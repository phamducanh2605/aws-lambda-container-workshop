#!/bin/bash

FUNCTION_NAME=${1:-workshop-product-api}
CONCURRENCY=${2:-10}

echo "Configuring provisioned concurrency for ${FUNCTION_NAME}..."

# Create or update provisioned concurrency
aws lambda put-provisioned-concurrency-config \
    --function-name ${FUNCTION_NAME} \
    --provisioned-concurrency-executions ${CONCURRENCY} \
    --qualifier '$LATEST'

echo "Provisioned concurrency configured: ${CONCURRENCY} instances"

# Monitor provisioned concurrency status
aws lambda get-provisioned-concurrency-config \
    --function-name ${FUNCTION_NAME} \
    --qualifier '$LATEST'

# src/warmup/intelligent_warmup.py
import boto3
import json
import time
from datetime import datetime, timedelta
from typing import List, Dict

class IntelligentWarmup:
    """Intelligent warm-up system based on traffic patterns"""
    
    def __init__(self):
        self.lambda_client = boto3.client('lambda')
        self.cloudwatch = boto3.client('cloudwatch')
        self.dynamodb = boto3.resource('dynamodb')
        self.warmup_table = self.dynamodb.Table('lambda-warmup-patterns')
    
    def analyze_traffic_patterns(self, function_name: str, days: int = 7) -> Dict:
        """Analyze historical traffic patterns"""
        end_time = datetime.utcnow()
        start_time = end_time - timedelta(days=days)
        
        # Get invocation metrics
        response = self.cloudwatch.get_metric_statistics(
            Namespace='AWS/Lambda',
            MetricName='Invocations',
            Dimensions=[{'Name': 'FunctionName', 'Value': function_name}],
            StartTime=start_time,
            EndTime=end_time,
            Period=3600,  # 1 hour periods
            Statistics=['Sum']
        )
        
        # Analyze patterns by hour of day
        hourly_patterns = {}
        for datapoint in response['Datapoints']:
            hour = datapoint['Timestamp'].hour
            invocations = datapoint['Sum']
            
            if hour not in hourly_patterns:
                hourly_patterns[hour] = []
            hourly_patterns[hour].append(invocations)
        
        # Calculate average invocations per hour
        hourly_averages = {}
        for hour, invocations in hourly_patterns.items():
            hourly_averages[hour] = sum(invocations) / len(invocations)
        
        return {
            'function_name': function_name,
            'analysis_period_days': days,
            'hourly_averages': hourly_averages,
            'peak_hour': max(hourly_averages, key=hourly_averages.get),
            'peak_invocations': max(hourly_averages.values()),
            'low_hour': min(hourly_averages, key=hourly_averages.get),
            'low_invocations': min(hourly_averages.values())
        }
    
    def calculate_optimal_concurrency(self, traffic_pattern: Dict) -> Dict:
        """Calculate optimal provisioned concurrency based on traffic"""
        hourly_averages = traffic_pattern['hourly_averages']
        
        # Calculate concurrency recommendations
        concurrency_schedule = {}
        for hour, avg_invocations in hourly_averages.items():
            # Estimate concurrent executions (assuming 2-second average duration)
            estimated_concurrency = max(1, int(avg_invocations * 2 / 3600))
            # Add 20% buffer for spikes
            recommended_concurrency = int(estimated_concurrency * 1.2)
            concurrency_schedule[hour] = min(recommended_concurrency, 100)  # Cap at 100
        
        return {
            'function_name': traffic_pattern['function_name'],
            'hourly_concurrency': concurrency_schedule,
            'max_concurrency': max(concurrency_schedule.values()),
            'min_concurrency': min(concurrency_schedule.values()),
            'avg_concurrency': sum(concurrency_schedule.values()) / len(concurrency_schedule)
        }
    
    def schedule_warmup(self, function_name: str, concurrency_schedule: Dict):
        """Schedule warm-up based on traffic patterns"""
        current_hour = datetime.utcnow().hour
        next_hour = (current_hour + 1) % 24
        
        # Get recommended concurrency for next hour
        next_concurrency = concurrency_schedule['hourly_concurrency'].get(next_hour, 5)
        
        # Update provisioned concurrency
        try:
            self.lambda_client.put_provisioned_concurrency_config(
                FunctionName=function_name,
                ProvisionedConcurrencyExecutions=next_concurrency
            )
            
            print(f"Updated provisioned concurrency for {function_name} to {next_concurrency}")
            
            # Store warmup event
            self.warmup_table.put_item(
                Item={
                    'function_name': function_name,
                    'timestamp': datetime.utcnow().isoformat(),
                    'hour': next_hour,
                    'concurrency': next_concurrency,
                    'action': 'scheduled_warmup'
                }
            )
            
        except Exception as e:
            print(f"Failed to update provisioned concurrency: {e}")
    
    def emergency_warmup(self, function_name: str, target_concurrency: int = 20):
        """Emergency warm-up for sudden traffic spikes"""
        try:
            # Get current provisioned concurrency
            current_config = self.lambda_client.get_provisioned_concurrency_config(
                FunctionName=function_name
            )
            current_concurrency = current_config['AllocatedConcurrencyExecutions']
            
            if target_concurrency > current_concurrency:
                self.lambda_client.put_provisioned_concurrency_config(
                    FunctionName=function_name,
                    ProvisionedConcurrencyExecutions=target_concurrency
                )
                
                print(f"Emergency warmup: increased concurrency from {current_concurrency} to {target_concurrency}")
                
        except self.lambda_client.exceptions.ResourceNotFoundException:
            # Create provisioned concurrency if it doesn't exist
            self.lambda_client.put_provisioned_concurrency_config(
                FunctionName=function_name,
                ProvisionedConcurrencyExecutions=target_concurrency
            )
            print(f"Emergency warmup: created provisioned concurrency with {target_concurrency}")

# Warmup Lambda handler
def warmup_handler(event, context):
    """Lambda function to handle intelligent warm-up"""
    warmup_system = IntelligentWarmup()
    
    # Get function name from event or environment
    function_name = event.get('function_name', os.environ.get('TARGET_FUNCTION'))
    
    if not function_name:
        return {'statusCode': 400, 'body': 'Function name not provided'}
    
    try:
        # Analyze traffic patterns
        traffic_pattern = warmup_system.analyze_traffic_patterns(function_name)
        
        # Calculate optimal concurrency
        concurrency_schedule = warmup_system.calculate_optimal_concurrency(traffic_pattern)
        
        # Schedule warmup
        warmup_system.schedule_warmup(function_name, concurrency_schedule)
        
        return {
            'statusCode': 200,
            'body': json.dumps({
                'message': 'Warmup scheduled successfully',
                'traffic_pattern': traffic_pattern,
                'concurrency_schedule': concurrency_schedule
            })
        }
        
    except Exception as e:
        return {
            'statusCode': 500,
            'body': json.dumps({'error': str(e)})
        }

# src/caching/connection_pool.py
import boto3
import pymysql
import redis
from pymysql.cursors import DictCursor
import os
from typing import Optional

class ConnectionPool:
    """Singleton connection pool for database connections"""
    _instance = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._initialized = False
        return cls._instance
    
    def __init__(self):
        if self._initialized:
            return
        
        # Database connections
        self._db_connection = None
        self._redis_connection = None
        self._dynamodb_resource = None
        self._s3_client = None
        
        self._initialized = True
    
    def get_db_connection(self):
        """Get MySQL database connection"""
        if self._db_connection is None or not self._is_connection_alive(self._db_connection):
            self._db_connection = pymysql.connect(
                host=os.environ.get('DB_HOST'),
                user=os.environ.get('DB_USER'),
                password=os.environ.get('DB_PASSWORD'),
                database=os.environ.get('DB_NAME'),
                cursorclass=DictCursor,
                autocommit=True,
                charset='utf8mb4',
                connect_timeout=5,
                read_timeout=10,
                write_timeout=10
            )
        return self._db_connection
    
    def get_redis_connection(self):
        """Get Redis connection"""
        if self._redis_connection is None:
            self._redis_connection = redis.Redis(
                host=os.environ.get('REDIS_HOST'),
                port=int(os.environ.get('REDIS_PORT', 6379)),
                db=0,
                socket_connect_timeout=5,
                socket_timeout=5,
                retry_on_timeout=True,
                health_check_interval=30
            )
        return self._redis_connection
    
    def get_dynamodb_resource(self):
        """Get DynamoDB resource"""
        if self._dynamodb_resource is None:
            self._dynamodb_resource = boto3.resource(
                'dynamodb',
                region_name=os.environ.get('AWS_REGION', 'us-east-1')
            )
        return self._dynamodb_resource
    
    def get_s3_client(self):
        """Get S3 client"""
        if self._s3_client is None:
            self._s3_client = boto3.client(
                's3',
                region_name=os.environ.get('AWS_REGION', 'us-east-1')
            )
        return self._s3_client
    
    def _is_connection_alive(self, connection) -> bool:
        """Check if database connection is alive"""
        try:
            connection.ping(reconnect=False)
            return True
        except:
            return False
    
    def close_all_connections(self):
        """Close all connections"""
        if self._db_connection:
            self._db_connection.close()
            self._db_connection = None
        
        if self._redis_connection:
            self._redis_connection.close()
            self._redis_connection = None

# Global connection pool
connection_pool = ConnectionPool()
