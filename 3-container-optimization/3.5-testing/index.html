import pytest
import json
from unittest.mock import Mock, patch
from moto import mock_dynamodb
import boto3

# Add src to path
import sys
import os
sys.path.insert(0, os.path.join(os.path.dirname(__file__), '../../..'))

from src.api.product_service import ProductService

@mock_dynamodb
class TestProductService:
    
    def setup_method(self):
        """Setup test environment"""
        # Create mock DynamoDB table
        self.dynamodb = boto3.resource('dynamodb', region_name='us-east-1')
        self.table = self.dynamodb.create_table(
            TableName='workshop-products',
            KeySchema=[{'AttributeName': 'id', 'KeyType': 'HASH'}],
            AttributeDefinitions=[{'AttributeName': 'id', 'AttributeType': 'S'}],
            BillingMode='PAY_PER_REQUEST'
        )
        
        # Initialize service
        with patch.dict(os.environ, {'PRODUCTS_TABLE': 'workshop-products'}):
            self.service = ProductService()
    
    def test_list_products_empty(self):
        """Test listing products when table is empty"""
        result = self.service.list_products()
        
        assert 'products' in result
        assert result['count'] == 0
        assert isinstance(result['products'], list)
    
    def test_create_product(self):
        """Test creating a new product"""
        product_data = {
            'name': 'Test Product',
            'description': 'A test product',
            'price': 29.99,
            'category': 'test'
        }
        
        result = self.service.create_product(product_data)
        
        assert 'product' in result
        assert result['product']['name'] == 'Test Product'
        assert result['product']['price'] == 29.99
        assert 'id' in result['product']
    
    def test_get_product_not_found(self):
        """Test getting a product that doesn't exist"""
        result = self.service.get_product('nonexistent-id')
        
        assert 'error' in result
        assert result['error'] == 'Product not found'
    
    def test_create_and_get_product(self):
        """Test creating and then retrieving a product"""
        # Create product
        product_data = {
            'name': 'Test Product',
            'description': 'A test product',
            'price': 29.99,
            'category': 'test'
        }
        
        create_result = self.service.create_product(product_data)
        product_id = create_result['product']['id']
        
        # Get product
        get_result = self.service.get_product(product_id)
        
        assert 'product' in get_result
        assert get_result['product']['name'] == 'Test Product'
        assert get_result['product']['id'] == product_id
    
    def test_update_product(self):
        """Test updating an existing product"""
        # Create product first
        product_data = {
            'name': 'Original Product',
            'price': 19.99
        }
        
        create_result = self.service.create_product(product_data)
        product_id = create_result['product']['id']
        
        # Update product
        update_data = {
            'name': 'Updated Product',
            'price': 39.99
        }
        
        update_result = self.service.update_product(product_id, update_data)
        
        assert 'message' in update_result
        assert 'updated successfully' in update_result['message']
    
    def test_delete_product(self):
        """Test deleting a product"""
        # Create product first
        product_data = {'name': 'Product to Delete'}
        create_result = self.service.create_product(product_data)
        product_id = create_result['product']['id']
        
        # Delete product
        delete_result = self.service.delete_product(product_id)
        
        assert 'message' in delete_result
        assert 'deleted successfully' in delete_result['message']
    
    def test_mock_products_fallback(self):
        """Test fallback to mock products when DynamoDB unavailable"""
        # Create service without DynamoDB connection
        service = ProductService()
        service.table = None
        
        result = service.list_products()
        
        assert 'products' in result
        assert result['count'] > 0
        assert 'note' in result
        assert 'mock data' in result['note']

sys.path.insert(0, os.path.join(os.path.dirname(__file__), '../..'))

from lambda_function import lambda_handler, create_response

class TestLambdaHandler:
    
    def test_health_endpoint(self):
        """Test health endpoint"""
        event = {
            'httpMethod': 'GET',
            'path': '/health',
            'body': None,
            'pathParameters': None
        }
        context = Mock()
        
        response = lambda_handler(event, context)
        
        assert response['statusCode'] == 200
        body = json.loads(response['body'])
        assert body['status'] == 'healthy'
    
    @patch('lambda_function.product_service')
    def test_list_products(self, mock_service):
        """Test list products endpoint"""
        # Mock service response
        mock_service.list_products.return_value = {
            'products': [{'id': '1', 'name': 'Test Product'}],
            'count': 1
        }
        
        event = {
            'httpMethod': 'GET',
            'path': '/products',
            'body': None,
            'pathParameters': None
        }
        context = Mock()
        
        response = lambda_handler(event, context)
        
        assert response['statusCode'] == 200
        body = json.loads(response['body'])
        assert body['count'] == 1
        mock_service.list_products.assert_called_once()
    
    @patch('lambda_function.product_service')
    def test_create_product(self, mock_service):
        """Test create product endpoint"""
        # Mock service response
        mock_service.create_product.return_value = {
            'product': {'id': '123', 'name': 'New Product'},
            'message': 'Product created successfully'
        }
        
        product_data = {
            'name': 'New Product',
            'price': 29.99
        }
        
        event = {
            'httpMethod': 'POST',
            'path': '/products',
            'body': json.dumps(product_data),
            'pathParameters': None
        }
        context = Mock()
        
        response = lambda_handler(event, context)
        
        assert response['statusCode'] == 200
        body = json.loads(response['body'])
        assert body['product']['name'] == 'New Product'
        mock_service.create_product.assert_called_once_with(product_data)
    
    def test_not_found_endpoint(self):
        """Test non-existent endpoint"""
        event = {
            'httpMethod': 'GET',
            'path': '/nonexistent',
            'body': None,
            'pathParameters': None
        }
        context = Mock()
        
        response = lambda_handler(event, context)
        
        assert response['statusCode'] == 404
        body = json.loads(response['body'])
        assert body['error'] == 'Not Found'
    
    @patch('lambda_function.product_service')
    def test_exception_handling(self, mock_service):
        """Test exception handling"""
        # Mock service to raise exception
        mock_service.list_products.side_effect = Exception("Database error")
        
        event = {
            'httpMethod': 'GET',
            'path': '/products',
            'body': None,
            'pathParameters': None
        }
        context = Mock()
        
        response = lambda_handler(event, context)
        
        assert response['statusCode'] == 500
        body = json.loads(response['body'])
        assert body['error'] == 'Internal Server Error'
    
    def test_create_response(self):
        """Test response creation utility"""
        test_data = {'message': 'test'}
        response = create_response(200, test_data)
        
        assert response['statusCode'] == 200
        assert 'Content-Type' in response['headers']
        assert response['headers']['Content-Type'] == 'application/json'
        assert 'Access-Control-Allow-Origin' in response['headers']
        
        body = json.loads(response['body'])
        assert body['message'] == 'test'

import pytest
import requests
import json
import time
import subprocess
import docker

class TestContainerIntegration:
    
    def setup_method(self):
        """Setup container for testing"""
        self.docker_client = docker.from_env()
        self.container = None
        self.base_url = "http://localhost:9000/2015-03-31/functions/function/invocations"
    
    def teardown_method(self):
        """Cleanup container"""
        if self.container:
            self.container.stop()
            self.container.remove()
    
    def start_container(self, image_name):
        """Start Lambda container for testing"""
        try:
            self.container = self.docker_client.containers.run(
                image_name,
                ports={'8080/tcp': 9000},
                detach=True,
                remove=False
            )
            
            # Wait for container to be ready
            time.sleep(3)
            
            # Health check
            for _ in range(10):
                try:
                    response = requests.get("http://localhost:9000/", timeout=1)
                    if response.status_code == 200:
                        return True
                except:
                    time.sleep(1)
            
            return False
        except Exception as e:
            print(f"Failed to start container: {e}")
            return False
    
    @pytest.mark.parametrize("image_name", [
        "lambda-workshop:base",
        "lambda-workshop:optimized", 
        "lambda-workshop:ultra"
    ])
    def test_container_health_endpoint(self, image_name):
        """Test health endpoint across all container variants"""
        if not self.start_container(image_name):
            pytest.skip(f"Could not start container {image_name}")
        
        event = {
            "httpMethod": "GET",
            "path": "/health",
            "body": None,
            "pathParameters": None
        }
        
        response = requests.post(
            self.base_url,
            json=event,
            timeout=10
        )
        
        assert response.status_code == 200
        result = response.json()
        assert result['statusCode'] == 200
        
        body = json.loads(result['body'])
        assert body['status'] == 'healthy'
    
    def test_container_performance_comparison(self):
        """Compare performance across container variants"""
        results = {}
        
        for image_name in ["lambda-workshop:base", "lambda-workshop:optimized", "lambda-workshop:ultra"]:
            if not self.start_container(image_name):
                continue
            
            # Warm up
            event = {"httpMethod": "GET", "path": "/health"}
            requests.post(self.base_url, json=event, timeout=5)
            
            # Measure performance
            times = []
            for _ in range(5):
                start_time = time.time()
                response = requests.post(self.base_url, json=event, timeout=10)
                end_time = time.time()
                
                if response.status_code == 200:
                    times.append((end_time - start_time) * 1000)  # Convert to ms
            
            if times:
                results[image_name] = {
                    'avg_time': sum(times) / len(times),
                    'min_time': min(times),
                    'max_time': max(times)
                }
            
            # Cleanup for next iteration
            self.teardown_method()
        
        # Print results
        print("\nContainer Performance Comparison:")
        for image, metrics in results.items():
            print(f"{image}: avg={metrics['avg_time']:.2f}ms, min={metrics['min_time']:.2f}ms, max={metrics['max_time']:.2f}ms")
        
        # Assert that optimized versions are faster (if all containers tested)
        if len(results) == 3:
            base_time = results.get("lambda-workshop:base", {}).get('avg_time', float('inf'))
            optimized_time = results.get("lambda-workshop:optimized", {}).get('avg_time', float('inf'))
            ultra_time = results.get("lambda-workshop:ultra", {}).get('avg_time', float('inf'))
            
            # Ultra should be fastest, optimized should be faster than base
            assert ultra_time <= optimized_time <= base_time, "Performance optimization not working as expected"
EOF

# Create performance benchmark
cat > tests/performance/test_performance_benchmark.py << 'EOF'
import pytest
import time
import statistics
from unittest.mock import patch
import sys
import os

# Add src to path
sys.path.insert(0, os.path.join(os.path.dirname(__file__), '../..'))

from lambda_function import lambda_handler

class TestPerformanceBenchmark:
    
    def benchmark_lambda_handler(self, event, iterations=100):
        """Benchmark Lambda handler performance"""
        times = []
        
        for _ in range(iterations):
            start_time = time.perf_counter()
            response = lambda_handler(event, {})
            end_time = time.perf_counter()
            
            execution_time = (end_time - start_time) * 1000  # Convert to milliseconds
            times.append(execution_time)
        
        return {
            'mean': statistics.mean(times),
            'median': statistics.median(times),
            'stdev': statistics.stdev(times) if len(times) > 1 else 0,
            'min': min(times),
            'max': max(times),
            'p95': sorted(times)[int(0.95 * len(times))],
            'p99': sorted(times)[int(0.99 * len(times))]
        }
    
    def test_health_endpoint_performance(self):
        """Benchmark health endpoint performance"""
        event = {
            'httpMethod': 'GET',
            'path': '/health',
            'body': None,
            'pathParameters': None
        }
        
        results = self.benchmark_lambda_handler(event, iterations=100)
        
        print(f"\nHealth Endpoint Performance:")
        print(f"Mean: {results['mean']:.2f}ms")
        print(f"Median: {results['median']:.2f}ms")
        print(f"P95: {results['p95']:.2f}ms")
        print(f"P99: {results['p99']:.2f}ms")
        
        # Assert reasonable performance (adjust thresholds as needed)
        assert results['mean'] < 100, f"Health endpoint too slow: {results['mean']:.2f}ms"
        assert results['p95'] < 200, f"P95 too high: {results['p95']:.2f}ms"
    
    @patch('lambda_function.product_service')
    def test_list_products_performance(self, mock_service):
        """Benchmark list products performance"""
        # Mock service response
        mock_service.list_products.return_value = {
            'products': [{'id': str(i), 'name': f'Product {i}'} for i in range(100)],
            'count': 100
        }
        
        event = {
            'httpMethod': 'GET',

from locust import HttpUser, task, between
import json
import random

        response = self.client.get(f"{self.api_base}/health")
        if response.status_code == 200:
            data = response.json()
            assert data.get('status') == 'healthy'
    
    @task(2)
    def test_list_products(self):
        """Test list products endpoint"""
        response = self.client.get(f"{self.api_base}/products")
        if response.status_code == 200:
            data = response.json()
            if 'products' in data:
                self.products = data['products']
    
    @task(1)
    def test_create_product(self):
        """Test create product endpoint"""
        product_data = {
            'name': f'Load Test Product {random.randint(1, 1000)}',
            'description': 'Product created during load testing',
            'price': round(random.uniform(10, 100), 2),
            'category': random.choice(['electronics', 'books', 'clothing'])
        }
        
        response = self.client.post(
            f"{self.api_base}/products",

            if 'product' in data:
                self.products.append(data['product'])

    def test_get_product(self):

            product = random.choice(self.products)
            product_id = product.get('id')
            if product_id:
                response = self.client.get(f"{self.api_base}/products/{product_id}")

        response = self.client.get(f"{self.api_base}/invalid-endpoint")
        assert response.status_code == 404
        
        # Test invalid product ID
        response = self.client.get(f"{self.api_base}/products/invalid-id")

#!/bin/bash

echo "=== Lambda Container Load Testing ==="

    echo "Error: API_URL environment variable not set"
    echo "Please set API_URL to your API Gateway endpoint"
    exit 1

echo "Testing API: $API_URL"

# Function to run load test
run_load_test() {
    local test_name=$1
    local users=$2
    local spawn_rate=$3
    local duration=$4
    local user_class=$5
    
    echo "Running $test_name..."
    echo "Users: $users, Spawn Rate: $spawn_rate/sec, Duration: ${duration}s"

        --host="$API_URL" \
        --users="$users" \
        --spawn-rate="$spawn_rate" \
        --run-time="${duration}s" \
        --user-class="$user_class" \
        --headless \
        --csv="results/${test_name}" \
        --html="results/${test_name}.html"
    
    echo "$test_name completed"
    echo ""
}

# Create results directory

# Light load test
run_load_test "light_load" 10 2 60 "LambdaAPIUser"

# Medium load test  
run_load_test "medium_load" 50 5 120 "LambdaAPIUser"

# High load test
run_load_test "high_load" 100 10 180 "LambdaAPIUser"

# Cold start test
run_load_test "cold_start" 5 1 300 "ColdStartUser"

# Spike test
run_load_test "spike_test" 200 50 60 "HighLoadUser"

echo "=== Load testing completed ==="
echo "Results saved in results/ directory"

# Generate summary report
echo "=== Load Test Summary ==="
for file in results/*_stats.csv; do
    if [ -f "$file" ]; then
        test_name=$(basename "$file" _stats.csv)
        echo "Test: $test_name"
        echo "Results: $(tail -n 1 "$file")"

    fi
done

cat > scripts/run-all-tests.sh << 'EOF'
#!/bin/bash

echo "=== Lambda Container Testing Suite ==="

# Set test environment
export PYTHONPATH="${PYTHONPATH}:$(pwd)"
export AWS_DEFAULT_REGION=us-east-1

# Create test results directory
mkdir -p test-results

# Function to run test suite
run_test_suite() {
    local test_type=$1
    local test_path=$2
    local output_file="test-results/${test_type}-results.xml"
    
    echo "Running $test_type tests..."
    
    pytest "$test_path" \
        --verbose \
        --tb=short \
        --junit-xml="$output_file" \
        --cov=src \
        --cov-report=html:test-results/${test_type}-coverage \
        --cov-report=term-missing
    
    local exit_code=$?
    
    if [ $exit_code -eq 0 ]; then
        echo "✅ $test_type tests PASSED"
    else
        echo "❌ $test_type tests FAILED"
    fi
    
    echo ""
    return $exit_code
}

# Run all test suites
echo "Starting test execution..."

# Unit tests
run_test_suite "unit" "tests/unit/"
unit_result=$?

# Integration tests  
run_test_suite "integration" "tests/integration/"
integration_result=$?

# Performance tests
run_test_suite "performance" "tests/performance/"
performance_result=$?

# Generate test summary
echo "=== Test Summary ==="
echo "Unit Tests: $([ $unit_result -eq 0 ] && echo "✅ PASSED" || echo "❌ FAILED")"
echo "Integration Tests: $([ $integration_result -eq 0 ] && echo "✅ PASSED" || echo "❌ FAILED")"
echo "Performance Tests: $([ $performance_result -eq 0 ] && echo "✅ PASSED" || echo "❌ FAILED")"

# Overall result
overall_result=$((unit_result + integration_result + performance_result))

if [ $overall_result -eq 0 ]; then
    echo ""
    echo "🎉 All tests PASSED!"
    exit 0
else
    echo ""
    echo "💥 Some tests FAILED!"
    exit 1
fi
EOF

Testing Best Practices Summary

Unit Testing

  • ✅ Test individual components in isolation
  • ✅ Mock external dependencies (DynamoDB, AWS services)
  • ✅ Achieve high code coverage (>80%)
  • ✅ Test error conditions and edge cases

Integration Testing

  • ✅ Test complete workflows end-to-end
  • ✅ Test container functionality
  • ✅ Verify AWS service integrations
  • ✅ Test API Gateway integration

Performance Testing

  • ✅ Benchmark execution times
  • ✅ Monitor memory usage
  • ✅ Compare container variants
  • ✅ Test cold start performance

Load Testing

  • ✅ Test under realistic load conditions
  • ✅ Identify performance bottlenecks
  • ✅ Test error handling under load
  • ✅ Validate auto-scaling behavior

Security Testing

  • ✅ Scan containers for vulnerabilities
  • ✅ Test IAM permissions
  • ✅ Validate input sanitization
  • ✅ Test authentication/authorization

Run tests frequently during development and always before deployment. Automated testing in CI/CD pipelines helps catch issues early.
