# Create secure Dockerfile
cat > docker/secure/Dockerfile << 'EOF'
# Use official AWS Lambda base image
FROM public.ecr.aws/lambda/python:3.11

# Set security labels
LABEL maintainer="your-team@company.com"
LABEL version="1.0"
LABEL description="Secure Lambda container for Product API"

# Create non-root user (Lambda runtime will still run as root)
RUN groupadd -r appgroup && useradd -r -g appgroup appuser

# Install security updates
RUN yum update -y && \
    yum install -y yum-plugin-security && \
    yum --security update -y && \
    yum clean all && \
    rm -rf /var/cache/yum

# Copy requirements and install with security checks
COPY requirements.txt ${LAMBDA_TASK_ROOT}
RUN pip install --no-cache-dir --upgrade pip && \
    pip install --no-cache-dir --require-hashes -r requirements.txt || \
    pip install --no-cache-dir -r requirements.txt

# Copy application code
COPY lambda_function.py ${LAMBDA_TASK_ROOT}
COPY src/ ${LAMBDA_TASK_ROOT}/src/

# Set secure file permissions
RUN find ${LAMBDA_TASK_ROOT} -type f -exec chmod 644 {} \; && \
    find ${LAMBDA_TASK_ROOT} -type d -exec chmod 755 {} \;

# Remove unnecessary packages and files
RUN yum remove -y gcc python3-devel && \
    yum autoremove -y && \
    find ${LAMBDA_TASK_ROOT} -name "*.pyc" -delete && \
    find ${LAMBDA_TASK_ROOT} -type d -name "__pycache__" -exec rm -rf {} + || true

# Set handler
CMD [ "lambda_function.lambda_handler" ]
EOF

# Create security scan script
cat > scripts/security-scan.sh << 'EOF'
#!/bin/bash

echo "=== Container Security Scanning ==="

# Function to scan image
scan_image() {
    local image_name=$1
    echo "Scanning $image_name for vulnerabilities..."
    
    # Scan for vulnerabilities
    trivy image --severity HIGH,CRITICAL $image_name
    
    # Generate detailed report
    trivy image --format json --output ${image_name//[:\\/]/_}-scan.json $image_name
    
    echo "Scan completed for $image_name"
    echo ""
}

# Scan all images
scan_image "lambda-workshop:base"
scan_image "lambda-workshop:optimized"
scan_image "lambda-workshop:ultra"

# Compare vulnerability counts
echo "=== Vulnerability Summary ==="
for image in base optimized ultra; do
    if [ -f "lambda-workshop_${image}-scan.json" ]; then
        HIGH_COUNT=$(jq '[.Results[]?.Vulnerabilities[]? | select(.Severity=="HIGH")] | length' lambda-workshop_${image}-scan.json)
        CRITICAL_COUNT=$(jq '[.Results[]?.Vulnerabilities[]? | select(.Severity=="CRITICAL")] | length' lambda-workshop_${image}-scan.json)
        echo "$image: $CRITICAL_COUNT critical, $HIGH_COUNT high vulnerabilities"
    fi
done

echo "=== Security scanning completed ==="
EOF

chmod +x scripts/security-scan.sh

# Create secure deployment script
cat > scripts/deploy-secure-lambda.sh << 'EOF'
#!/bin/bash

set -e

echo "=== Deploying Secure Lambda Function ==="

# Build secure image
docker build -t lambda-workshop:secure -f docker/secure/Dockerfile .

# Tag for ECR
docker tag lambda-workshop:secure $ECR_REGISTRY/lambda-workshop/product-api:secure

# Push to ECR
docker push $ECR_REGISTRY/lambda-workshop/product-api:secure

# Deploy secure Lambda function
aws lambda create-function \
    --function-name workshop-product-api-secure \
    --role $SECURE_ROLE_ARN \
    --code ImageUri=$ECR_REGISTRY/lambda-workshop/product-api:secure \
    --package-type Image \
    --memory-size 512 \
    --timeout 30 \
    --environment Variables='{
        "PRODUCTS_TABLE":"workshop-products",
        "AWS_DEFAULT_REGION":"us-east-1",
        "LOG_LEVEL":"INFO"
    }' \
    --description "Secure Lambda function with minimal permissions" \
    --dead-letter-config TargetArn=arn:aws:sqs:us-east-1:$(aws sts get-caller-identity --query Account --output text):lambda-dlq \
    --tracing-config Mode=Active \
    --reserved-concurrent-executions 100 \
    2>/dev/null || \
    aws lambda update-function-code \
        --function-name workshop-product-api-secure \
        --image-uri $ECR_REGISTRY/lambda-workshop/product-api:secure

echo "✅ Secure Lambda function deployed"
EOF

chmod +x scripts/deploy-secure-lambda.sh

# Create security configuration script
cat > scripts/configure-security.sh << 'EOF'
#!/bin/bash

echo "=== Configuring Additional Security Features ==="

# Enable X-Ray tracing
aws lambda put-function-configuration \
    --function-name workshop-product-api-secure \
    --tracing-config Mode=Active

# Set reserved concurrency
aws lambda put-reserved-concurrency \
    --function-name workshop-product-api-secure \
    --reserved-concurrent-executions 100

# Create DLQ (Dead Letter Queue)
aws sqs create-queue \
    --queue-name lambda-dlq \
    --attributes '{
        "MessageRetentionPeriod": "1209600",
        "VisibilityTimeoutSeconds": "60"
    }' 2>/dev/null || echo "DLQ already exists"

# Get DLQ ARN
DLQ_ARN=$(aws sqs get-queue-attributes \
    --queue-url https://sqs.us-east-1.amazonaws.com/$(aws sts get-caller-identity --query Account --output text)/lambda-dlq \
    --attribute-names QueueArn \
    --query 'Attributes.QueueArn' \
    --output text)

# Configure DLQ for Lambda
aws lambda update-function-configuration \
    --function-name workshop-product-api-secure \
    --dead-letter-config TargetArn=$DLQ_ARN

echo "✅ Security features configured"
EOF

chmod +x scripts/configure-security.sh

# Create secure Lambda handler with secrets
cat > lambda_function_secure.py << 'EOF'
import json
import os
import boto3
import logging
from botocore.exceptions import ClientError

# Configure logging
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

# Initialize AWS clients
secrets_client = boto3.client('secretsmanager')

def get_secret(secret_name):
    """Retrieve secret from AWS Secrets Manager"""
    try:
        response = secrets_client.get_secret_value(SecretId=secret_name)
        return json.loads(response['SecretString'])
    except ClientError as e:
        logger.error(f"Error retrieving secret {secret_name}: {e}")
        return None

def lambda_handler(event, context):
    """Secure Lambda handler with secrets management"""
    
    # Log request (without sensitive data)
    logger.info(f"Processing request: {event.get('httpMethod')} {event.get('path')}")
    
    try:
        # Retrieve secrets if needed
        if event.get('path') == '/secure-endpoint':
            db_credentials = get_secret('workshop/database/credentials')
            if not db_credentials:
                return {
                    'statusCode': 500,
                    'body': json.dumps({'error': 'Unable to retrieve database credentials'})
                }
        
        # Process request based on path
        http_method = event.get('httpMethod', 'GET')
        path = event.get('path', '/')
        
        if path == '/health':
            response = {
                'status': 'healthy',
                'version': '1.0.0',
                'security_features': [
                    'secrets_manager',
                    'iam_least_privilege',
                    'x_ray_tracing',
                    'dead_letter_queue'
                ]
            }
        elif path == '/secure-info':
            # Example of secure data handling
            response = {
                'message': 'This is secure information',
                'user_id': event.get('requestContext', {}).get('identity', {}).get('userArn', 'anonymous'),
                'timestamp': context.aws_request_id
            }
        else:
            response = {'error': 'Endpoint not found'}
            return create_secure_response(404, response)
        
        return create_secure_response(200, response)
        
    except Exception as e:
        logger.error(f"Error processing request: {str(e)}")
        return create_secure_response(500, {
            'error': 'Internal server error',
            'request_id': context.aws_request_id
        })

def create_secure_response(status_code, body):
    """Create secure HTTP response with security headers"""
    return {
        'statusCode': status_code,
        'headers': {
            'Content-Type': 'application/json',
            'X-Content-Type-Options': 'nosniff',
            'X-Frame-Options': 'DENY',
            'X-XSS-Protection': '1; mode=block',
            'Strict-Transport-Security': 'max-age=31536000; includeSubDomains',
            'Content-Security-Policy': "default-src 'self'",
            'Referrer-Policy': 'strict-origin-when-cross-origin'
        },
        'body': json.dumps(body, default=str)
    }
EOF

#!/bin/bash

echo "=== Security Monitoring Dashboard ==="

# Check IAM role permissions
echo "IAM Role Analysis:"
aws iam list-attached-role-policies \
    --role-name lambda-workshop-secure-role \
    --output table

echo ""

# Check Lambda function security configuration
echo "Lambda Security Configuration:"
aws lambda get-function-configuration \
    --function-name workshop-product-api-secure \
    --query '{
        Role: Role,
        TracingConfig: TracingConfig,
        DeadLetterConfig: DeadLetterConfig,
        ReservedConcurrency: ReservedConcurrencyExecutions,
        Environment: Environment.Variables
    }' \
    --output table

echo ""

# Check recent CloudTrail events
echo "Recent Security Events (CloudTrail):"
aws logs filter-log-events \
    --log-group-name CloudTrail/SecurityEvents \
    --start-time $(date -d '1 hour ago' +%s)000 \
    --filter-pattern "{ $.eventName = CreateFunction || $.eventName = UpdateFunctionCode }" \
    --query 'events[*].[eventTime,eventName,sourceIPAddress]' \
    --output table 2>/dev/null || echo "CloudTrail logs not available"

echo ""

# Check X-Ray traces
echo "X-Ray Tracing Status:"
aws xray get-trace-summaries \
    --time-range-type TimeRangeByStartTime \
    --start-time $(date -d '1 hour ago' +%s) \
    --end-time $(date +%s) \
    --query 'TraceSummaries[*].[Id,Duration,ResponseTime,HasError]' \
    --output table 2>/dev/null || echo "No X-Ray traces found"

echo "=== Security monitoring completed ==="
EOF

chmod +x scripts/security-monitoring.sh

# Create security audit script
cat > scripts/security-audit.sh << 'EOF'
#!/bin/bash

echo "=== Security Audit Report ==="

# Function security checklist
check_security_feature() {
    local feature=$1
    local check_command=$2
    
    echo -n "Checking $feature: "
    if eval $check_command &>/dev/null; then
        echo "✅ PASS"
    else
        echo "❌ FAIL"
    fi
}

echo "Lambda Function Security Checklist:"
check_security_feature "IAM Least Privilege" "aws iam get-role --role-name lambda-workshop-secure-role"
check_security_feature "X-Ray Tracing Enabled" "aws lambda get-function-configuration --function-name workshop-product-api-secure --query 'TracingConfig.Mode' --output text | grep -q Active"
check_security_feature "Dead Letter Queue" "aws lambda get-function-configuration --function-name workshop-product-api-secure --query 'DeadLetterConfig.TargetArn' --output text | grep -q sqs"
check_security_feature "Reserved Concurrency" "aws lambda get-function-configuration --function-name workshop-product-api-secure --query 'ReservedConcurrencyExecutions' --output text | grep -q '[0-9]'"
check_security_feature "Environment Variables Encrypted" "aws lambda get-function-configuration --function-name workshop-product-api-secure --query 'KMSKeyArn' --output text"

echo ""
echo "Container Security Checklist:"
check_security_feature "Base Image from AWS" "docker inspect lambda-workshop:secure --format '{{.Config.Image}}' | grep -q 'public.ecr.aws/lambda'"
check_security_feature "No Root User" "docker inspect lambda-workshop:secure --format '{{.Config.User}}' | grep -v '^root$' || echo 'Lambda runs as root by design'"
check_security_feature "Minimal Packages" "docker history lambda-workshop:secure | wc -l | awk '{print \$1 < 20}'"

echo ""
echo "Network Security:"
check_security_feature "VPC Configuration" "aws lambda get-function-configuration --function-name workshop-product-api-secure --query 'VpcConfig.VpcId' --output text"
check_security_feature "Security Groups" "aws lambda get-function-configuration --function-name workshop-product-api-secure --query 'VpcConfig.SecurityGroupIds' --output text"

echo ""
echo "=== Security audit completed ==="
EOF

chmod +x scripts/security-audit.sh

Security Best Practices Summary

Container Security

  • ✅ Use official AWS Lambda base images
  • ✅ Keep base images updated
  • ✅ Scan for vulnerabilities regularly
  • ✅ Remove unnecessary packages and files
  • ✅ Use multi-stage builds to minimize attack surface

IAM Security

  • ✅ Apply least privilege principle
  • ✅ Use specific resource ARNs
  • ✅ Avoid wildcard permissions
  • ✅ Regular permission audits

Runtime Security

  • ✅ Enable X-Ray tracing for monitoring
  • ✅ Configure dead letter queues
  • ✅ Set appropriate timeout and memory limits
  • ✅ Use reserved concurrency to prevent abuse

Data Security

  • ✅ Use AWS Secrets Manager for sensitive data
  • ✅ Encrypt environment variables
  • ✅ Implement proper error handling
  • ✅ Add security headers to responses

Monitoring & Compliance

  • ✅ Enable CloudTrail logging
  • ✅ Monitor function invocations
  • ✅ Set up security alerts
  • ✅ Regular security audits

Security is an ongoing process. Regularly update your images, review permissions, and monitor for new vulnerabilities.
