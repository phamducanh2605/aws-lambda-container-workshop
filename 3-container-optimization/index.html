# Remove unnecessary files
RUN find /opt/python -name "*.pyc" -delete && \
    find /opt/python -name "__pycache__" -type d -exec rm -rf {} + && \
    find /opt/python -name "*.so" -exec strip {} \;

# Production stage
FROM public.ecr.aws/lambda/python:3.11

# Copy optimized dependencies
COPY --from=builder /opt/python ${LAMBDA_TASK_ROOT}

COPY src/ ${LAMBDA_TASK_ROOT}/src/
COPY lambda_function.py ${LAMBDA_TASK_ROOT}/

ENV PYTHONUNBUFFERED=1
ENV PYTHONHASHSEED=0

RUN python -m compileall -b ${LAMBDA_TASK_ROOT} && \
    find ${LAMBDA_TASK_ROOT} -name "*.py" -not -path "*/tests/*" -delete

CMD ["lambda_function.lambda_handler"]

# scripts/analyze_dependencies.py
import pkg_resources
import importlib
import sys
import os

def analyze_imports(directory):
    """Analyze actual imports in Python files"""
    imports = set()
    
    for root, dirs, files in os.walk(directory):
        for file in files:
            if file.endswith('.py'):
                filepath = os.path.join(root, file)
                with open(filepath, 'r') as f:
                    content = f.read()
                    # Extract import statements
                    lines = content.split('\n')
                    for line in lines:
                        line = line.strip()
                        if line.startswith('import ') or line.startswith('from '):
                            imports.add(line.split()[1].split('.')[0])
    
    return imports

def get_installed_packages():
    """Get list of installed packages"""
    return {pkg.project_name for pkg in pkg_resources.working_set}

def find_unused_packages():
    """Find packages that are installed but not used"""
    used_imports = analyze_imports('./src')
    installed_packages = get_installed_packages()
    
    # Map common import names to package names
    import_to_package = {
        'boto3': 'boto3',
        'botocore': 'botocore',
        'requests': 'requests',
        'numpy': 'numpy',
        'pandas': 'pandas',
        'PIL': 'Pillow',
        'cv2': 'opencv-python'
    }
    
    used_packages = set()
    for imp in used_imports:
        if imp in import_to_package:
            used_packages.add(import_to_package[imp])
        else:
            used_packages.add(imp)
    
    unused = installed_packages - used_packages
    return unused

if __name__ == "__main__":
    unused = find_unused_packages()
    print("Unused packages:")
    for pkg in sorted(unused):
        print(f"  - {pkg}")

# src/optimizations/connection_pool.py
import boto3
import requests
from functools import lru_cache
import os

class OptimizedConnections:
    """Singleton class for connection pooling"""
    _instance = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._initialized = False
        return cls._instance
    
    def __init__(self):
        if self._initialized:
            return
            
        # Initialize connections
        self.dynamodb = boto3.resource('dynamodb', region_name=os.environ.get('AWS_REGION', 'us-east-1'))
        self.s3_client = boto3.client('s3', region_name=os.environ.get('AWS_REGION', 'us-east-1'))
        
        # HTTP session with connection pooling
        self.http_session = requests.Session()
        adapter = requests.adapters.HTTPAdapter(
            pool_connections=10,
            pool_maxsize=20,
            max_retries=3
        )
        self.http_session.mount('http://', adapter)
        self.http_session.mount('https://', adapter)
        
        self._initialized = True

# Global instance
connections = OptimizedConnections()

@lru_cache(maxsize=128)
def get_table(table_name):
    """Cached table reference"""
    return connections.dynamodb.Table(table_name)

# src/optimizations/lazy_imports.py
import importlib
from functools import wraps

def lazy_import(module_name):
    """Decorator for lazy importing modules"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # Import module only when function is called
            module = importlib.import_module(module_name)
            # Add module to function's globals
            func.__globals__[module_name.split('.')[-1]] = module
            return func(*args, **kwargs)
        return wrapper
    return decorator

# Usage example
@lazy_import('pandas')
@lazy_import('numpy')
def process_data(data):
    """Function that uses pandas and numpy"""
    df = pandas.DataFrame(data)
    return numpy.array(df.values)

# src/optimizations/memory_profiler.py
import psutil
import os
import json
from datetime import datetime

class MemoryProfiler:
    def __init__(self):
        self.process = psutil.Process(os.getpid())
        self.initial_memory = self.process.memory_info().rss
        self.checkpoints = []
    
    def checkpoint(self, name):
        """Record memory usage at checkpoint"""
        current_memory = self.process.memory_info().rss
        memory_delta = current_memory - self.initial_memory
        
        checkpoint = {
            'name': name,
            'timestamp': datetime.utcnow().isoformat(),
            'memory_mb': current_memory / 1024 / 1024,
            'delta_mb': memory_delta / 1024 / 1024
        }
        
        self.checkpoints.append(checkpoint)
        print(f"Memory checkpoint '{name}': {checkpoint['memory_mb']:.2f}MB (+{checkpoint['delta_mb']:.2f}MB)")
    
    def get_report(self):
        """Generate memory usage report"""
        return {
            'initial_memory_mb': self.initial_memory / 1024 / 1024,
            'checkpoints': self.checkpoints,
            'peak_memory_mb': max(cp['memory_mb'] for cp in self.checkpoints) if self.checkpoints else 0
        }

# Global profiler instance
profiler = MemoryProfiler()

# lambda_function.py
import json
import os
from src.optimizations.connection_pool import connections, get_table
from src.optimizations.memory_profiler import profiler
from src.optimizations.lazy_imports import lazy_import

# Pre-initialize connections outside handler
table = get_table(os.environ.get('DYNAMODB_TABLE', 'products'))

def lambda_handler(event, context):
    """Optimized Lambda handler"""
    profiler.checkpoint('handler_start')
    
    try:
        # Parse request
        http_method = event.get('httpMethod', 'GET')
        path = event.get('path', '/')
        body = json.loads(event.get('body', '{}')) if event.get('body') else {}
        
        profiler.checkpoint('request_parsed')
        
        # Route request
        if path.startswith('/products'):
            response = handle_products(http_method, path, body)
        elif path.startswith('/orders'):
            response = handle_orders(http_method, path, body)
        else:
            response = {
                'statusCode': 404,
                'body': json.dumps({'error': 'Not found'})
            }
        
        profiler.checkpoint('request_processed')
        
        # Add performance headers
        response['headers'] = response.get('headers', {})
        response['headers']['X-Memory-Report'] = json.dumps(profiler.get_report())
        
        return response
        
    except Exception as e:
        profiler.checkpoint('error_occurred')
        return {
            'statusCode': 500,
            'body': json.dumps({'error': str(e)}),
            'headers': {
                'X-Memory-Report': json.dumps(profiler.get_report())
            }
        }

def handle_products(method, path, body):
    """Handle product-related requests"""
    profiler.checkpoint('products_start')
    
    if method == 'GET':
        # Get product list or specific product
        product_id = path.split('/')[-1] if path != '/products' else None
        
        if product_id and product_id != 'products':
            # Get specific product
            response = table.get_item(Key={'id': product_id})
            item = response.get('Item')
            
            if item:
                return {
                    'statusCode': 200,
                    'body': json.dumps(item, default=str)
                }
            else:
                return {
                    'statusCode': 404,
                    'body': json.dumps({'error': 'Product not found'})
                }
        else:
            # Get product list
            response = table.scan(Limit=50)
            items = response.get('Items', [])
            
            return {
                'statusCode': 200,
                'body': json.dumps({
                    'products': items,
                    'count': len(items)
                }, default=str)
            }
    
    elif method == 'POST':
        # Create new product
        product_data = body
        product_data['id'] = str(uuid.uuid4())
        
        table.put_item(Item=product_data)
        
        return {
            'statusCode': 201,
            'body': json.dumps(product_data, default=str)
        }
    
    else:
        return {
            'statusCode': 405,
            'body': json.dumps({'error': 'Method not allowed'})
        }

def handle_orders(method, path, body):

# tests/test_cold_start.py
import boto3
import time
import json
import statistics

def test_cold_start_performance():
    """Test cold start performance"""
    lambda_client = boto3.client('lambda')
    function_name = 'workshop-product-api-optimized'
    
    # Force cold start by updating environment variable
    lambda_client.update_function_configuration(
        FunctionName=function_name,
        Environment={
            'Variables': {
                'FORCE_COLD_START': str(time.time())
            }
        }
    )
    
    # Wait for update to complete
    time.sleep(5)
    
    # Measure cold start time
    cold_start_times = []
    
    for i in range(5):
        start_time = time.time()
        
        response = lambda_client.invoke(
            FunctionName=function_name,
            Payload=json.dumps({
                'httpMethod': 'GET',
                'path': '/products'
            })
        )
        
        end_time = time.time()
        duration = (end_time - start_time) * 1000  # Convert to milliseconds
        cold_start_times.append(duration)
        
        # Force another cold start
        lambda_client.update_function_configuration(
            FunctionName=function_name,
            Environment={
                'Variables': {
                    'FORCE_COLD_START': str(time.time())
                }
            }
        )
        time.sleep(5)
    
    # Calculate statistics
    avg_cold_start = statistics.mean(cold_start_times)
    min_cold_start = min(cold_start_times)
    max_cold_start = max(cold_start_times)
    
    print(f"Cold Start Performance Results:")
    print(f"  Average: {avg_cold_start:.2f}ms")
    print(f"  Minimum: {min_cold_start:.2f}ms")
    print(f"  Maximum: {max_cold_start:.2f}ms")
    print(f"  All times: {[f'{t:.2f}ms' for t in cold_start_times]}")
    
    # Assert performance target
    assert avg_cold_start < 1000, f"Cold start too slow: {avg_cold_start:.2f}ms"
    
    return {
        'average': avg_cold_start,
        'minimum': min_cold_start,
        'maximum': max_cold_start,
        'all_times': cold_start_times
    }

if __name__ == "__main__":
    results = test_cold_start_performance()
    print(f"✅ Cold start test passed! Average: {results['average']:.2f}ms")
