# Create optimized Dockerfile
cat > docker/optimized/Dockerfile << 'EOF'
# Multi-stage build for optimization
FROM public.ecr.aws/lambda/python:3.11 as builder

# Install build dependencies
RUN yum update -y && \
    yum install -y gcc python3-devel && \
    yum clean all

# Create virtual environment
RUN python -m venv /opt/venv
ENV PATH="/opt/venv/bin:$PATH"

# Copy and install requirements
COPY requirements.txt .
RUN pip install --no-cache-dir --upgrade pip && \
    pip install --no-cache-dir -r requirements.txt

# Production stage
FROM public.ecr.aws/lambda/python:3.11

# Copy virtual environment from builder
COPY --from=builder /opt/venv /opt/venv
ENV PATH="/opt/venv/bin:$PATH"

# Copy application code
COPY lambda_function.py ${LAMBDA_TASK_ROOT}
COPY src/ ${LAMBDA_TASK_ROOT}/src/

# Remove unnecessary files
RUN find ${LAMBDA_TASK_ROOT} -type f -name "*.pyc" -delete && \
    find ${LAMBDA_TASK_ROOT} -type d -name "__pycache__" -exec rm -rf {} + || true

# Set handler
CMD [ "lambda_function.lambda_handler" ]
EOF

# Create ultra-optimized Dockerfile
cat > docker/optimized/Dockerfile.ultra << 'EOF'
# Ultra-optimized multi-stage build
FROM public.ecr.aws/lambda/python:3.11 as builder

# Install minimal build dependencies
RUN yum update -y && \
    yum install -y gcc python3-devel && \
    yum clean all && \
    rm -rf /var/cache/yum

# Create optimized virtual environment
RUN python -m venv /opt/venv
ENV PATH="/opt/venv/bin:$PATH"

# Install dependencies with optimizations
COPY requirements.txt .
RUN pip install --no-cache-dir --upgrade pip setuptools wheel && \
    pip install --no-cache-dir --compile -r requirements.txt && \
    find /opt/venv -name "*.pyc" -delete && \
    find /opt/venv -name "__pycache__" -exec rm -rf {} + || true

# Production stage
FROM public.ecr.aws/lambda/python:3.11

# Copy only necessary files from builder
COPY --from=builder /opt/venv/lib/python3.11/site-packages /var/lang/lib/python3.11/site-packages

# Copy application code
COPY lambda_function.py ${LAMBDA_TASK_ROOT}
COPY src/ ${LAMBDA_TASK_ROOT}/src/

# Optimize Python bytecode
RUN python -m compileall -b ${LAMBDA_TASK_ROOT} && \
    find ${LAMBDA_TASK_ROOT} -name "*.py" -delete && \
    find ${LAMBDA_TASK_ROOT} -name "*.pyc" -exec rename .pyc .py {} \;

# Remove unnecessary files and optimize
RUN find ${LAMBDA_TASK_ROOT} -type d -name "__pycache__" -exec rm -rf {} + || true && \
    find /var/lang/lib/python3.11/site-packages -name "*.dist-info" -exec rm -rf {} + || true && \
    find /var/lang/lib/python3.11/site-packages -name "tests" -type d -exec rm -rf {} + || true

# Set handler
CMD [ "lambda_function.lambda_handler" ]
EOF

# Create build script
cat > scripts/build-containers.sh << 'EOF'
#!/bin/bash

set -e

echo "=== Building Lambda Container Images ==="

# Get ECR registry from environment
if [ -z "$ECR_REGISTRY" ]; then
    export ECR_REGISTRY=$(aws ecr describe-repositories \
        --repository-names lambda-workshop/product-api \
        --query 'repositories[0].repositoryUri' \
        --output text | cut -d'/' -f1)
fi

echo "ECR Registry: $ECR_REGISTRY"

# Build base image
echo "Building base image..."
docker build -t lambda-workshop:base -f docker/base/Dockerfile .
docker tag lambda-workshop:base $ECR_REGISTRY/lambda-workshop/product-api:base

# Build optimized image
echo "Building optimized image..."
docker build -t lambda-workshop:optimized -f docker/optimized/Dockerfile .
docker tag lambda-workshop:optimized $ECR_REGISTRY/lambda-workshop/product-api:optimized

# Build ultra-optimized image
echo "Building ultra-optimized image..."
docker build -t lambda-workshop:ultra -f docker/optimized/Dockerfile.ultra .
docker tag lambda-workshop:ultra $ECR_REGISTRY/lambda-workshop/product-api:ultra

# Show image sizes
echo "=== Image Sizes Comparison ==="
docker images | grep lambda-workshop | sort -k7 -h

echo "=== Build Complete ==="
EOF

chmod +x scripts/build-containers.sh

# Create size analysis script
cat > scripts/analyze-sizes.sh << 'EOF'
#!/bin/bash

echo "=== Container Image Size Analysis ==="

# Function to get image size in MB
get_size_mb() {
    docker images --format "table {{.Repository}}:{{.Tag}}\t{{.Size}}" | grep "$1" | awk '{print $2}' | sed 's/MB//'
}

# Get sizes
BASE_SIZE=$(docker inspect lambda-workshop:base --format='{{.Size}}')
OPT_SIZE=$(docker inspect lambda-workshop:optimized --format='{{.Size}}')
ULTRA_SIZE=$(docker inspect lambda-workshop:ultra --format='{{.Size}}')

# Convert to MB
BASE_MB=$((BASE_SIZE / 1024 / 1024))
OPT_MB=$((OPT_SIZE / 1024 / 1024))
ULTRA_MB=$((ULTRA_SIZE / 1024 / 1024))

echo "Base Image:           ${BASE_MB} MB"
echo "Optimized Image:      ${OPT_MB} MB"
echo "Ultra-Optimized:      ${ULTRA_MB} MB"

# Calculate savings
OPT_SAVINGS=$(((BASE_SIZE - OPT_SIZE) * 100 / BASE_SIZE))
ULTRA_SAVINGS=$(((BASE_SIZE - ULTRA_SIZE) * 100 / BASE_SIZE))

echo ""
echo "Size Reduction:"
echo "Optimized:       ${OPT_SAVINGS}% smaller"
echo "Ultra-Optimized: ${ULTRA_SAVINGS}% smaller"

# Layer analysis
echo ""
echo "=== Layer Analysis ==="
echo "Base Image Layers:"
docker history lambda-workshop:base --format "table {{.CreatedBy}}\t{{.Size}}" | head -10

echo ""
echo "Ultra-Optimized Layers:"
docker history lambda-workshop:ultra --format "table {{.CreatedBy}}\t{{.Size}}" | head -10
EOF

# Create container test script
cat > scripts/test-containers.sh << 'EOF'
#!/bin/bash

echo "=== Testing Container Images ==="

# Test event
TEST_EVENT='{
    "httpMethod": "GET",
    "path": "/health",
    "body": null,
    "pathParameters": null
}'

# Function to test container
test_container() {
    local image_name=$1
    echo "Testing $image_name..."
    
    # Run container
    CONTAINER_ID=$(docker run -d -p 9000:8080 $image_name)
    
    # Wait for container to start
    sleep 3
    
    # Test the function
    RESPONSE=$(curl -s -X POST "http://localhost:9000/2015-03-31/functions/function/invocations" \
        -H "Content-Type: application/json" \
        -d "$TEST_EVENT")
    
    echo "Response: $RESPONSE"
    
    # Stop container
    docker stop $CONTAINER_ID > /dev/null
    docker rm $CONTAINER_ID > /dev/null
    
    echo "✅ $image_name test completed"
    echo ""
}

# Test all images
test_container "lambda-workshop:base"
test_container "lambda-workshop:optimized"
test_container "lambda-workshop:ultra"

echo "=== All container tests completed ==="
EOF

# Create benchmark script
cat > scripts/benchmark-containers.sh << 'EOF'
#!/bin/bash

echo "=== Container Performance Benchmark ==="

# Benchmark function
benchmark_container() {
    local image_name=$1
    local iterations=10
    
    echo "Benchmarking $image_name ($iterations iterations)..."
    
    # Start container
    CONTAINER_ID=$(docker run -d -p 9000:8080 $image_name)
    sleep 3
    
    # Warm up
    curl -s -X POST "http://localhost:9000/2015-03-31/functions/function/invocations" \
        -H "Content-Type: application/json" \
        -d '{"httpMethod":"GET","path":"/health"}' > /dev/null
    
    # Benchmark
    total_time=0
    for i in $(seq 1 $iterations); do
        start_time=$(date +%s%N)
        curl -s -X POST "http://localhost:9000/2015-03-31/functions/function/invocations" \
            -H "Content-Type: application/json" \
            -d '{"httpMethod":"GET","path":"/products"}' > /dev/null
        end_time=$(date +%s%N)
        
        duration=$((($end_time - $start_time) / 1000000))  # Convert to milliseconds
        total_time=$(($total_time + $duration))
    done
    
    avg_time=$(($total_time / $iterations))
    echo "$image_name average response time: ${avg_time}ms"
    
    # Stop container
    docker stop $CONTAINER_ID > /dev/null
    docker rm $CONTAINER_ID > /dev/null
    
    echo ""
}

# Benchmark all images
benchmark_container "lambda-workshop:base"
benchmark_container "lambda-workshop:optimized"
benchmark_container "lambda-workshop:ultra"

echo "=== Benchmark completed ==="
EOF

chmod +x scripts/benchmark-containers.sh

Key Optimization Techniques

The optimized containers implement several key techniques:

  1. Multi-stage builds: Separate build and runtime environments
  2. Dependency optimization: Remove build tools from final image
  3. Python bytecode compilation: Pre-compile Python files
  4. Layer optimization: Minimize number and size of layers
  5. File cleanup: Remove unnecessary files and caches
  6. Virtual environment optimization: Efficient dependency management

Size Comparison Results

Typical size reductions achieved:

  • Base Image: ~800MB
  • Optimized Image: ~400MB (50% reduction)
  • Ultra-Optimized: ~200MB (75% reduction)

The ultra-optimized image provides the best cold start performance due to its smaller size, but requires more careful dependency management.
