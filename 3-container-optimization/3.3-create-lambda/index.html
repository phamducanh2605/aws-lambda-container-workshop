# Create Lambda deployment script
cat > scripts/deploy-lambda-functions.sh << 'EOF'
#!/bin/bash

set -e

echo "=== Deploying Lambda Functions ==="

# Get ECR registry and role ARN
if [ -z "$ECR_REGISTRY" ]; then
    export ECR_REGISTRY=$(aws ecr describe-repositories \
        --repository-names lambda-workshop/product-api \
        --query 'repositories[0].repositoryUri' \
        --output text | cut -d'/' -f1)
fi

if [ -z "$LAMBDA_ROLE_ARN" ]; then
    export LAMBDA_ROLE_ARN=$(aws iam get-role \
        --role-name lambda-container-workshop-role \
        --query 'Role.Arn' \
        --output text)
fi

echo "ECR Registry: $ECR_REGISTRY"
echo "Lambda Role: $LAMBDA_ROLE_ARN"

# Function to create Lambda function
create_lambda_function() {
    local function_name=$1
    local image_tag=$2
    local memory_size=$3
    local timeout=$4
    
    echo "Creating Lambda function: $function_name"
    
    # Delete function if exists
    aws lambda delete-function --function-name $function_name 2>/dev/null || true
    sleep 5
    
    # Create function
    aws lambda create-function \
        --function-name $function_name \
        --role $LAMBDA_ROLE_ARN \
        --code ImageUri=$ECR_REGISTRY/lambda-workshop/product-api:$image_tag \
        --package-type Image \
        --memory-size $memory_size \
        --timeout $timeout \
        --environment Variables='{
            "PRODUCTS_TABLE":"workshop-products",
            "AWS_DEFAULT_REGION":"us-east-1"
        }' \
        --description "Workshop Lambda function - $image_tag variant"
    
    # Wait for function to be active
    aws lambda wait function-active --function-name $function_name
    
    echo "✅ $function_name created successfully"
    echo ""
}

# Deploy all variants
create_lambda_function "workshop-product-api-base" "base" 512 30
create_lambda_function "workshop-product-api-optimized" "optimized" 512 30
create_lambda_function "workshop-product-api-ultra" "ultra" 256 15

echo "=== All Lambda functions deployed ==="

# List functions
echo "Deployed functions:"
aws lambda list-functions --query 'Functions[?starts_with(FunctionName, `workshop-product-api`)].{Name:FunctionName,Runtime:PackageType,Memory:MemorySize,Timeout:Timeout}' --output table
EOF

chmod +x scripts/deploy-lambda-functions.sh

# Create API setup script
cat > scripts/setup-api-gateway.sh << 'EOF'
#!/bin/bash

set -e

API_ID=$1
ROOT_RESOURCE_ID=$2
LAMBDA_FUNCTION_NAME=${3:-"workshop-product-api-optimized"}

if [ -z "$API_ID" ] || [ -z "$ROOT_RESOURCE_ID" ]; then
    echo "Usage: $0 <API_ID> <ROOT_RESOURCE_ID> [LAMBDA_FUNCTION_NAME]"
    exit 1
fi

echo "Setting up API Gateway for Lambda function: $LAMBDA_FUNCTION_NAME"

# Get Lambda function ARN
LAMBDA_ARN=$(aws lambda get-function \
    --function-name $LAMBDA_FUNCTION_NAME \
    --query 'Configuration.FunctionArn' \
    --output text)

echo "Lambda ARN: $LAMBDA_ARN"

# Create /products resource
PRODUCTS_RESOURCE_ID=$(aws apigateway create-resource \
    --rest-api-id $API_ID \
    --parent-id $ROOT_RESOURCE_ID \
    --path-part "products" \
    --query 'id' \
    --output text)

# Create /products/{id} resource
PRODUCT_ID_RESOURCE_ID=$(aws apigateway create-resource \
    --rest-api-id $API_ID \
    --parent-id $PRODUCTS_RESOURCE_ID \
    --path-part "{id}" \
    --query 'id' \
    --output text)

# Create /health resource
HEALTH_RESOURCE_ID=$(aws apigateway create-resource \
    --rest-api-id $API_ID \
    --parent-id $ROOT_RESOURCE_ID \
    --path-part "health" \
    --query 'id' \
    --output text)

# Function to create method and integration
create_method_integration() {
    local resource_id=$1
    local http_method=$2
    local path=$3
    
    echo "Creating $http_method method for $path"
    
    # Create method
    aws apigateway put-method \
        --rest-api-id $API_ID \
        --resource-id $resource_id \
        --http-method $http_method \
        --authorization-type NONE
    
    # Create integration
    aws apigateway put-integration \
        --rest-api-id $API_ID \
        --resource-id $resource_id \
        --http-method $http_method \
        --type AWS_PROXY \
        --integration-http-method POST \
        --uri "arn:aws:apigateway:us-east-1:lambda:path/2015-03-31/functions/$LAMBDA_ARN/invocations"
    
    # Add Lambda permission
    aws lambda add-permission \
        --function-name $LAMBDA_FUNCTION_NAME \
        --statement-id "apigateway-$resource_id-$http_method" \
        --action lambda:InvokeFunction \
        --principal apigateway.amazonaws.com \
        --source-arn "arn:aws:execute-api:us-east-1:$(aws sts get-caller-identity --query Account --output text):$API_ID/*/*" \
        2>/dev/null || true
}

# Create methods for all endpoints
create_method_integration $PRODUCTS_RESOURCE_ID "GET" "/products"
create_method_integration $PRODUCTS_RESOURCE_ID "POST" "/products"
create_method_integration $PRODUCT_ID_RESOURCE_ID "GET" "/products/{id}"
create_method_integration $PRODUCT_ID_RESOURCE_ID "PUT" "/products/{id}"
create_method_integration $PRODUCT_ID_RESOURCE_ID "DELETE" "/products/{id}"
create_method_integration $HEALTH_RESOURCE_ID "GET" "/health"

# Deploy API
aws apigateway create-deployment \
    --rest-api-id $API_ID \
    --stage-name prod

echo "✅ API Gateway setup completed"
echo "API URL: https://$API_ID.execute-api.us-east-1.amazonaws.com/prod"
EOF

chmod +x scripts/setup-api-gateway.sh

#!/bin/bash

echo "=== Testing Lambda Functions ==="

# Test event
TEST_EVENT='{
    "httpMethod": "GET",
    "path": "/health",
    "body": null,
    "pathParameters": null,
    "headers": {},
    "queryStringParameters": null
}'

# Function to test Lambda
test_lambda_function() {
    local function_name=$1
    echo "Testing $function_name..."
    
    # Invoke function
    RESPONSE=$(aws lambda invoke \
        --function-name $function_name \
        --payload "$TEST_EVENT" \
        --output json \
        response.json)
    
    # Check response
    STATUS_CODE=$(echo $RESPONSE | jq -r '.StatusCode')
    if [ "$STATUS_CODE" = "200" ]; then
        echo "✅ $function_name: SUCCESS"
        cat response.json | jq -r '.body' | jq .
    else
        echo "❌ $function_name: FAILED"
        cat response.json
    fi
    
    echo ""
    rm -f response.json
}

# Test all functions
test_lambda_function "workshop-product-api-base"
test_lambda_function "workshop-product-api-optimized"
test_lambda_function "workshop-product-api-ultra"

echo "=== Lambda function tests completed ==="
EOF

# Create performance comparison script
cat > scripts/compare-performance.sh << 'EOF'
#!/bin/bash

echo "=== Lambda Function Performance Comparison ==="

# Test configuration
ITERATIONS=10
TEST_EVENT='{
    "httpMethod": "GET",
    "path": "/products",
    "body": null,
    "pathParameters": null,
    "headers": {},
    "queryStringParameters": null
}'

# Function to measure performance
measure_performance() {
    local function_name=$1
    echo "Measuring $function_name performance ($ITERATIONS iterations)..."
    
    total_duration=0
    total_billed_duration=0
    total_memory_used=0
    cold_starts=0
    
    for i in $(seq 1 $ITERATIONS); do
        # Invoke function
        RESPONSE=$(aws lambda invoke \
            --function-name $function_name \
            --payload "$TEST_EVENT" \
            --output json \
            response.json 2>&1)
        
        # Extract metrics from logs
        DURATION=$(aws logs filter-log-events \
            --log-group-name "/aws/lambda/$function_name" \
            --start-time $(date -d '1 minute ago' +%s)000 \
            --filter-pattern "REPORT RequestId" \
            --query 'events[-1].message' \
            --output text | grep -o 'Duration: [0-9.]*' | cut -d' ' -f2 2>/dev/null || echo "0")
        
        BILLED_DURATION=$(aws logs filter-log-events \
            --log-group-name "/aws/lambda/$function_name" \
            --start-time $(date -d '1 minute ago' +%s)000 \
            --filter-pattern "REPORT RequestId" \
            --query 'events[-1].message' \
            --output text | grep -o 'Billed Duration: [0-9]*' | cut -d' ' -f3 2>/dev/null || echo "0")
        
        MEMORY_USED=$(aws logs filter-log-events \
            --log-group-name "/aws/lambda/$function_name" \
            --start-time $(date -d '1 minute ago' +%s)000 \
            --filter-pattern "REPORT RequestId" \
            --query 'events[-1].message' \
            --output text | grep -o 'Memory Size: [0-9]*' | cut -d' ' -f3 2>/dev/null || echo "0")
        
        # Check for cold start
        INIT_DURATION=$(aws logs filter-log-events \
            --log-group-name "/aws/lambda/$function_name" \
            --start-time $(date -d '1 minute ago' +%s)000 \
            --filter-pattern "REPORT RequestId" \
            --query 'events[-1].message' \
            --output text | grep -o 'Init Duration: [0-9.]*' 2>/dev/null || echo "")
        
        if [ ! -z "$INIT_DURATION" ]; then
            cold_starts=$((cold_starts + 1))
        fi
        
        # Accumulate totals
        total_duration=$(echo "$total_duration + $DURATION" | bc -l)
        total_billed_duration=$((total_billed_duration + BILLED_DURATION))
        total_memory_used=$((total_memory_used + MEMORY_USED))
        
        # Wait between invocations
        sleep 2
    done
    
    # Calculate averages
    avg_duration=$(echo "scale=2; $total_duration / $ITERATIONS" | bc -l)
    avg_billed_duration=$((total_billed_duration / ITERATIONS))
    avg_memory_used=$((total_memory_used / ITERATIONS))
    
    echo "Results for $function_name:"
    echo "  Average Duration: ${avg_duration}ms"
    echo "  Average Billed Duration: ${avg_billed_duration}ms"
    echo "  Average Memory Used: ${avg_memory_used}MB"
    echo "  Cold Starts: $cold_starts/$ITERATIONS"
    echo ""
    
    rm -f response.json
}

# Measure all functions
measure_performance "workshop-product-api-base"
measure_performance "workshop-product-api-optimized"
measure_performance "workshop-product-api-ultra"

echo "=== Performance comparison completed ==="
EOF

chmod +x scripts/compare-performance.sh

#!/bin/bash

echo "=== Lambda Function Monitoring ==="

# Function to get CloudWatch metrics
get_metrics() {
    local function_name=$1
    echo "Metrics for $function_name:"
    
    # Get function configuration
    aws lambda get-function-configuration \
        --function-name $function_name \
        --query '{
            MemorySize: MemorySize,
            Timeout: Timeout,
            CodeSize: CodeSize,
            LastModified: LastModified
        }' \
        --output table
    
    echo ""
}

# Monitor all functions
get_metrics "workshop-product-api-base"
get_metrics "workshop-product-api-optimized"
get_metrics "workshop-product-api-ultra"

# Get recent invocations
echo "Recent invocations (last 5 minutes):"
aws logs filter-log-events \
    --log-group-name "/aws/lambda/workshop-product-api-optimized" \
    --start-time $(date -d '5 minutes ago' +%s)000 \
    --filter-pattern "REPORT RequestId" \
    --query 'events[*].message' \
    --output text | tail -5

echo "=== Monitoring completed ==="
EOF

Performance Expectations

Typical performance improvements:

  • Base Image: ~2-3 second cold starts
  • Optimized Image: ~1-2 second cold starts (30-50% improvement)
  • Ultra-Optimized: ~0.5-1 second cold starts (60-80% improvement)

Actual performance will vary based on function complexity, memory allocation, and AWS region. The ultra-optimized variant typically shows the most significant improvement in cold start times.

Next, we will implement comprehensive monitoring and performance analysis.
