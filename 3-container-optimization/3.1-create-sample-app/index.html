# Create lambda_function.py file
cat > lambda_function.py << 'EOF'
import json
import os
import time
import boto3
from datetime import datetime
from src.api.product_service import ProductService
from src.monitoring.performance_metrics import PerformanceMetrics

# Initialize connections outside handler for reuse
product_service = ProductService()
metrics = PerformanceMetrics()

def lambda_handler(event, context):
    """
    Main Lambda handler for Product API
    """
    start_time = time.time()
    
    try:
        # Log request info
        print(f"Request: {json.dumps(event)}")
        
        # Parse request
        http_method = event.get('httpMethod', 'GET')
        path = event.get('path', '/')
        body = json.loads(event.get('body', '{}')) if event.get('body') else {}
        path_parameters = event.get('pathParameters') or {}
        
        # Route requests
        if path == '/products' and http_method == 'GET':
            response = product_service.list_products()
        elif path == '/products' and http_method == 'POST':
            response = product_service.create_product(body)
        elif path.startswith('/products/') and http_method == 'GET':
            product_id = path_parameters.get('id')
            response = product_service.get_product(product_id)
        elif path.startswith('/products/') and http_method == 'PUT':
            product_id = path_parameters.get('id')
            response = product_service.update_product(product_id, body)
        elif path.startswith('/products/') and http_method == 'DELETE':
            product_id = path_parameters.get('id')
            response = product_service.delete_product(product_id)
        elif path == '/health' and http_method == 'GET':
            response = {'status': 'healthy', 'timestamp': datetime.utcnow().isoformat()}
        else:
            response = {'error': 'Not Found', 'path': path, 'method': http_method}
            return create_response(404, response)
        
        # Record metrics
        execution_time = time.time() - start_time
        metrics.record_execution_time(execution_time)
        
        return create_response(200, response)
        
    except Exception as e:
        print(f"Error: {str(e)}")
        error_response = {
            'error': 'Internal Server Error',
            'message': str(e),
            'timestamp': datetime.utcnow().isoformat()
        }
        return create_response(500, error_response)

def create_response(status_code, body):
    """Create HTTP response"""
    return {
        'statusCode': status_code,
        'headers': {
            'Content-Type': 'application/json',
            'Access-Control-Allow-Origin': '*',
            'Access-Control-Allow-Methods': 'GET,POST,PUT,DELETE,OPTIONS',
            'Access-Control-Allow-Headers': 'Content-Type,X-Amz-Date,Authorization,X-Api-Key'
        },
        'body': json.dumps(body, default=str)
    }
EOF

# Create product service
cat > src/api/product_service.py << 'EOF'
import json
import boto3
import uuid
from datetime import datetime
from typing import Dict, List, Optional

class ProductService:
    def __init__(self):
        self.dynamodb = boto3.resource('dynamodb')
        self.table_name = os.environ.get('PRODUCTS_TABLE', 'workshop-products')
        try:
            self.table = self.dynamodb.Table(self.table_name)
        except Exception as e:
            print(f"Warning: Could not connect to DynamoDB table {self.table_name}: {e}")
            self.table = None
    
    def list_products(self) -> Dict:
        """List all products"""
        try:
            if not self.table:
                return self._get_mock_products()
            
            response = self.table.scan()
            products = response.get('Items', [])
            
            return {
                'products': products,
                'count': len(products),
                'timestamp': datetime.utcnow().isoformat()
            }
        except Exception as e:
            print(f"Error listing products: {e}")
            return self._get_mock_products()
    
    def get_product(self, product_id: str) -> Dict:
        """Get product by ID"""
        try:
            if not self.table:
                return self._get_mock_product(product_id)
            
            response = self.table.get_item(Key={'id': product_id})
            product = response.get('Item')
            
            if not product:
                return {'error': 'Product not found', 'id': product_id}
            
            return {'product': product}
        except Exception as e:
            print(f"Error getting product {product_id}: {e}")
            return self._get_mock_product(product_id)
    
    def create_product(self, product_data: Dict) -> Dict:
        """Create new product"""
        try:
            product_id = str(uuid.uuid4())
            product = {
                'id': product_id,
                'name': product_data.get('name', 'Unknown Product'),
                'description': product_data.get('description', ''),
                'price': product_data.get('price', 0.0),
                'category': product_data.get('category', 'general'),
                'created_at': datetime.utcnow().isoformat(),
                'updated_at': datetime.utcnow().isoformat()
            }
            
            if self.table:
                self.table.put_item(Item=product)
            
            return {'product': product, 'message': 'Product created successfully'}
        except Exception as e:
            print(f"Error creating product: {e}")
            return {'error': 'Failed to create product', 'message': str(e)}
    
    def update_product(self, product_id: str, product_data: Dict) -> Dict:
        """Update existing product"""
        try:
            if not self.table:
                return {'message': f'Product {product_id} updated (mock)'}
            
            # Check if product exists
            response = self.table.get_item(Key={'id': product_id})
            if 'Item' not in response:
                return {'error': 'Product not found', 'id': product_id}
            
            # Update product
            update_expression = "SET updated_at = :updated_at"
            expression_values = {':updated_at': datetime.utcnow().isoformat()}
            
            for key, value in product_data.items():
                if key != 'id':
                    update_expression += f", {key} = :{key}"
                    expression_values[f':{key}'] = value
            
            self.table.update_item(
                Key={'id': product_id},
                UpdateExpression=update_expression,
                ExpressionAttributeValues=expression_values
            )
            
            return {'message': f'Product {product_id} updated successfully'}
        except Exception as e:
            print(f"Error updating product {product_id}: {e}")
            return {'error': 'Failed to update product', 'message': str(e)}
    
    def delete_product(self, product_id: str) -> Dict:
        """Delete product"""
        try:
            if not self.table:
                return {'message': f'Product {product_id} deleted (mock)'}
            
            self.table.delete_item(Key={'id': product_id})
            return {'message': f'Product {product_id} deleted successfully'}
        except Exception as e:
            print(f"Error deleting product {product_id}: {e}")
            return {'error': 'Failed to delete product', 'message': str(e)}
    
    def _get_mock_products(self) -> Dict:
        """Return mock products for testing"""
        mock_products = [
            {
                'id': '1',
                'name': 'Sample Product 1',
                'description': 'This is a sample product for testing',
                'price': 29.99,
                'category': 'electronics',
                'created_at': '2025-01-01T00:00:00Z'
            },
            {
                'id': '2',
                'name': 'Sample Product 2',
                'description': 'Another sample product',
                'price': 49.99,
                'category': 'books',
                'created_at': '2025-01-01T00:00:00Z'
            }
        ]
        
        return {
            'products': mock_products,
            'count': len(mock_products),
            'timestamp': datetime.utcnow().isoformat(),
            'note': 'Using mock data - DynamoDB not available'
        }
    
    def _get_mock_product(self, product_id: str) -> Dict:
        """Return mock product for testing"""
        mock_product = {
            'id': product_id,
            'name': f'Sample Product {product_id}',
            'description': 'This is a mock product for testing',
            'price': 19.99,
            'category': 'test',
            'created_at': '2025-01-01T00:00:00Z'
        }
        
        return {
            'product': mock_product,
            'note': 'Using mock data - DynamoDB not available'
        }

# Create performance monitoring
cat > src/monitoring/performance_metrics.py << 'EOF'
import time
import psutil
import os
from datetime import datetime
from typing import Dict

class PerformanceMetrics:
    def __init__(self):
        self.start_time = time.time()
        self.metrics = {
            'execution_times': [],
            'memory_usage': [],
            'cpu_usage': []
        }
    
    def record_execution_time(self, execution_time: float):
        """Record function execution time"""
        self.metrics['execution_times'].append({
            'time': execution_time,
            'timestamp': datetime.utcnow().isoformat()
        })
        print(f"Execution time: {execution_time:.3f}s")
    
    def get_memory_usage(self) -> Dict:
        """Get current memory usage"""
        try:
            process = psutil.Process(os.getpid())
            memory_info = process.memory_info()
            
            memory_data = {
                'rss': memory_info.rss,  # Resident Set Size
                'vms': memory_info.vms,  # Virtual Memory Size
                'percent': process.memory_percent(),
                'timestamp': datetime.utcnow().isoformat()
            }
            
            self.metrics['memory_usage'].append(memory_data)
            return memory_data
        except Exception as e:
            print(f"Error getting memory usage: {e}")
            return {'error': str(e)}
    
    def get_cpu_usage(self) -> Dict:
        """Get current CPU usage"""
        try:
            cpu_percent = psutil.cpu_percent(interval=0.1)
            cpu_data = {
                'percent': cpu_percent,
                'timestamp': datetime.utcnow().isoformat()
            }
            
            self.metrics['cpu_usage'].append(cpu_data)
            return cpu_data
        except Exception as e:
            print(f"Error getting CPU usage: {e}")
            return {'error': str(e)}
    
    def get_container_stats(self) -> Dict:
        """Get container performance statistics"""
        return {
            'uptime': time.time() - self.start_time,
            'memory': self.get_memory_usage(),
            'cpu': self.get_cpu_usage(),
            'execution_count': len(self.metrics['execution_times']),
            'avg_execution_time': self._calculate_avg_execution_time()
        }
    
    def _calculate_avg_execution_time(self) -> float:
        """Calculate average execution time"""
        if not self.metrics['execution_times']:
            return 0.0
        
        total_time = sum(metric['time'] for metric in self.metrics['execution_times'])
        return total_time / len(self.metrics['execution_times'])
EOF

Key Features

The sample application includes:

  • RESTful API: Complete CRUD operations for products
  • Error Handling: Comprehensive error handling and logging
  • Performance Monitoring: Built-in metrics collection
  • Mock Data: Fallback to mock data when DynamoDB is unavailable
  • Health Checks: Health endpoint for monitoring
  • Modular Design: Organized code structure for maintainability

This application is designed to work both locally and in Lambda containers, with graceful fallbacks when AWS services are not available.
