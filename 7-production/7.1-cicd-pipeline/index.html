# Create buildspec for CodeBuild
cat > buildspec.yml << 'EOF'
version: 0.2

phases:
  pre_build:
    commands:
      - echo Logging in to Amazon ECR...
      - aws ecr get-login-password --region $AWS_DEFAULT_REGION | docker login --username AWS --password-stdin $AWS_ACCOUNT_ID.dkr.ecr.$AWS_DEFAULT_REGION.amazonaws.com
      - REPOSITORY_URI=$AWS_ACCOUNT_ID.dkr.ecr.$AWS_DEFAULT_REGION.amazonaws.com/$IMAGE_REPO_NAME
      - COMMIT_HASH=$(echo $CODEBUILD_RESOLVED_SOURCE_VERSION | cut -c 1-7)
      - IMAGE_TAG=${COMMIT_HASH:=latest}
      
  build:
    commands:
      - echo Build started on `date`
      - echo Installing dependencies...
      - pip install -r requirements.txt
      - pip install -r tests/requirements.txt
      
      - echo Running tests...
      - python -m pytest tests/unit/ --verbose --junit-xml=test-results/unit.xml
      - python -m pytest tests/integration/ --verbose --junit-xml=test-results/integration.xml
      
      - echo Running security scans...
      - docker build -t $IMAGE_REPO_NAME:$IMAGE_TAG -f docker/optimized/Dockerfile .
      - trivy image --severity HIGH,CRITICAL $IMAGE_REPO_NAME:$IMAGE_TAG
      
      - echo Building the Docker images...
      - docker build -t $IMAGE_REPO_NAME:base -f docker/base/Dockerfile .
      - docker build -t $IMAGE_REPO_NAME:optimized -f docker/optimized/Dockerfile .
      - docker build -t $IMAGE_REPO_NAME:ultra -f docker/optimized/Dockerfile.ultra .
      
      - echo Tagging images...
      - docker tag $IMAGE_REPO_NAME:base $REPOSITORY_URI:base-$IMAGE_TAG
      - docker tag $IMAGE_REPO_NAME:optimized $REPOSITORY_URI:optimized-$IMAGE_TAG
      - docker tag $IMAGE_REPO_NAME:ultra $REPOSITORY_URI:ultra-$IMAGE_TAG
      - docker tag $IMAGE_REPO_NAME:optimized $REPOSITORY_URI:latest
      
  post_build:
    commands:
      - echo Build completed on `date`
      - echo Pushing the Docker images...
      - docker push $REPOSITORY_URI:base-$IMAGE_TAG
      - docker push $REPOSITORY_URI:optimized-$IMAGE_TAG
      - docker push $REPOSITORY_URI:ultra-$IMAGE_TAG
      - docker push $REPOSITORY_URI:latest
      
      - echo Writing image definitions file...
      - printf '[{"name":"lambda-container","imageUri":"%s"}]' $REPOSITORY_URI:optimized-$IMAGE_TAG > imagedefinitions.json

artifacts:
  files:
    - imagedefinitions.json
    - scripts/**/*
    - environments/**/*
  name: BuildArtifact

reports:
  unit-tests:
    files:
      - test-results/unit.xml
    file-format: JUNITXML
  integration-tests:
    files:
      - test-results/integration.xml
    file-format: JUNITXML
EOF

# Create CodeBuild service role

{
  "Version": "2012-10-17",
  "Statement": [
    {
      "Effect": "Allow",
      "Principal": {
        "Service": "codebuild.amazonaws.com"
      },
      "Action": "sts:AssumeRole"
    }
  ]
}
EOF

# Create the role
aws iam create-role \
    --role-name lambda-workshop-codebuild-role \
    --assume-role-policy-document file://codebuild-trust-policy.json

# Create policy for CodeBuild
cat > codebuild-policy.json << 'EOF'
{
    "Version": "2012-10-17",
    "Statement": [
        {
            "Effect": "Allow",
            "Action": [
                "logs:CreateLogGroup",
                "logs:CreateLogStream",
                "logs:PutLogEvents"
            ],
            "Resource": "arn:aws:logs:*:*:*"
        },
        {
            "Effect": "Allow",
            "Action": [
                "ecr:BatchCheckLayerAvailability",
                "ecr:GetDownloadUrlForLayer",
                "ecr:BatchGetImage",
                "ecr:GetAuthorizationToken",
                "ecr:PutImage",
                "ecr:InitiateLayerUpload",
                "ecr:UploadLayerPart",
                "ecr:CompleteLayerUpload"
            ],
            "Resource": "*"
        },
        {
            "Effect": "Allow",
            "Action": [
                "s3:GetObject",
                "s3:GetObjectVersion",
                "s3:PutObject"
            ],
            "Resource": "*"
        }
    ]
}
EOF

# Attach policy to role
aws iam put-role-policy \
    --role-name lambda-workshop-codebuild-role \
    --policy-name CodeBuildPolicy \
    --policy-document file://codebuild-policy.json

# Get role ARN
CODEBUILD_ROLE_ARN=$(aws iam get-role \
    --role-name lambda-workshop-codebuild-role \
    --query 'Role.Arn' \
    --output text)

echo "CodeBuild Role ARN: $CODEBUILD_ROLE_ARN"

# Create GitHub Actions workflow
mkdir -p .github/workflows

cat > .github/workflows/ci-cd.yml << 'EOF'
name: Lambda Container CI/CD

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main ]

env:
  AWS_REGION: us-east-1
  ECR_REPOSITORY: lambda-workshop/product-api

jobs:
  test:
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v3
    
    - name: Set up Python
      uses: actions/setup-python@v4
      with:
        python-version: '3.11'
    
    - name: Install dependencies
      run: |
        python -m pip install --upgrade pip
        pip install -r requirements.txt
        pip install -r tests/requirements.txt
    
    - name: Run unit tests
      run: |
        pytest tests/unit/ --verbose --junit-xml=test-results/unit.xml --cov=src --cov-report=xml
    
    - name: Run integration tests
      run: |
        pytest tests/integration/ --verbose --junit-xml=test-results/integration.xml
    
    - name: Upload test results
      uses: actions/upload-artifact@v3
      if: always()
      with:
        name: test-results
        path: test-results/
    
    - name: Upload coverage to Codecov
      uses: codecov/codecov-action@v3
      with:
        file: ./coverage.xml
        flags: unittests

  security:
    runs-on: ubuntu-latest
    needs: test
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v3
    
    - name: Run Trivy vulnerability scanner
      uses: aquasecurity/trivy-action@master
      with:
        image-ref: 'public.ecr.aws/lambda/python:3.11'
        format: 'sarif'
        output: 'trivy-results.sarif'
    
    - name: Upload Trivy scan results
      uses: github/codeql-action/upload-sarif@v2
      with:
        sarif_file: 'trivy-results.sarif'

  build-and-deploy:
    runs-on: ubuntu-latest
    needs: [test, security]
    if: github.ref == 'refs/heads/main'
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v3
    
    - name: Configure AWS credentials
      uses: aws-actions/configure-aws-credentials@v2
      with:
        aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
        aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
        aws-region: ${{ env.AWS_REGION }}
    
    - name: Login to Amazon ECR
      id: login-ecr
      uses: aws-actions/amazon-ecr-login@v1
    
    - name: Build, tag, and push images
      env:
        ECR_REGISTRY: ${{ steps.login-ecr.outputs.registry }}
        IMAGE_TAG: ${{ github.sha }}
      run: |
        # Build all variants
        docker build -t $ECR_REGISTRY/$ECR_REPOSITORY:base-$IMAGE_TAG -f docker/base/Dockerfile .
        docker build -t $ECR_REGISTRY/$ECR_REPOSITORY:optimized-$IMAGE_TAG -f docker/optimized/Dockerfile .
        docker build -t $ECR_REGISTRY/$ECR_REPOSITORY:ultra-$IMAGE_TAG -f docker/optimized/Dockerfile.ultra .
        
        # Tag latest
        docker tag $ECR_REGISTRY/$ECR_REPOSITORY:optimized-$IMAGE_TAG $ECR_REGISTRY/$ECR_REPOSITORY:latest
        
        # Push all images
        docker push $ECR_REGISTRY/$ECR_REPOSITORY:base-$IMAGE_TAG
        docker push $ECR_REGISTRY/$ECR_REPOSITORY:optimized-$IMAGE_TAG
        docker push $ECR_REGISTRY/$ECR_REPOSITORY:ultra-$IMAGE_TAG
        docker push $ECR_REGISTRY/$ECR_REPOSITORY:latest
    
    - name: Deploy to staging
      env:
        ECR_REGISTRY: ${{ steps.login-ecr.outputs.registry }}
        IMAGE_TAG: ${{ github.sha }}
      run: |
        # Set environment variables
        export ECR_REGISTRY=$ECR_REGISTRY
        export LAMBDA_ROLE_ARN=$(aws iam get-role --role-name lambda-container-workshop-role --query 'Role.Arn' --output text)
        
        # Deploy to staging
        ./scripts/deploy-pipeline.sh staging optimized-$IMAGE_TAG blue-green
    
    - name: Run staging tests
      run: |
        ./scripts/test-pipeline.sh staging
    
    - name: Deploy to production
      if: success()
      env:
        ECR_REGISTRY: ${{ steps.login-ecr.outputs.registry }}
        IMAGE_TAG: ${{ github.sha }}
      run: |
        # Deploy to production with approval (manual step in real scenario)
        export ECR_REGISTRY=$ECR_REGISTRY
        export LAMBDA_ROLE_ARN=$(aws iam get-role --role-name lambda-container-workshop-role --query 'Role.Arn' --output text)
        
        ./scripts/deploy-pipeline.sh prod ultra-$IMAGE_TAG blue-green
    
    - name: Run production smoke tests
      run: |
        ./scripts/test-pipeline.sh prod
    
    - name: Notify deployment
      if: always()
      uses: 8398a7/action-slack@v3
      with:
        status: ${{ job.status }}
        channel: '#deployments'
        webhook_url: ${{ secrets.SLACK_WEBHOOK }}
EOF

# Create production deployment approval workflow
cat > .github/workflows/production-deploy.yml << 'EOF'
name: Production Deployment

on:
  workflow_dispatch:
    inputs:
      image_tag:
        description: 'Image tag to deploy'
        required: true
        default: 'latest'
      deployment_type:
        description: 'Deployment type'
        required: true
        default: 'blue-green'
        type: choice
        options:
        - blue-green
        - standard

jobs:
  deploy-production:
    runs-on: ubuntu-latest
    environment: production
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v3
    
    - name: Configure AWS credentials
      uses: aws-actions/configure-aws-credentials@v2
      with:
        aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
        aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
        aws-region: us-east-1
    
    - name: Validate image exists
      run: |
        aws ecr describe-images \
          --repository-name lambda-workshop/product-api \
          --image-ids imageTag=${{ github.event.inputs.image_tag }}
    
    - name: Deploy to production
      run: |
        export ECR_REGISTRY=$(aws ecr describe-repositories --repository-names lambda-workshop/product-api --query 'repositories[0].repositoryUri' --output text | cut -d'/' -f1)
        export LAMBDA_ROLE_ARN=$(aws iam get-role --role-name lambda-container-workshop-role --query 'Role.Arn' --output text)
        
        ./scripts/deploy-pipeline.sh prod ${{ github.event.inputs.image_tag }} ${{ github.event.inputs.deployment_type }}
    
    - name: Run production tests
      run: |
        ./scripts/test-pipeline.sh prod
    
    - name: Create deployment tag
      run: |
        git tag -a "prod-${{ github.event.inputs.image_tag }}-$(date +%Y%m%d-%H%M%S)" -m "Production deployment of ${{ github.event.inputs.image_tag }}"
        git push origin --tags
EOF

# Create production deployment Lambda
cat > deploy-production.py << 'EOF'
import json
import boto3
import os

def lambda_handler(event, context):
    """Deploy to production environment with blue-green strategy"""
    
    codepipeline = boto3.client('codepipeline')
    lambda_client = boto3.client('lambda')
    
    try:
        job_id = event['CodePipeline.job']['id']
        
        # Blue-green deployment logic
        function_name = 'workshop-product-api-prod'
        alias_name = 'live'
        
        # Get current version
        current_version = lambda_client.get_alias(
            FunctionName=function_name,
            Name=alias_name
        )['FunctionVersion']
        
        # Update function code
        image_uri = os.environ.get('ECR_REGISTRY') + '/lambda-workshop/product-api:latest'
        lambda_client.update_function_code(
            FunctionName=function_name,
            ImageUri=image_uri
        )
        
        # Wait for update
        lambda_client.get_waiter('function_updated').wait(
            FunctionName=function_name
        )
        
        # Publish new version
        new_version = lambda_client.publish_version(
            FunctionName=function_name,
            Description=f'Production deployment {context.aws_request_id}'
        )['Version']
        
        # Test new version
        test_response = lambda_client.invoke(
            FunctionName=f'{function_name}:{new_version}',
            Payload=json.dumps({
                'httpMethod': 'GET',
                'path': '/health'
            })
        )
        
        if test_response['StatusCode'] != 200:
            raise Exception('New version test failed')
        
        # Gradual traffic shift
        # 10% traffic to new version
        lambda_client.update_alias(
            FunctionName=function_name,
            Name=alias_name,
            FunctionVersion=new_version,
            RoutingConfig={
                'AdditionalVersionWeights': {
                    current_version: 0.9
                }
            }
        )
        
        # In a real scenario, you would monitor metrics here
        # and gradually increase traffic
        
        # Complete shift to new version
        lambda_client.update_alias(
            FunctionName=function_name,
            Name=alias_name,
            FunctionVersion=new_version
        )
        
        codepipeline.put_job_success_result(jobId=job_id)
        
        return {
            'statusCode': 200,
            'body': json.dumps('Production deployment successful')
        }
        
    except Exception as e:
        print(f"Error: {str(e)}")
        codepipeline.put_job_failure_result(
            jobId=job_id,
            failureDetails={'message': str(e)}
        )
        raise e
