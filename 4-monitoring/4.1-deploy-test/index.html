# Create environment deployment script
cat > scripts/deploy-environment.sh << 'EOF'
#!/bin/bash

set -e

ENVIRONMENT=${1:-dev}
IMAGE_TAG=${2:-optimized}

echo "=== Deploying to $ENVIRONMENT environment ==="

# Load environment configuration
CONFIG_FILE="environments/$ENVIRONMENT/config.json"

if [ ! -f "$CONFIG_FILE" ]; then
    echo "Error: Configuration file $CONFIG_FILE not found"
    exit 1
fi

# Extract configuration values
MEMORY_SIZE=$(jq -r '.lambda.memory_size' "$CONFIG_FILE")
TIMEOUT=$(jq -r '.lambda.timeout' "$CONFIG_FILE")
RESERVED_CONCURRENCY=$(jq -r '.lambda.reserved_concurrency' "$CONFIG_FILE")
LOG_LEVEL=$(jq -r '.lambda.environment_variables.LOG_LEVEL' "$CONFIG_FILE")
LOG_RETENTION=$(jq -r '.monitoring.log_retention_days' "$CONFIG_FILE")

echo "Configuration:"
echo "  Memory: ${MEMORY_SIZE}MB"
echo "  Timeout: ${TIMEOUT}s"
echo "  Concurrency: $RESERVED_CONCURRENCY"
echo "  Log Level: $LOG_LEVEL"

# Function name with environment suffix
FUNCTION_NAME="workshop-product-api-$ENVIRONMENT"

# Deploy Lambda function
echo "Deploying Lambda function: $FUNCTION_NAME"

aws lambda create-function \
    --function-name "$FUNCTION_NAME" \
    --role "$LAMBDA_ROLE_ARN" \
    --code ImageUri="$ECR_REGISTRY/lambda-workshop/product-api:$IMAGE_TAG" \
    --package-type Image \
    --memory-size "$MEMORY_SIZE" \
    --timeout "$TIMEOUT" \
    --environment Variables="{
        \"PRODUCTS_TABLE\":\"workshop-products-$ENVIRONMENT\",
        \"AWS_DEFAULT_REGION\":\"us-east-1\",
        \"LOG_LEVEL\":\"$LOG_LEVEL\",
        \"ENVIRONMENT\":\"$ENVIRONMENT\"
    }" \
    --description "Workshop Lambda function - $ENVIRONMENT environment" \
    --tracing-config Mode=Active \
    --reserved-concurrent-executions "$RESERVED_CONCURRENCY" \
    2>/dev/null || \
    aws lambda update-function-code \
        --function-name "$FUNCTION_NAME" \
        --image-uri "$ECR_REGISTRY/lambda-workshop/product-api:$IMAGE_TAG"

# Update function configuration
aws lambda update-function-configuration \
    --function-name "$FUNCTION_NAME" \
    --memory-size "$MEMORY_SIZE" \
    --timeout "$TIMEOUT" \
    --environment Variables="{
        \"PRODUCTS_TABLE\":\"workshop-products-$ENVIRONMENT\",
        \"AWS_DEFAULT_REGION\":\"us-east-1\",
        \"LOG_LEVEL\":\"$LOG_LEVEL\",
        \"ENVIRONMENT\":\"$ENVIRONMENT\"
    }" \
    --tracing-config Mode=Active

# Set reserved concurrency
aws lambda put-reserved-concurrency \
    --function-name "$FUNCTION_NAME" \
    --reserved-concurrent-executions "$RESERVED_CONCURRENCY"

# Configure log retention
LOG_GROUP="/aws/lambda/$FUNCTION_NAME"
aws logs create-log-group --log-group-name "$LOG_GROUP" 2>/dev/null || true
aws logs put-retention-policy \
    --log-group-name "$LOG_GROUP" \
    --retention-in-days "$LOG_RETENTION"

echo "✅ Deployment to $ENVIRONMENT completed successfully"
EOF

chmod +x scripts/deploy-environment.sh

# Create blue-green deployment script
cat > scripts/blue-green-deploy.sh << 'EOF'
#!/bin/bash

set -e

ENVIRONMENT=${1:-staging}
NEW_IMAGE_TAG=${2:-optimized}

echo "=== Blue-Green Deployment to $ENVIRONMENT ==="

FUNCTION_NAME="workshop-product-api-$ENVIRONMENT"
ALIAS_NAME="live"

# Get current version (blue)
CURRENT_VERSION=$(aws lambda get-alias \
    --function-name "$FUNCTION_NAME" \
    --name "$ALIAS_NAME" \
    --query 'FunctionVersion' \
    --output text 2>/dev/null || echo "\$LATEST")

echo "Current version (blue): $CURRENT_VERSION"

# Deploy new version (green)
echo "Deploying new version (green)..."

# Update function code
aws lambda update-function-code \
    --function-name "$FUNCTION_NAME" \
    --image-uri "$ECR_REGISTRY/lambda-workshop/product-api:$NEW_IMAGE_TAG"

# Wait for update to complete
aws lambda wait function-updated \
    --function-name "$FUNCTION_NAME"

# Publish new version
NEW_VERSION=$(aws lambda publish-version \
    --function-name "$FUNCTION_NAME" \
    --description "Blue-green deployment $(date)" \
    --query 'Version' \
    --output text)

echo "New version (green): $NEW_VERSION"

# Test new version
echo "Testing new version..."
TEST_RESULT=$(aws lambda invoke \
    --function-name "$FUNCTION_NAME:$NEW_VERSION" \
    --payload '{"httpMethod":"GET","path":"/health"}' \
    --output json \
    response.json)

STATUS_CODE=$(echo "$TEST_RESULT" | jq -r '.StatusCode')

if [ "$STATUS_CODE" != "200" ]; then
    echo "❌ New version test failed"
    cat response.json
    exit 1
fi

echo "✅ New version test passed"

# Gradual traffic shift
echo "Starting gradual traffic shift..."

# Create or update alias with weighted routing
if [ "$CURRENT_VERSION" != "\$LATEST" ]; then
    # Update alias with 10% traffic to new version
    aws lambda update-alias \
        --function-name "$FUNCTION_NAME" \
        --name "$ALIAS_NAME" \
        --function-version "$NEW_VERSION" \
        --routing-config AdditionalVersionWeights="{\"$CURRENT_VERSION\":0.9}"
    
    echo "10% traffic shifted to new version"
    
    # Wait and monitor
    echo "Monitoring for 2 minutes..."
    sleep 120
    
    # Check error rates
    ERROR_RATE=$(aws logs filter-log-events \
        --log-group-name "/aws/lambda/$FUNCTION_NAME" \
        --start-time $(date -d '2 minutes ago' +%s)000 \
        --filter-pattern "ERROR" \
        --query 'length(events)' \
        --output text)
    
    if [ "$ERROR_RATE" -gt 5 ]; then
        echo "❌ High error rate detected, rolling back..."
        aws lambda update-alias \
            --function-name "$FUNCTION_NAME" \
            --name "$ALIAS_NAME" \
            --function-version "$CURRENT_VERSION"
        echo "Rollback completed"
        exit 1
    fi
    
    # Shift 50% traffic
    aws lambda update-alias \
        --function-name "$FUNCTION_NAME" \
        --name "$ALIAS_NAME" \
        --function-version "$NEW_VERSION" \
        --routing-config AdditionalVersionWeights="{\"$CURRENT_VERSION\":0.5}"
    
    echo "50% traffic shifted to new version"
    sleep 60
    
    # Final shift - 100% to new version
    aws lambda update-alias \
        --function-name "$FUNCTION_NAME" \
        --name "$ALIAS_NAME" \
        --function-version "$NEW_VERSION"
    
    echo "100% traffic shifted to new version"
else
    # First deployment - create alias
    aws lambda create-alias \
        --function-name "$FUNCTION_NAME" \
        --name "$ALIAS_NAME" \
        --function-version "$NEW_VERSION" \
        --description "Live traffic alias" 2>/dev/null || \
    aws lambda update-alias \
        --function-name "$FUNCTION_NAME" \
        --name "$ALIAS_NAME" \
        --function-version "$NEW_VERSION"
fi

echo "✅ Blue-green deployment completed successfully"

# Cleanup old versions (keep last 3)
echo "Cleaning up old versions..."
aws lambda list-versions-by-function \
    --function-name "$FUNCTION_NAME" \
    --query 'Versions[?Version!=`$LATEST`].Version' \
    --output text | \
    tr '\t' '\n' | \
    sort -nr | \
    tail -n +4 | \
    while read version; do
        echo "Deleting version $version"
        aws lambda delete-function \
            --function-name "$FUNCTION_NAME:$version" 2>/dev/null || true
    done

rm -f response.json
echo "✅ Cleanup completed"
EOF

chmod +x scripts/blue-green-deploy.sh

# Create test pipeline script
cat > scripts/test-pipeline.sh << 'EOF'
#!/bin/bash

set -e

ENVIRONMENT=${1:-staging}
FUNCTION_NAME="workshop-product-api-$ENVIRONMENT"

echo "=== Automated Testing Pipeline for $ENVIRONMENT ==="

# Test configuration
API_URL="https://$(aws apigateway get-rest-apis \
    --query 'items[?name==`workshop-product-api`].id' \
    --output text).execute-api.us-east-1.amazonaws.com/prod"

echo "Testing API: $API_URL"

# Function to run test and capture results
run_test() {
    local test_name=$1
    local test_command=$2
    
    echo "Running $test_name..."
    
    if eval "$test_command"; then
        echo "✅ $test_name: PASSED"
        return 0
    else
        echo "❌ $test_name: FAILED"
        return 1
    fi
}

# Health check test
health_test() {
    response=$(curl -s -w "%{http_code}" "$API_URL/health")
    http_code="${response: -3}"
    
    if [ "$http_code" = "200" ]; then
        body="${response%???}"
        if echo "$body" | jq -e '.status == "healthy"' > /dev/null; then
            return 0
        fi
    fi
    return 1
}

# Load test
load_test() {
    # Simple load test with curl
    local success_count=0
    local total_requests=20
    
    for i in $(seq 1 $total_requests); do
        if curl -s "$API_URL/health" | jq -e '.status == "healthy"' > /dev/null; then
            success_count=$((success_count + 1))
        fi
        sleep 0.1
    done
    
    success_rate=$((success_count * 100 / total_requests))
    echo "Success rate: $success_rate%"
    
    [ $success_rate -ge 95 ]  # 95% success rate required
}

# Performance test
performance_test() {
    local total_time=0
    local iterations=10
    
    for i in $(seq 1 $iterations); do
        start_time=$(date +%s%N)
        curl -s "$API_URL/health" > /dev/null
        end_time=$(date +%s%N)
        
        duration=$(( (end_time - start_time) / 1000000 ))  # Convert to milliseconds
        total_time=$((total_time + duration))
    done
    
    avg_time=$((total_time / iterations))
    echo "Average response time: ${avg_time}ms"
    
    [ $avg_time -lt 1000 ]  # Less than 1 second average
}

# Error handling test
error_test() {
    # Test invalid endpoint
    response=$(curl -s -w "%{http_code}" "$API_URL/invalid-endpoint")
    http_code="${response: -3}"
    
    [ "$http_code" = "404" ]
}

# Security test
security_test() {
    # Test for security headers
    headers=$(curl -s -I "$API_URL/health")
    
    if echo "$headers" | grep -q "X-Content-Type-Options: nosniff" && \
       echo "$headers" | grep -q "X-Frame-Options: DENY"; then
        return 0
    fi
    return 1
}

# Run all tests
test_results=()

run_test "Health Check" "health_test"
test_results+=($?)

run_test "Load Test" "load_test"
test_results+=($?)

run_test "Performance Test" "performance_test"
test_results+=($?)

run_test "Error Handling" "error_test"
test_results+=($?)

run_test "Security Headers" "security_test"
test_results+=($?)

# Calculate overall result
failed_tests=0
for result in "${test_results[@]}"; do
    if [ $result -ne 0 ]; then
        failed_tests=$((failed_tests + 1))
    fi
done

echo ""
echo "=== Test Results Summary ==="
echo "Total tests: ${#test_results[@]}"
echo "Failed tests: $failed_tests"
echo "Success rate: $(( (${#test_results[@]} - failed_tests) * 100 / ${#test_results[@]} ))%"

if [ $failed_tests -eq 0 ]; then
    echo "🎉 All tests passed!"
    exit 0
else
    echo "💥 $failed_tests test(s) failed!"
    exit 1
fi
EOF

chmod +x scripts/test-pipeline.sh

# Create monitoring setup script
cat > scripts/setup-monitoring.sh << 'EOF'
#!/bin/bash

set -e

ENVIRONMENT=${1:-staging}
FUNCTION_NAME="workshop-product-api-$ENVIRONMENT"

echo "=== Setting up monitoring for $ENVIRONMENT ==="

# Get AWS account ID
ACCOUNT_ID=$(aws sts get-caller-identity --query Account --output text)

# Create SNS topic for alerts
TOPIC_ARN=$(aws sns create-topic \
    --name "lambda-workshop-alerts-$ENVIRONMENT" \
    --query 'TopicArn' \
    --output text)

echo "SNS Topic: $TOPIC_ARN"

# Subscribe email to topic (replace with your email)
# aws sns subscribe \
#     --topic-arn "$TOPIC_ARN" \
#     --protocol email \
#     --notification-endpoint your-email@example.com

# Create CloudWatch alarms
create_alarm() {
    local alarm_name=$1
    local metric_name=$2
    local threshold=$3
    local comparison_operator=$4
    local description=$5
    
    aws cloudwatch put-metric-alarm \
        --alarm-name "$alarm_name" \
        --alarm-description "$description" \
        --metric-name "$metric_name" \
        --namespace AWS/Lambda \
        --statistic Average \
        --period 300 \
        --threshold "$threshold" \
        --comparison-operator "$comparison_operator" \
        --evaluation-periods 2 \
        --alarm-actions "$TOPIC_ARN" \
        --dimensions Name=FunctionName,Value="$FUNCTION_NAME"
    
    echo "Created alarm: $alarm_name"
}

# Error rate alarm
create_alarm \
    "lambda-workshop-error-rate-$ENVIRONMENT" \
    "Errors" \
    5 \
    "GreaterThanThreshold" \
    "Lambda function error rate is high"

# Duration alarm
create_alarm \
    "lambda-workshop-duration-$ENVIRONMENT" \
    "Duration" \
    10000 \
    "GreaterThanThreshold" \
    "Lambda function duration is high"

# Throttle alarm
create_alarm \
    "lambda-workshop-throttles-$ENVIRONMENT" \
    "Throttles" \
    1 \
    "GreaterThanOrEqualToThreshold" \
    "Lambda function is being throttled"

# Create custom metric for cold starts
aws logs put-metric-filter \
    --log-group-name "/aws/lambda/$FUNCTION_NAME" \
    --filter-name "ColdStartFilter" \
    --filter-pattern "[timestamp, requestId, version=\"INIT_START\"]" \
    --metric-transformations \
        metricName=ColdStarts,metricNamespace=Lambda/Workshop,metricValue=1,defaultValue=0

# Cold start alarm
create_alarm \
    "lambda-workshop-cold-starts-$ENVIRONMENT" \
    "ColdStarts" \
    10 \
    "GreaterThanThreshold" \
    "High number of cold starts detected"

echo "✅ Monitoring setup completed"
EOF

chmod +x scripts/setup-monitoring.sh

# Create rollback script
cat > scripts/rollback.sh << 'EOF'
#!/bin/bash

set -e

ENVIRONMENT=${1:-staging}
TARGET_VERSION=${2}

echo "=== Rollback for $ENVIRONMENT environment ==="

FUNCTION_NAME="workshop-product-api-$ENVIRONMENT"
ALIAS_NAME="live"

if [ -z "$TARGET_VERSION" ]; then
    # Get previous version
    VERSIONS=$(aws lambda list-versions-by-function \
        --function-name "$FUNCTION_NAME" \
        --query 'Versions[?Version!=`$LATEST`].Version' \
        --output text | tr '\t' '\n' | sort -nr)
    
    CURRENT_VERSION=$(aws lambda get-alias \
        --function-name "$FUNCTION_NAME" \
        --name "$ALIAS_NAME" \
        --query 'FunctionVersion' \
        --output text)
    
    # Get the version before current
    TARGET_VERSION=$(echo "$VERSIONS" | grep -A1 "^$CURRENT_VERSION$" | tail -1)
    
    if [ -z "$TARGET_VERSION" ] || [ "$TARGET_VERSION" = "$CURRENT_VERSION" ]; then
        echo "❌ No previous version found for rollback"
        exit 1
    fi
fi

echo "Current version: $(aws lambda get-alias --function-name "$FUNCTION_NAME" --name "$ALIAS_NAME" --query 'FunctionVersion' --output text)"
echo "Rolling back to version: $TARGET_VERSION"

# Test target version before rollback
echo "Testing target version..."
TEST_RESULT=$(aws lambda invoke \
    --function-name "$FUNCTION_NAME:$TARGET_VERSION" \
    --payload '{"httpMethod":"GET","path":"/health"}' \
    --output json \
    response.json)

STATUS_CODE=$(echo "$TEST_RESULT" | jq -r '.StatusCode')

if [ "$STATUS_CODE" != "200" ]; then
    echo "❌ Target version test failed"
    cat response.json
    exit 1
fi

echo "✅ Target version test passed"

# Perform rollback
aws lambda update-alias \
    --function-name "$FUNCTION_NAME" \
    --name "$ALIAS_NAME" \
    --function-version "$TARGET_VERSION"

echo "✅ Rollback completed successfully"

# Verify rollback
NEW_VERSION=$(aws lambda get-alias \
    --function-name "$FUNCTION_NAME" \
    --name "$ALIAS_NAME" \
    --query 'FunctionVersion' \
    --output text)

echo "Current version after rollback: $NEW_VERSION"

# Test after rollback
echo "Testing after rollback..."
if curl -s "https://$(aws apigateway get-rest-apis --query 'items[?name==`workshop-product-api`].id' --output text).execute-api.us-east-1.amazonaws.com/prod/health" | jq -e '.status == "healthy"' > /dev/null; then
    echo "✅ Post-rollback test passed"
else
    echo "❌ Post-rollback test failed"
    exit 1
fi

rm -f response.json
EOF

chmod +x scripts/rollback.sh

    --query 'Configuration.State' \

    --log-group-name "/aws/lambda/$FUNCTION_NAME" \

    --filter-pattern "REPORT RequestId" \
    --query 'length(events)' \

# Create deployment validation script
cat > scripts/validate-deployment.sh << 'EOF'
#!/bin/bash

ENVIRONMENT=${1:-staging}
FUNCTION_NAME="workshop-product-api-$ENVIRONMENT"

echo "=== Deployment Validation for $ENVIRONMENT ==="

# Validation checklist
validate_function_exists() {
    aws lambda get-function --function-name "$FUNCTION_NAME" > /dev/null 2>&1
}

validate_function_active() {
    local state=$(aws lambda get-function \
        --function-name "$FUNCTION_NAME" \
        --query 'Configuration.State' \
        --output text)
    [ "$state" = "Active" ]
}

validate_alias_exists() {
    aws lambda get-alias \
        --function-name "$FUNCTION_NAME" \
        --name "live" > /dev/null 2>&1
}

validate_monitoring() {
    aws cloudwatch describe-alarms \
        --alarm-names "lambda-workshop-error-rate-$ENVIRONMENT" > /dev/null 2>&1
}

validate_api_gateway() {
    local api_id=$(aws apigateway get-rest-apis \
        --query 'items[?name==`workshop-product-api`].id' \
        --output text)
    [ ! -z "$api_id" ]
}

# Run validations
validations=(
    "Function Exists:validate_function_exists"
    "Function Active:validate_function_active"
    "Alias Exists:validate_alias_exists"
    "Monitoring Setup:validate_monitoring"
    "API Gateway:validate_api_gateway"
)

failed_validations=0

for validation in "${validations[@]}"; do
    name="${validation%%:*}"
    func="${validation##*:}"
    
    echo -n "Validating $name: "
    
    if $func; then
        echo "✅ PASS"
    else
        echo "❌ FAIL"
        failed_validations=$((failed_validations + 1))
    fi
done

echo ""
echo "=== Validation Summary ==="
echo "Total validations: ${#validations[@]}"
echo "Failed validations: $failed_validations"

if [ $failed_validations -eq 0 ]; then
    echo "🎉 All validations passed!"
    exit 0
else
    echo "💥 $failed_validations validation(s) failed!"
    exit 1
fi
EOF

chmod +x scripts/validate-deployment.sh
