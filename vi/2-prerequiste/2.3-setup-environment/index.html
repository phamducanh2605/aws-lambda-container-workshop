# Tạo ECR repository
aws ecr create-repository \
    --repository-name lambda-workshop/product-api \
    --region us-east-1

# Lấy URI của repository
export ECR_REGISTRY=$(aws ecr describe-repositories \
    --repository-names lambda-workshop/product-api \
    --query 'repositories[0].repositoryUri' \
    --output text | cut -d'/' -f1)

echo "ECR Registry: $ECR_REGISTRY"

# Lưu vào file để sử dụng sau
echo "export ECR_REGISTRY=$ECR_REGISTRY" >> ~/.bashrc

cat > scripts/check-environment.sh << 'EOF'
#!/bin/bash

echo "=== Kiểm tra môi trường phát triển ==="

# Kiểm tra Docker
echo -n "Docker: "
if command -v docker &> /dev/null && docker info &> /dev/null; then
    echo "✅ $(docker --version)"
else
    echo "❌ Chưa cài đặt hoặc không chạy"
fi

# Kiểm tra AWS CLI
echo -n "AWS CLI: "
if command -v aws &> /dev/null; then
    echo "✅ $(aws --version)"
else
    echo "❌ Chưa cài đặt"
fi

# Kiểm tra Python
echo -n "Python 3.11: "
if command -v python3.11 &> /dev/null; then
    echo "✅ $(python3.11 --version)"
else
    echo "❌ Chưa cài đặt"
fi

# Kiểm tra AWS credentials
echo -n "AWS Credentials: "
if aws sts get-caller-identity &> /dev/null; then
    echo "✅ Account: $(aws sts get-caller-identity --query 'Account' --output text)"
else
    echo "❌ Chưa cấu hình"
fi

# Kiểm tra ECR access
echo -n "ECR Access: "
if aws ecr describe-repositories --repository-names lambda-workshop/product-api &> /dev/null; then
    echo "✅ Repository đã tạo"
else
    echo "❌ Repository chưa tạo"
fi

echo "=== Hoàn thành kiểm tra ==="
EOF

chmod +x scripts/check-environment.sh

Kết luận

Bạn đã ho à n th à nh vi ệ c thiết lập môi trường phát triển với:

  • ✅ Docker để build Container Images
  • ✅ AWS CLI để t ươ ng t á c với AWS services
  • ✅ Python 3.11 runtime environment
  • ✅ ECR repository để l ư u tr ữ images
  • ✅ C ấ u tr ú c project organized

Hãy chạy ./scripts/check-environment.sh để đảm bảo tất cả đã được cài đặt đúng cách trước khi tiếp tục.

Tiếp theo, chúng ta sẽ b ắ t đầ u tạo ứng dụng mẫu và build Container Images.
