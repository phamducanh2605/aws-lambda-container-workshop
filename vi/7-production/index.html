#!/bin/bash
# scripts/blue-green-deploy.sh

set -e

FUNCTION_NAME=${1:-workshop-product-api}
NEW_IMAGE_URI=${2}
TRAFFIC_SHIFT_PERCENTAGE=${3:-10}

if [ -z "$NEW_IMAGE_URI" ]; then
    echo "Usage: $0 <function-name> <image-uri> [traffic-percentage]"
    exit 1
fi

echo "Starting blue-green Triển khai for $FUNCTION_NAME"

# Step 1: Update function code with new image
echo "Updating function code..."
NEW_VERSION=$(AWS Lambda update-function-code \
    --function-name $FUNCTION_NAME \
    --image-uri $NEW_IMAGE_URI \
    --query 'Version' --output text)

echo "New version created: $NEW_VERSION"

# Step 2: Run smoke tests
echo "Running smoke tests..."
./scripts/smoke-test.sh $FUNCTION_NAME $NEW_VERSION

# Step 3: Gradual traffic shift
echo "Shifting $TRAFFIC_SHIFT_PERCENTAGE% traffic to new version..."
AWS Lambda update-alias \
    --function-name $FUNCTION_NAME \
    --name PROD \
    --routing-config "AdditionalVersionWeights={\"$NEW_VERSION\":$(echo "scale=2; $TRAFFIC_SHIFT_PERCENTAGE/100" | bc)}"

echo "Triển khaiment completed successfully!"

#!/bin/bash
# scripts/smoke-test.sh

FUNCTION_NAME=$1
VERSION=$2

echo "Running smoke tests for $FUNCTION_NAME:$VERSION"

# Kiểm thử basic functionality
RESPONSE=$(AWS Lambda invoke \
    --function-name $FUNCTION_NAME:$VERSION \
    --payload '{"httpMethod":"GET","path":"/health"}' \
    response.json)

STATUS_CODE=$(echo $RESPONSE | jq -r '.StatusCode')

if [ "$STATUS_CODE" != "200" ]; then
    echo "Smoke test failed: Status code $STATUS_CODE"
    exit 1
fi

echo "Smoke tests passed!"

#!/bin/bash
# scripts/setup-Môi trường sản xuất-alarms.sh

FUNCTION_NAME="workshop-product-api"
SNS_TOPIC="arn:aws:sns:${AWS_REGION}:${AWS_ACCOUNT_ID}:Môi trường sản xuất-alerts"

# Error Rate Alarm
aws CloudWatch put-metric-alarm \
    --alarm-name "PROD-${FUNCTION_NAME}-ErrorRate" \
    --alarm-description "Môi trường sản xuất error rate too high" \
    --metric-name Errors \
    --namespace AWS/Lambda \
    --statistic Sum \
    --period 300 \
    --threshold 5 \
    --comparison-operator GreaterThanThreshold \
    --evaluation-periods 2 \
    --alarm-actions $SNS_TOPIC \
    --dimensions Name=FunctionName,Value=$FUNCTION_NAME

# Duration Alarm
aws CloudWatch put-metric-alarm \
    --alarm-name "PROD-${FUNCTION_NAME}-Duration" \
    --alarm-description "Môi trường sản xuất duration too high" \
    --metric-name Duration \
    --namespace AWS/Lambda \
    --statistic Average \
    --period 300 \
    --threshold 5000 \
    --comparison-operator GreaterThanThreshold \
    --evaluation-periods 2 \
    --alarm-actions $SNS_TOPIC \
    --dimensions Name=FunctionName,Value=$FUNCTION_NAME

echo "Môi trường sản xuất alarms configured!"

# src/Triển khai/rollback.py
import boto3
import json

def rollback_deployment(function_name: str, reason: str = "Hiệu suất degradation"):
    """Automated rollback to previous version"""
    lambda_client = boto3.client('lambda')

    try:
        # Get current alias configuration
        alias_config = lambda_client.get_alias(
            FunctionName=function_name,
            Name='PROD'
        )

        current_version = alias_config['FunctionVersion']

        # Get previous version from versions list
        versions = lambda_client.list_versions_by_function(
            FunctionName=function_name
        )

        # Find previous version
        version_numbers = [int(v['Version']) for v in versions['Versions'] if v['Version'] != '$LATEST']
        version_numbers.sort(reverse=True)

        if len(version_numbers) < 2:
            raise Exception("No previous version available for rollback")

        previous_version = str(version_numbers[1])

        # Update alias to point to previous version
        lambda_client.update_alias(
            FunctionName=function_name,
            Name='PROD',
            FunctionVersion=previous_version,
            Description=f"Rollback from {current_version} - {reason}"
        )

        print(f"Rollback completed: {current_version} -> {previous_version}")
        return True

    except Exception as e:
        print(f"Rollback failed: {e}")
        return False
