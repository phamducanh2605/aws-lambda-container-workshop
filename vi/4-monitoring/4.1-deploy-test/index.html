cat > scripts/run-load-test.sh << 'EOF'
#!/bin/bash

FUNCTION_TYPE=${1:-"optimized"}
TEST_DURATION=${2:-300}

if [ "$FUNCTION_TYPE" != "optimized" ] && [ "$FUNCTION_TYPE" != "base" ]; then
    echo "Usage: $0 [optimized|base] [duration_seconds]"
    echo "Example: $0 optimized 300"
    exit 1
fi

echo "=== Running Load Test for $FUNCTION_TYPE function ==="

# Get API endpoint based on function type
if [ "$FUNCTION_TYPE" = "optimized" ]; then
    FUNCTION_NAME="workshop-product-api-optimized"
else
    FUNCTION_NAME="workshop-product-api-base"
fi

# Get API Gateway ID (assuming it was created)
API_ID=$(aws apigateway get-rest-apis \
    --query "items[?name=='workshop-product-api'].id" \
    --output text)

if [ -z "$API_ID" ] || [ "$API_ID" = "None" ]; then
    echo "API Gateway not found. Creating one..."
    ./scripts/create-api-gateway.sh $FUNCTION_NAME
    API_ID=$(aws apigateway get-rest-apis \
        --query "items[?name=='workshop-product-api'].id" \
        --output text)
fi

API_ENDPOINT="https://${API_ID}.execute-api.us-east-1.amazonaws.com/prod"
echo "Kiểm thử endpoint: $API_ENDPOINT"

# Update config file with actual endpoint
sed "s|target: 'https://YOUR_API_ID.execute-api.us-east-1.amazonaws.com/prod'|target: '$API_ENDPOINT'|" \
    tests/load-test-config.yml > tests/load-test-${FUNCTION_TYPE}.yml

# Update test duration
sed -i "s/duration: 300/duration: $TEST_DURATION/" tests/load-test-${FUNCTION_TYPE}.yml

# Run load test
echo "Starting load test..."
artillery run tests/load-test-${FUNCTION_TYPE}.yml \
    --output tests/load-test-${FUNCTION_TYPE}-results.json

# Generate HTML report
artillery report tests/load-test-${FUNCTION_TYPE}-results.json \
    --output tests/load-test-${FUNCTION_TYPE}-report.html

echo "✅ Load test completed!"
echo "Results: tests/load-test-${FUNCTION_TYPE}-results.json"
echo "Report: tests/load-test-${FUNCTION_TYPE}-report.html"
EOF

chmod +x scripts/run-load-test.sh

cat > scripts/benchmark-cold-starts.sh << 'EOF'
#!/bin/bash

FUNCTION_NAME=${1:-"workshop-product-api-optimized"}
NUM_TESTS=${2:-10}

echo "=== Cold Start Benchmark for $FUNCTION_NAME ==="
echo "Number of tests: $NUM_TESTS"

# Create results file
RESULTS_FILE="tests/cold-start-${FUNCTION_NAME}-$(date +%Y%m%d-%H%M%S).csv"
echo "Test,Duration,InitDuration,BilledDuration,MemoryUsed,MaxMemoryUsed" > $RESULTS_FILE

for i in $(seq 1 $NUM_TESTS); do
    echo "Test $i/$NUM_TESTS..."

    # Wait to ensure Cold Start
    sleep 30

    # Invoke function and capture metrics
    RESPONSE=$(AWS Lambda invoke \
        --function-name $FUNCTION_NAME \
        --payload '{"httpMethod":"GET","path":"/health"}' \
        --log-type Tail \
        response.json 2>&1)

    # Extract duration from log
    DURATION=$(echo "$RESPONSE" | grep -o '"Duration": [0-9.]*' | cut -d' ' -f2)
    INIT_DURATION=$(echo "$RESPONSE" | grep -o '"Init Duration": [0-9.]*' | cut -d' ' -f3 || echo "0")
    BILLED_DURATION=$(echo "$RESPONSE" | grep -o '"Billed Duration": [0-9]*' | cut -d' ' -f3)
    MEMORY_USED=$(echo "$RESPONSE" | grep -o '"Memory Used": [0-9]*' | cut -d' ' -f3)
    MAX_MEMORY=$(echo "$RESPONSE" | grep -o '"Max Memory Used": [0-9]*' | cut -d' ' -f4)

    # Save to CSV
    echo "$i,$DURATION,$INIT_DURATION,$BILLED_DURATION,$MEMORY_USED,$MAX_MEMORY" >> $RESULTS_FILE

    echo "  Duration: ${DURATION}ms, Init: ${INIT_DURATION}ms, Memory: ${MEMORY_USED}MB"
done

# Calculate averages
echo "=== Cold Start Results ==="
echo "Results saved to: $RESULTS_FILE"

# Calculate average Cold Start time
AVG_DURATION=$(awk -F',' 'NR>1 {sum+=$2; count++} END {print sum/count}' $RESULTS_FILE)
AVG_INIT=$(awk -F',' 'NR>1 {sum+=$3; count++} END {print sum/count}' $RESULTS_FILE)
AVG_MEMORY=$(awk -F',' 'NR>1 {sum+=$5; count++} END {print sum/count}' $RESULTS_FILE)

echo "Average Duration: ${AVG_DURATION}ms"
echo "Average Init Duration: ${AVG_INIT}ms"
echo "Average Memory Used: ${AVG_MEMORY}MB"

# Dọn dẹp
rm -f response.json
EOF

chmod +x scripts/benchmark-cold-starts.sh

cat > scripts/compare-functions.sh << 'EOF'
#!/bin/bash

echo "=== Comparing Function Hiệu suất ==="

OPTIMIZED_FUNCTION="workshop-product-api-optimized"
BASE_FUNCTION="workshop-product-api-base"

# Test cold starts
echo "1. Kiểm thử Cold Starts..."
./scripts/benchmark-cold-starts.sh $OPTIMIZED_FUNCTION 5
./scripts/benchmark-cold-starts.sh $BASE_FUNCTION 5

# Test warm invocations
echo "2. Kiểm thử Warm Invocations..."

echo "Optimized Function (Warm):"
for i in {1..5}; do
    time AWS Lambda invoke \
        --function-name $OPTIMIZED_FUNCTION \
        --payload '{"httpMethod":"GET","path":"/products"}' \
        /dev/null 2>&1 | grep real
done

echo "Base Function (Warm):"
for i in {1..5}; do
    time AWS Lambda invoke \
        --function-name $BASE_FUNCTION \
        --payload '{"httpMethod":"GET","path":"/products"}' \
        /dev/null 2>&1 | grep real
done

# Get function sizes
echo "3. Function Sizes:"
AWS Lambda get-function \
    --function-name $OPTIMIZED_FUNCTION \
    --query 'Configuration.CodeSize' \
    --output text | xargs -I {} echo "Optimized: {} bytes"

AWS Lambda get-function \
    --function-name $BASE_FUNCTION \
    --query 'Configuration.CodeSize' \
    --output text | xargs -I {} echo "Base: {} bytes"

echo "✅ Hiệu suất comparison completed!"
EOF

chmod +x scripts/compare-functions.sh

cat > scripts/analyze-metrics.sh << 'EOF'
#!/bin/bash

FUNCTION_NAME=${1:-"workshop-product-api-optimized"}
START_TIME=${2:-$(date -d '1 hour ago' --iso-8601)}
END_TIME=${3:-$(date --iso-8601)}

echo "=== Analyzing CloudWatch Metrics ==="
echo "Function: $FUNCTION_NAME"
echo "Time Range: $START_TIME to $END_TIME"

# Get Duration metrics
echo "1. Duration Metrics:"
aws CloudWatch get-metric-statistics \
    --namespace AWS/Lambda \
    --metric-name Duration \
    --dimensions Name=FunctionName,Value=$FUNCTION_NAME \
    --start-time $START_TIME \
    --end-time $END_TIME \
    --period 300 \
    --statistics Average,Maximum,Minimum \
    --query 'Datapoints[*].[Timestamp,Average,Maximum,Minimum]' \
    --output table

# Get Invocation metrics
echo "2. Invocation Metrics:"
aws CloudWatch get-metric-statistics \
    --namespace AWS/Lambda \
    --metric-name Invocations \
    --dimensions Name=FunctionName,Value=$FUNCTION_NAME \
    --start-time $START_TIME \
    --end-time $END_TIME \
    --period 300 \
    --statistics Sum \
    --query 'Datapoints[*].[Timestamp,Sum]' \
    --output table

# Get Error metrics
echo "3. Error Metrics:"
aws CloudWatch get-metric-statistics \
    --namespace AWS/Lambda \
    --metric-name Errors \
    --dimensions Name=FunctionName,Value=$FUNCTION_NAME \
    --start-time $START_TIME \
    --end-time $END_TIME \
    --period 300 \
    --statistics Sum \
    --query 'Datapoints[*].[Timestamp,Sum]' \
    --output table

# Get Throttle metrics
echo "4. Throttle Metrics:"
aws CloudWatch get-metric-statistics \
    --namespace AWS/Lambda \
    --metric-name Throttles \
    --dimensions Name=FunctionName,Value=$FUNCTION_NAME \
    --start-time $START_TIME \
    --end-time $END_TIME \
    --period 300 \
    --statistics Sum \
    --query 'Datapoints[*].[Timestamp,Sum]' \
    --output table

echo "✅ Metrics analysis completed!"
EOF

chmod +x scripts/analyze-metrics.sh

cat > scripts/generate-report.sh << 'EOF'
#!/bin/bash

echo "=== Lambda Container Hiệu suất Report ==="
echo "Generated: $(date)"
echo

# Function information
echo "## Function Information"
echo "| Function | Package Type | Memory | Timeout | Code Size |"
echo "|----------|--------------|--------|---------|-----------|"

for func in workshop-product-api-optimized workshop-product-api-base; do
    INFO=$(AWS Lambda get-function --function-name $func --query 'Configuration.[FunctionName,PackageType,MemorySize,Timeout,CodeSize]' --output text)
    echo "| $INFO |" | tr '\t' '|'
done

echo

# Cold Start analysis
echo "## Cold Start Hiệu suất"
if [ -f "tests/cold-start-workshop-product-api-optimized-*.csv" ]; then
    OPTIMIZED_FILE=$(ls tests/cold-start-workshop-product-api-optimized-*.csv | tail -1)
    OPT_AVG=$(awk -F',' 'NR>1 {sum+=$2; count++} END {print sum/count}' $OPTIMIZED_FILE)
    echo "Optimized Average Cold Start: ${OPT_AVG}ms"
fi

if [ -f "tests/cold-start-workshop-product-api-base-*.csv" ]; then
    BASE_FILE=$(ls tests/cold-start-workshop-product-api-base-*.csv | tail -1)
    BASE_AVG=$(awk -F',' 'NR>1 {sum+=$2; count++} END {print sum/count}' $BASE_FILE)
    echo "Base Average Cold Start: ${BASE_AVG}ms"
fi

echo

# Load test results
echo "## Load Test Results"
if [ -f "tests/load-test-optimized-results.json" ]; then
    echo "Optimized Function Load Test:"
    node -e "
        const fs = require('fs');
        const data = JSON.parse(fs.readFileSync('tests/load-test-optimized-results.json'));
        console.log('- Total Requests:', data.aggregate.counters['http.requests']);
        console.log('- Success Rate:', (100 - (data.aggregate.counters['http.request_rate'] || 0)).toFixed(2) + '%');
        console.log('- Average Response Time:', data.aggregate.latency.mean.toFixed(2) + 'ms');
        console.log('- 95th Percentile:', data.aggregate.latency.p95.toFixed(2) + 'ms');
    "
fi

echo "✅ Hiệu suất report generated!"
EOF

chmod +x scripts/generate-report.sh

Kết luận

Bạn đã ho à n th à nh:

  • ✅ Setup Kiểm thử t ả i (Load Kiểm thử) với Artillery
  • ✅ Benchmark Cold Start hiệu suất
  • ✅ So s á nh hiệu suất gi ữ a optimized và base functions

  • ✅ Tạo hiệu suất reports

Expected Results:

  • Optimized Container Images th ườ ng c ó Cold Start nhanh h ơ n 20-40%
  • Image size nh ỏ h ơ n 30-50%
  • Memory usage t ố i ư u h ơ n
  • Response time cải thiện trong sustained load

Tiếp theo, chúng ta sẽ t ì m hi ể u v ề các chiến lược bộ nhớ đệm (Bộ nhớ đệm) và Khởi độ ng tr ướ c (Warm-up) để cải thiện hiệu suất h ơ n n ữ a.
