# src/monitoring/custom_metrics.py
import boto3
import time
import json
from datetime import datetime
from functools import wraps

class Hiệu suấtMetrics:
    def __init__(self):
        self.CloudWatch = boto3.client('CloudWatch')
        self.namespace = 'Lambda/ContainerHiệu suất'
        self.function_name = os.environ.get('AWS_LAMBDA_FUNCTION_NAME', 'unknown')

    def put_metric(self, metric_name, value, unit='Count', dimensions=None):
        """Gửi custom metric lên CloudWatch"""
        try:
            dimensions = dimensions or []
            dimensions.append({
                'Name': 'FunctionName',
                'Value': self.function_name
            })

            self.CloudWatch.put_metric_data(
                Namespace=self.namespace,
                MetricData=[
                    {
                        'MetricName': metric_name,
                        'Value': value,
                        'Unit': unit,
                        'Dimensions': dimensions,
                        'Timestamp': datetime.utcnow()
                    }
                ]
            )
        except Exception as e:
            print(f"Không thể gửi metric {metric_name}: {e}")

    def track_cold_start(self):
        """Theo dõi Cold Start occurrence"""
        # Kiểm tra xem đây có phải Cold Start không
        if not hasattr(self, '_warm_start'):
            self.put_metric('ColdStart', 1)
            self._warm_start = True
        else:
            self.put_metric('WarmStart', 1)

    def track_memory_usage(self):
        """Theo dõi việc sử dụng bộ nhớ"""
        import psutil
        process = psutil.Process()
        memory_mb = process.memory_info().rss / 1024 / 1024

        self.put_metric('MemoryUsageMB', memory_mb, 'None')

        # Tính toán hiệu quả bộ nhớ
        allocated_mb = int(os.environ.get('AWS_LAMBDA_FUNCTION_MEMORY_SIZE', 128))
        efficiency = (memory_mb / allocated_mb) * 100
        self.put_metric('MemoryEfficiency', efficiency, 'Percent')

    def track_execution_time(self, start_time, operation_name):
        """Theo dõi thời gian thực thi operation"""
        duration_ms = (time.time() - start_time) * 1000
        self.put_metric(f'{operation_name}Duration', duration_ms, 'Milliseconds')
        return duration_ms

# Global metrics instance
metrics = Hiệu suấtMetrics()

def monitor_hiệu suất(operation_name):
    """Decorator để giám sát hiệu suất function"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            start_time = time.time()

            try:
                result = func(*args, **kwargs)
                metrics.put_metric(f'{operation_name}Success', 1)
                return result
            except Exception as e:
                metrics.put_metric(f'{operation_name}Error', 1)
                raise
            finally:
                metrics.track_execution_time(start_time, operation_name)

        return wrapper
    return decorator

import json
import os
import time
from src.monitoring.custom_metrics import metrics, monitor_hiệu suất
from src.optimizations.memory_profiler import profiler

def lambda_handler(event, context):
    """Lambda handler nâng cao với giám sát toàn diện"""
    start_time = time.time()

    # Theo dõi cold/warm start
    metrics.track_cold_start()
    profiler.checkpoint('handler_start')

    try:
        # Theo dõi việc sử dụng bộ nhớ lúc bắt đầu
        metrics.track_memory_usage()

        # Parse request
        http_method = event.get('httpMethod', 'GET')
        path = event.get('path', '/')

        profiler.checkpoint('request_parsed')

        # Route và xử lý request
        if path.startswith('/products'):
            response = handle_products_monitored(http_method, path, event.get('body'))
        elif path.startswith('/orders'):
            response = handle_orders_monitored(http_method, path, event.get('body'))
        else:
            response = {
                'statusCode': 404,
                'body': json.dumps({'error': 'Not found'})
            }

        profiler.checkpoint('request_processed')

        # Theo dõi việc sử dụng bộ nhớ cuối cùng
        metrics.track_memory_usage()

        # Thêm hiệu suất headers
        total_duration = (time.time() - start_time) * 1000
        response['headers'] = response.get('headers', {})
        response['headers'].update({
            'X-Execution-Time': f'{total_duration:.2f}ms',
            'X-Memory-Report': json.dumps(profiler.get_report()),
            'X-Function-Version': context.function_version,
            'X-Request-Id': context.aws_request_id
        })

        # Theo dõi thành công tổng thể
        metrics.put_metric('RequestSuccess', 1)
        metrics.track_execution_time(start_time, 'TotalRequest')

        return response

    except Exception as e:
        profiler.checkpoint('error_occurred')
        metrics.put_metric('RequestError', 1)

        # Theo dõi chi tiết lỗi
        error_type = type(e).__name__
        metrics.put_metric(f'Error_{error_type}', 1)

        return {
            'statusCode': 500,
            'body': json.dumps({'error': str(e)}),
            'headers': {
                'X-Execution-Time': f'{(time.time() - start_time) * 1000:.2f}ms',
                'X-Memory-Report': json.dumps(profiler.get_report())
            }
        }

# src/monitoring/xray_tracing.py
from aws_xray_sdk.core import xray_recorder, patch_all
from aws_xray_sdk.core.context import Context
import boto3

# Patch AWS SDK calls
patch_all()

class XRayTracing:
    def __init__(self):
        self.recorder = xray_recorder

    def trace_subsegment(self, name):
        """Tạo subsegment để tracing chi tiết"""
        return self.recorder.in_subsegment(name)

    def add_annotation(self, key, value):
        """Thêm annotation vào segment hiện tại"""
        try:
            self.recorder.current_subsegment().add_annotation(key, value)
        except:
            pass  # Bỏ qua nếu không có active segment

    def add_metadata(self, namespace, data):
        """Thêm metadata vào segment hiện tại"""
        try:
            self.recorder.current_subsegment().add_metadata(namespace, data)
        except:
            pass

# Global tracer instance
tracer = XRayTracing()

# Handler nâng cao với X-Ray tracing
def lambda_handler(event, context):
    """Lambda handler với X-Ray tracing"""

    # Thêm annotations để filtering
    tracer.add_annotation('function_name', context.function_name)
    tracer.add_annotation('function_version', context.function_version)
    tracer.add_annotation('memory_size', context.memory_limit_in_mb)

    # Thêm request metadata
    tracer.add_metadata('request', {
        'method': event.get('httpMethod'),
        'path': event.get('path'),
        'user_agent': event.get('headers', {}).get('User-Agent'),
        'source_ip': event.get('requestContext', {}).get('identity', {}).get('sourceIp')
    })

    with tracer.trace_subsegment('request_processing'):
        # Logic handler hiện tại của bạn
        response = process_request(event, context)

    # Thêm response metadata
    tracer.add_metadata('response', {
        'status_code': response.get('statusCode'),
        'response_size': len(response.get('body', ''))
    })

    return response

# src/monitoring/database_tracing.py
import boto3
from aws_xray_sdk.core import xray_recorder

class TracedDynamoDB:
    def __init__(self):
        self.DynamoDB = boto3.resource('DynamoDB')

    def get_item_traced(self, table_name, key):
        """Get item với X-Ray tracing"""
        with xray_recorder.in_subsegment(f'dynamodb_get_{table_name}'):
            table = self.DynamoDB.Table(table_name)

            # Thêm annotations
            xray_recorder.current_subsegment().add_annotation('table_name', table_name)
            xray_recorder.current_subsegment().add_annotation('operation', 'get_item')

            response = table.get_item(Key=key)

            # Thêm metadata
            xray_recorder.current_subsegment().add_metadata('DynamoDB', {
                'consumed_capacity': response.get('ConsumedCapacity'),
                'item_found': 'Item' in response
            })

            return response

    def scan_traced(self, table_name, **kwargs):
        """Scan table với X-Ray tracing"""
        with xray_recorder.in_subsegment(f'dynamodb_scan_{table_name}'):
            table = self.DynamoDB.Table(table_name)

            xray_recorder.current_subsegment().add_annotation('table_name', table_name)
            xray_recorder.current_subsegment().add_annotation('operation', 'scan')

            response = table.scan(**kwargs)

            xray_recorder.current_subsegment().add_metadata('DynamoDB', {
                'scanned_count': response.get('ScannedCount'),
                'item_count': response.get('Count'),
                'consumed_capacity': response.get('ConsumedCapacity')
            })

            return response

# Global traced DynamoDB instance
traced_db = TracedDynamoDB()

import json
import boto3
from datetime import datetime, timedelta

def process_cloudwatch_alarm(event, context):
    """Xử lý CloudWatch alarm và thực hiện hành động"""

    # Parse SNS message
    message = json.loads(event['Records'][0]['Sns']['Message'])
    alarm_name = message['AlarmName']
    new_state = message['NewStateValue']
    reason = message['NewStateReason']

    if new_state == 'ALARM':
        if 'Cold-Start' in alarm_name:
            handle_cold_start_alarm(message)
        elif 'Error-Rate' in alarm_name:
            handle_error_rate_alarm(message)
        elif 'Duration' in alarm_name:
            handle_duration_alarm(message)

    return {'statusCode': 200}

def handle_cold_start_alarm(alarm_data):
    """Xử lý cảnh báo tỷ lệ Cold Start cao"""
    function_name = extract_function_name(alarm_data)

    # Tăng provisioned concurrency
    lambda_client = boto3.client('lambda')

    try:
        # Lấy provisioned concurrency hiện tại
        response = lambda_client.get_provisioned_concurrency_config(
            FunctionName=function_name
        )
        current_concurrency = response['AllocatedConcurrencyExecutions']
        new_concurrency = min(current_concurrency + 10, 100)  # Giới hạn tại 100

        # Cập nhật provisioned concurrency
        lambda_client.put_provisioned_concurrency_config(
            FunctionName=function_name,
            ProvisionedConcurrencyExecutions=new_concurrency
        )

        print(f"Đã tăng provisioned concurrency cho {function_name} từ {current_concurrency} lên {new_concurrency}")

    except lambda_client.exceptions.ResourceNotFoundException:
        # Tạo provisioned concurrency nếu chưa tồn tại
        lambda_client.put_provisioned_concurrency_config(
            FunctionName=function_name,
            ProvisionedConcurrencyExecutions=10
        )
        print(f"Đã tạo provisioned concurrency cho {function_name} với 10 executions")

def handle_error_rate_alarm(alarm_data):

# tests/test_hiệu suất_monitoring.py
import boto3
import json
import time
import pytest
from concurrent.futures import ThreadPoolExecutor

class TestHiệu suấtMonitoring:
    def setup_method(self):
        self.lambda_client = boto3.client('lambda')
        self.CloudWatch = boto3.client('CloudWatch')
        self.function_name = 'workshop-product-api'

    def test_cold_start_monitoring(self):
        """Kiểm thử việc giám sát Cold Start"""
        # Buộc Cold Start
        self.lambda_client.update_function_configuration(
            FunctionName=self.function_name,
            Environment={'Variables': {'FORCE_COLD': str(time.time())}}
        )
        time.sleep(5)

        # Invoke function
        response = self.lambda_client.invoke(
            FunctionName=self.function_name,
            Payload=json.dumps({'httpMethod': 'GET', 'path': '/products'})
        )

        # Chờ metrics được publish
        time.sleep(60)

        # Kiểm tra xem Cold Start metric đã được publish chưa
        metrics = self.CloudWatch.get_metric_statistics(
            Namespace='Lambda/ContainerHiệu suất',
            MetricName='ColdStart',
            Dimensions=[{'Name': 'FunctionName', 'Value': self.function_name}],
            StartTime=datetime.utcnow() - timedelta(minutes=5),
            EndTime=datetime.utcnow(),
            Period=300,
            Statistics=['Sum']
        )

        assert len(metrics['Datapoints']) > 0, "Không tìm thấy Cold Start metric"
        assert metrics['Datapoints'][0]['Sum'] >= 1, "Cold Start không được ghi lại"

    def test_memory_monitoring(self):
        """Kiểm thử giám sát việc sử dụng bộ nhớ"""
        response = self.lambda_client.invoke(
            FunctionName=self.function_name,
            Payload=json.dumps({'httpMethod': 'GET', 'path': '/products'})
        )

        # Kiểm tra response headers cho memory report
        payload = json.loads(response['Payload'].read())
        headers = payload.get('headers', {})

        assert 'X-Memory-Report' in headers, "Không tìm thấy memory report trong headers"

        memory_report = json.loads(headers['X-Memory-Report'])
        assert 'checkpoints' in memory_report, "Không tìm thấy memory checkpoints"
        assert len(memory_report['checkpoints']) > 0, "Không có memory checkpoints được ghi lại"

    def test_concurrent_hiệu suất(self):
        """Kiểm thử hiệu suất dưới tải đồng thời"""
        def invoke_function():
            return self.lambda_client.invoke(
                FunctionName=self.function_name,
                Payload=json.dumps({'httpMethod': 'GET', 'path': '/products'})
            )

        # Kiểm thử với 10 invocations đồng thời
        with ThreadPoolExecutor(max_workers=10) as executor:
            futures = [executor.submit(invoke_function) for _ in range(10)]
            responses = [future.result() for future in futures]

        # Tất cả invocations phải thành công
        for response in responses:
            assert response['StatusCode'] == 200
            payload = json.loads(response['Payload'].read())
            assert payload['statusCode'] == 200

    def test_x_ray_tracing(self):
        """Kiểm thử tích hợp X-Ray tracing"""
        response = self.lambda_client.invoke(
            FunctionName=self.function_name,
            Payload=json.dumps({'httpMethod': 'GET', 'path': '/products'})
        )

        # Kiểm tra xem trace ID có tồn tại không
        assert 'X-Amzn-Trace-Id' in response['ResponseMetadata']['HTTPHeaders']

        # Chờ trace được xử lý
        time.sleep(30)

        # Query X-Ray cho traces
        xray = boto3.client('xray')
        traces = xray.get_trace_summaries(
            TimeRangeType='TimeRangeByStartTime',
            StartTime=datetime.utcnow() - timedelta(minutes=5),
            EndTime=datetime.utcnow(),
            FilterExpression=f'service("{self.function_name}")'
        )

        assert len(traces['TraceSummaries']) > 0, "Không tìm thấy X-Ray traces"
