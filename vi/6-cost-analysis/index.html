# src/cost_analysis/cost_tracker.py
import boto3
import json
from datetime import datetime, timedelta
from typing import Dict, List
import os

class LambdaCostTracker:
    """Track and analyze Lambda function costs"""

    def __init__(self):
        self.ce_client = boto3.client('ce')  # Cost Explorer
        self.lambda_client = boto3.client('lambda')
        self.CloudWatch = boto3.client('CloudWatch')

    def get_lambda_costs(self, function_name: str, days: int = 30) -> Dict:
        """Get Lambda function costs for specified period"""
        end_date = datetime.utcnow().date()
        start_date = end_date - timedelta(days=days)

        try:
            response = self.ce_client.get_cost_and_usage(
                TimePeriod={
                    'Start': start_date.strftime('%Y-%m-%d'),
                    'End': end_date.strftime('%Y-%m-%d')
                },
                Granularity='DAILY',
                Metrics=['BlendedCost', 'UsageQuantity'],
                GroupBy=[
                    {'Type': 'DIMENSION', 'Key': 'SERVICE'},
                    {'Type': 'DIMENSION', 'Key': 'USAGE_TYPE'}
                ],
                Filter={
                    'Dimensions': {
                        'Key': 'SERVICE',
                        'Values': ['AWS Lambda']
                    }
                }
            )

            return self._process_cost_data(response, function_name)

        except Exception as e:
            print(f"Error getting cost data: {e}")
            return {}

    def _process_cost_data(self, cost_response: Dict, function_name: str) -> Dict:
        """Process cost explorer response"""
        total_cost = 0
        daily_costs = []
        usage_breakdown = {}

        for result in cost_response['ResultsByTime']:
            date = result['TimePeriod']['Start']
            daily_cost = 0

            for group in result['Groups']:
                service = group['Keys'][0]
                usage_type = group['Keys'][1]
                cost = float(group['Metrics']['BlendedCost']['Amount'])
                usage = float(group['Metrics']['UsageQuantity']['Amount'])

                if service == 'AWS Lambda':
                    daily_cost += cost

                    if usage_type not in usage_breakdown:
                        usage_breakdown[usage_type] = {'cost': 0, 'usage': 0}

                    usage_breakdown[usage_type]['cost'] += cost
                    usage_breakdown[usage_type]['usage'] += usage

            daily_costs.append({'date': date, 'cost': daily_cost})
            total_cost += daily_cost

        return {
            'function_name': function_name,
            'total_cost': total_cost,
            'daily_costs': daily_costs,
            'usage_breakdown': usage_breakdown,
            'average_daily_cost': total_cost / len(daily_costs) if daily_costs else 0
        }

    def calculate_cost_per_invocation(self, function_name: str, days: int = 30) -> Dict:
        """Calculate cost per invocation"""
        # Get cost data
        cost_data = self.get_lambda_costs(function_name, days)

        # Get invocation count
        end_time = datetime.utcnow()
        start_time = end_time - timedelta(days=days)

        invocation_response = self.CloudWatch.get_metric_statistics(
            Namespace='AWS/Lambda',
            MetricName='Invocations',
            Dimensions=[{'Name': 'FunctionName', 'Value': function_name}],
            StartTime=start_time,
            EndTime=end_time,
            Period=86400,  # Daily
            Statistics=['Sum']
        )

        total_invocations = sum(point['Sum'] for point in invocation_response['Datapoints'])

        cost_per_invocation = cost_data['total_cost'] / total_invocations if total_invocations > 0 else 0

        return {
            'function_name': function_name,
            'period_days': days,
            'total_cost': cost_data['total_cost'],
            'total_invocations': total_invocations,
            'cost_per_invocation': cost_per_invocation,
            'cost_per_million_invocations': cost_per_invocation * 1000000
        }

    def analyze_memory_cost_efficiency(self, function_name: str) -> Dict:
        """Phân tích memory allocation cost efficiency"""
        # Get function configuration
        function_config = self.lambda_client.get_function_configuration(
            FunctionName=function_name
        )

        allocated_memory = function_config['MemorySize']

        # Get actual memory usage from CloudWatch
        end_time = datetime.utcnow()
        start_time = end_time - timedelta(days=7)

        # Get custom memory usage metrics if available
        memory_response = self.CloudWatch.get_metric_statistics(
            Namespace='Lambda/ContainerHiệu suất',
            MetricName='MemoryUsageMB',
            Dimensions=[{'Name': 'FunctionName', 'Value': function_name}],
            StartTime=start_time,
            EndTime=end_time,
            Period=3600,
            Statistics=['Average']
        )

        if memory_response['Datapoints']:
            avg_memory_used = sum(point['Average'] for point in memory_response['Datapoints']) / len(memory_response['Datapoints'])
            memory_efficiency = (avg_memory_used / allocated_memory) * 100

            # Calculate potential savings
            optimal_memory = int(avg_memory_used * 1.2)  # 20% buffer
            optimal_memory = max(128, min(optimal_memory, 10240))  # Lambda limits

            # Estimate cost savings (memory cost is proportional to allocation)
            current_cost_factor = allocated_memory / 128
            optimal_cost_factor = optimal_memory / 128
            potential_savings_percent = ((current_cost_factor - optimal_cost_factor) / current_cost_factor) * 100

            return {
                'function_name': function_name,
                'allocated_memory_mb': allocated_memory,
                'average_used_memory_mb': avg_memory_used,
                'memory_efficiency_percent': memory_efficiency,
                'recommended_memory_mb': optimal_memory,
                'potential_cost_savings_percent': potential_savings_percent
            }

        return {'error': 'No memory usage data available'}

# Cost tracking handler
def cost_analysis_handler(event, context):
    """Lambda handler for Phân tích chi phí"""
    tracker = LambdaCostTracker()
    function_name = event.get('function_name', 'workshop-product-api')

    try:
        # Perform Phân tích chi phí
        cost_data = tracker.get_lambda_costs(function_name)
        cost_per_invocation = tracker.calculate_cost_per_invocation(function_name)
        memory_efficiency = tracker.analyze_memory_cost_efficiency(function_name)

        return {
            'statusCode': 200,
            'body': json.dumps({
                'cost_analysis': cost_data,
                'cost_per_invocation': cost_per_invocation,
                'memory_efficiency': memory_efficiency
            }, default=str)
        }

    except Exception as e:
        return {
            'statusCode': 500,
            'body': json.dumps({'error': str(e)})
        }

# tests/benchmark_hiệu suất.py
import boto3
import json
import time
import statistics
import concurrent.futures
from datetime import datetime
from typing import List, Dict

class Hiệu suấtBenchmark:
    """Comprehensive hiệu suất Đánh giá hiệu suất (Benchmarking) suite"""

    def __init__(self, function_name: str):
        self.function_name = function_name
        self.lambda_client = boto3.client('lambda')
        self.results = []

    def benchmark_cold_start(self, iterations: int = 10) -> Dict:
        """Benchmark Cold Start hiệu suất"""
        cold_start_times = []

        for i in range(iterations):
            # Force Cold Start by updating environment
            self.lambda_client.update_function_configuration(
                FunctionName=self.function_name,
                Environment={
                    'Variables': {
                        'FORCE_COLD_START': str(time.time())
                    }
                }
            )

            # Wait for update to complete
            time.sleep(5)

            # Measure invocation time
            start_time = time.time()

            response = self.lambda_client.invoke(
                FunctionName=self.function_name,
                Payload=json.dumps({
                    'httpMethod': 'GET',
                    'path': '/products',
                    'benchmark': True
                })
            )

            end_time = time.time()
            duration_ms = (end_time - start_time) * 1000
            cold_start_times.append(duration_ms)

            print(f"Cold Start {i+1}: {duration_ms:.2f}ms")

        return {
            'test_type': 'cold_start',
            'iterations': iterations,
            'times_ms': cold_start_times,
            'average_ms': statistics.mean(cold_start_times),
            'median_ms': statistics.median(cold_start_times),
            'min_ms': min(cold_start_times),
            'max_ms': max(cold_start_times),
            'std_dev_ms': statistics.stdev(cold_start_times) if len(cold_start_times) > 1 else 0
        }

    def benchmark_warm_execution(self, iterations: int = 50) -> Dict:
        """Benchmark warm execution hiệu suất"""
        warm_times = []

        # First invocation to warm up
        self.lambda_client.invoke(
            FunctionName=self.function_name,
            Payload=json.dumps({'httpMethod': 'GET', 'path': '/products'})
        )

        time.sleep(2)  # Ensure warm state

        for i in range(iterations):
            start_time = time.time()

            response = self.lambda_client.invoke(
                FunctionName=self.function_name,
                Payload=json.dumps({
                    'httpMethod': 'GET',
                    'path': '/products',
                    'benchmark': True
                })
            )

            end_time = time.time()
            duration_ms = (end_time - start_time) * 1000
            warm_times.append(duration_ms)

            # Small delay to avoid throttling
            time.sleep(0.1)

        return {
            'test_type': 'warm_execution',
            'iterations': iterations,
            'times_ms': warm_times,
            'average_ms': statistics.mean(warm_times),
            'median_ms': statistics.median(warm_times),
            'min_ms': min(warm_times),
            'max_ms': max(warm_times),
            'std_dev_ms': statistics.stdev(warm_times) if len(warm_times) > 1 else 0
        }

    def benchmark_concurrent_execution(self, concurrency: int = 10, iterations: int = 5) -> Dict:
        """Benchmark concurrent execution hiệu suất"""
        concurrent_results = []

        for iteration in range(iterations):
            def invoke_function():
                start_time = time.time()
                response = self.lambda_client.invoke(
                    FunctionName=self.function_name,
                    Payload=json.dumps({
                        'httpMethod': 'GET',
                        'path': '/products',
                        'benchmark': True
                    })
                )
                end_time = time.time()
                return (end_time - start_time) * 1000

            # Execute concurrent invocations
            with concurrent.futures.ThreadPoolExecutor(max_workers=concurrency) as executor:
                futures = [executor.submit(invoke_function) for _ in range(concurrency)]
                iteration_times = [future.result() for future in futures]

            concurrent_results.extend(iteration_times)
            print(f"Concurrent iteration {iteration + 1}: avg {statistics.mean(iteration_times):.2f}ms")

        return {
            'test_type': 'concurrent_execution',
            'concurrency': concurrency,
            'iterations': iterations,
            'total_invocations': len(concurrent_results),
            'times_ms': concurrent_results,
            'average_ms': statistics.mean(concurrent_results),
            'median_ms': statistics.median(concurrent_results),
            'min_ms': min(concurrent_results),
            'max_ms': max(concurrent_results),
            'std_dev_ms': statistics.stdev(concurrent_results) if len(concurrent_results) > 1 else 0
        }

    def benchmark_memory_usage(self, test_payloads: List[Dict]) -> Dict:
        """Benchmark memory usage with different payloads"""
        memory_results = []

        for i, payload in enumerate(test_payloads):
            response = self.lambda_client.invoke(
                FunctionName=self.function_name,
                Payload=json.dumps(payload)
            )

            # Parse response to get memory report
            response_payload = json.loads(response['Payload'].read())
            headers = response_payload.get('headers', {})

            if 'X-Memory-Report' in headers:
                memory_report = json.loads(headers['X-Memory-Report'])
                peak_memory = memory_report.get('peak_memory_mb', 0)
                memory_results.append({
                    'payload_size': len(json.dumps(payload)),
                    'peak_memory_mb': peak_memory,
                    'payload_type': payload.get('test_type', f'test_{i}')
                })

        return {
            'test_type': 'memory_usage',
            'results': memory_results,
            'average_memory_mb': statistics.mean([r['peak_memory_mb'] for r in memory_results]),
            'max_memory_mb': max([r['peak_memory_mb'] for r in memory_results]),
            'min_memory_mb': min([r['peak_memory_mb'] for r in memory_results])
        }

    def run_comprehensive_benchmark(self) -> Dict:
        """Run comprehensive benchmark suite"""
        print(f"Starting comprehensive benchmark for {self.function_name}")

        results = {
            'function_name': self.function_name,
            'timestamp': datetime.utcnow().isoformat(),
            'benchmarks': {}
        }

        # Cold Start benchmark
        print("Running Cold Start benchmark...")
        results['benchmarks']['cold_start'] = self.benchmark_cold_start()

        # Warm execution benchmark
        print("Running warm execution benchmark...")
        results['benchmarks']['warm_execution'] = self.benchmark_warm_execution()

        # Concurrent execution benchmark
        print("Running concurrent execution benchmark...")
        results['benchmarks']['concurrent_execution'] = self.benchmark_concurrent_execution()

        # Memory usage benchmark
        print("Running memory usage benchmark...")
        test_payloads = [
            {'httpMethod': 'GET', 'path': '/products', 'test_type': 'simple_get'},
            {'httpMethod': 'POST', 'path': '/products', 'body': json.dumps({'name': 'Kiểm thử Product', 'price': 99.99}), 'test_type': 'simple_post'},
            {'httpMethod': 'GET', 'path': '/products', 'queryStringParameters': {'limit': '100'}, 'test_type': 'large_query'}
        ]
        results['benchmarks']['memory_usage'] = self.benchmark_memory_usage(test_payloads)

        return results

# Benchmark execution script
def run_benchmark():
    """Run benchmark and save results"""
    function_name = 'workshop-product-api'
    benchmark = Hiệu suấtBenchmark(function_name)

    results = benchmark.run_comprehensive_benchmark()

    # Save results to file
    timestamp = datetime.utcnow().strftime('%Y%m%d_%H%M%S')
    filename = f'benchmark_results_{timestamp}.json'

    with open(filename, 'w') as f:
        json.dump(results, f, indent=2)

    print(f"Benchmark results saved to {filename}")

    # Print summary
    print("\n=== BENCHMARK SUMMARY ===")
    for test_name, test_results in results['benchmarks'].items():
        print(f"\n{test_name.upper()}:")
        print(f"  Average: {test_results.get('average_ms', 0):.2f}ms")
        print(f"  Median: {test_results.get('median_ms', 0):.2f}ms")
        print(f"  Min: {test_results.get('min_ms', 0):.2f}ms")
        print(f"  Max: {test_results.get('max_ms', 0):.2f}ms")

if __name__ == "__main__":
    run_benchmark()
