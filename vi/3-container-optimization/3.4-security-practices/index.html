# Build stage với minimal tools
FROM public.ecr.aws/lambda/python:3.11 as builder

# Create non-root user
RUN groupadd -r appgroup && useradd -r -g appgroup appuser

# Install only necessary build dependencies
RUN yum update -y && \
    yum install -y gcc python3-devel && \
    yum clean all && \
    rm -rf /var/cache/yum

# Copy requirements và install
COPY requirements.txt .
RUN pip install --no-cache-dir --target /opt/python -r requirements.txt

# Remove unnecessary files và potential Bảo mật risks
RUN find /opt/python -name "*.pyc" -delete && \
    find /opt/python -name "__pycache__" -type d -exec rm -rf {} + && \
    find /opt/python -name "*.so" -exec strip {} \; 2>/dev/null || true && \
    find /opt/python -name "test*" -type d -exec rm -rf {} + 2>/dev/null || true

# Môi trường sản xuất stage
FROM public.ecr.aws/lambda/python:3.11

# Copy dependencies from builder
COPY --from=builder /opt/python ${LAMBDA_TASK_ROOT}/

# Copy application code với proper ownership
COPY --chown=nobody:nobody lambda_function.py ${LAMBDA_TASK_ROOT}/
COPY --chown=nobody:nobody src/ ${LAMBDA_TASK_ROOT}/src/

# Set Bảo mật-focused environment variables
ENV PYTHONPATH="${LAMBDA_TASK_ROOT}/src:${PYTHONPATH}"
ENV PYTHONDONTWRITEBYTECODE=1
ENV PYTHONUNBUFFERED=1
ENV PYTHONHASHSEED=random

# Remove potential Bảo mật risks
RUN rm -rf /tmp/* /var/tmp/* && \
    find ${LAMBDA_TASK_ROOT} -name "*.py" -exec chmod 644 {} \; && \
    find ${LAMBDA_TASK_ROOT} -type d -exec chmod 755 {} \;

# Health check
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
    CMD python -c "import lambda_function; print('OK')" || exit 1

CMD ["lambda_function.lambda_handler"]
EOF

cat > src/utils/secrets_manager.py << 'EOF'
import boto3
import json
from functools import lru_cache
import os

class SecretsManager:
    def __init__(self):
        self.client = boto3.client('secretsmanager')

    @lru_cache(maxsize=10)
    def get_secret(self, secret_name: str) -> dict:
        """Get secret from AWS Secrets Manager with Bộ nhớ đệm (Caching)"""
        try:
            response = self.client.get_secret_value(SecretId=secret_name)
            return json.loads(response['SecretString'])
        except Exception as e:
            print(f"Error retrieving secret {secret_name}: {e}")
            return {}

    def get_database_config(self) -> dict:
        """Get database configuration from secrets"""
        secret_name = os.environ.get('DB_SECRET_NAME', 'lambda-workshop/database')
        return self.get_secret(secret_name)

# Global instance
secrets_manager = SecretsManager()
EOF

cat > src/utils/validators.py << 'EOF'
import re
from typing import Any, Dict, List
import bleach

class InputValidator:
    """Input validation utilities"""

    @staticmethod
    def validate_product_data(data: Dict[str, Any]) -> Dict[str, Any]:
        """Validate product data input"""
        errors = []

        # Name validation
        if 'name' not in data or not data['name']:
            errors.append("Product name is required")
        elif len(data['name']) > 100:
            errors.append("Product name too long (max 100 characters)")
        else:
            # Sanitize HTML
            data['name'] = bleach.clean(data['name'])

        # Price validation
        if 'price' not in data:
            errors.append("Price is required")
        else:
            try:
                price = float(data['price'])
                if price < 0:
                    errors.append("Price must be positive")
                elif price > 999999:
                    errors.append("Price too high")
                data['price'] = price
            except (ValueError, TypeError):
                errors.append("Invalid price format")

        # Category validation
        if 'category' in data:
            if len(data['category']) > 50:
                errors.append("Category name too long")
            else:
                data['category'] = bleach.clean(data['category'])

        # Description validation
        if 'description' in data:
            if len(data['description']) > 1000:
                errors.append("Description too long")
            else:
                data['description'] = bleach.clean(data['description'])

        if errors:
            raise ValueError(f"Validation errors: {', '.join(errors)}")

        return data

    @staticmethod
    def validate_id(product_id: str) -> str:
        """Validate product ID"""
        if not product_id:
            raise ValueError("Product ID is required")

        # Check for SQL injection patterns
        if re.search(r'[;\'"\\]', product_id):
            raise ValueError("Invalid product ID format")

        if len(product_id) > 50:
            raise ValueError("Product ID too long")

        return product_id.strip()

# Update requirements.txt
echo "bleach==6.1.0" >> requirements.txt
EOF

cat > src/utils/security_logger.py << 'EOF'
import json
import logging
from datetime import datetime
from typing import Dict, Any

class SecurityLogger:
    """Bảo mật-focused logging"""

    def __init__(self):
        self.logger = logging.getLogger('Bảo mật')
        self.logger.setLevel(logging.INFO)

    def log_security_event(self, event_type: str, details: Dict[str, Any],
                          severity: str = 'INFO'):
        """Log Bảo mật events"""
        event = {
            'timestamp': datetime.utcnow().isoformat(),
            'event_type': event_type,
            'severity': severity,
            'details': self._sanitize_details(details)
        }

        if severity == 'CRITICAL':
            self.logger.critical(json.dumps(event))
        elif severity == 'WARNING':
            self.logger.warning(json.dumps(event))
        else:
            self.logger.info(json.dumps(event))

    def log_access_attempt(self, method: str, path: str, ip: str = None):
        """Log API access attempts"""
        self.log_security_event('api_access', {
            'method': method,
            'path': path,
            'source_ip': ip
        })

    def log_validation_error(self, error: str, input_data: Dict[str, Any]):
        """Log input validation errors"""
        self.log_security_event('validation_error', {
            'error': error,
            'input_size': len(str(input_data))
        }, 'WARNING')

    def _sanitize_details(self, details: Dict[str, Any]) -> Dict[str, Any]:
        """Remove sensitive information from logs"""
        sanitized = {}
        sensitive_keys = ['password', 'token', 'key', 'secret']

        for key, value in details.items():
            if any(sensitive in key.lower() for sensitive in sensitive_keys):
                sanitized[key] = '[REDACTED]'
            else:
                sanitized[key] = value

        return sanitized

# Global instance
security_logger = SecurityLogger()
EOF

#!/bin/bash

FUNCTION_NAME=${1:-"workshop-product-api-optimized"}

echo "=== Bảo mật Kiểm thử for $FUNCTION_NAME ==="

# Test 1: SQL Injection attempts
echo "1. Kiểm thử SQL Injection protection..."
AWS Lambda invoke \
    --function-name $FUNCTION_NAME \
    --payload '{
        "httpMethod":"GET",
        "path":"/products/1; DROP TABLE products;--",
        "pathParameters":{"id":"1; DROP TABLE products;--"}
    }' \
    response.json

echo "Response:"
cat response.json | jq .
echo

# Test 2: XSS attempts
echo "2. Kiểm thử XSS protection..."
AWS Lambda invoke \
    --function-name $FUNCTION_NAME \
    --payload '{
        "httpMethod":"POST",
        "path":"/products",
        "body":"{\"name\":\"<script>alert(1)</script>\",\"price\":99.99,\"category\":\"Test\"}"
    }' \
    response.json

echo "Response:"
cat response.json | jq .
echo

# Test 3: Large payload
echo "3. Kiểm thử large payload handling..."
LARGE_PAYLOAD=$(python3 -c "print('A' * 10000)")
AWS Lambda invoke \
    --function-name $FUNCTION_NAME \
    --payload "{
        \"httpMethod\":\"POST\",
        \"path\":\"/products\",
        \"body\":\"{\\\"name\\\":\\\"$LARGE_PAYLOAD\\\",\\\"price\\\":99.99}\"
    }" \
    response.json

echo "Response:"
cat response.json | jq .
echo

# Dọn dẹp
rm -f response.json

echo "✅ Bảo mật tests completed!"
EOF

cat > scripts/setup-Bảo mật-alarms.sh << 'EOF'
#!/bin/bash

FUNCTION_NAME="workshop-product-api-optimized"
SNS_TOPIC_ARN="arn:aws:sns:us-east-1:$(aws sts get-caller-identity --query Account --output text):Bảo mật-alerts"

# Create SNS topic for Bảo mật alerts
aws sns create-topic --name Bảo mật-alerts

# High error rate alarm
aws CloudWatch put-metric-alarm \
    --alarm-name "Bảo mật-${FUNCTION_NAME}-HighErrorRate" \
    --alarm-description "High error rate detected" \
    --metric-name Errors \
    --namespace AWS/Lambda \
    --statistic Sum \
    --period 300 \
    --threshold 10 \
    --comparison-operator GreaterThanThreshold \
    --dimensions Name=FunctionName,Value=$FUNCTION_NAME \
    --evaluation-periods 2 \
    --alarm-actions $SNS_TOPIC_ARN

# Unusual invocation pattern
aws CloudWatch put-metric-alarm \
    --alarm-name "Bảo mật-${FUNCTION_NAME}-UnusualInvocations" \
    --alarm-description "Unusual invocation pattern detected" \
    --metric-name Invocations \
    --namespace AWS/Lambda \
    --statistic Sum \
    --period 300 \
    --threshold 1000 \
    --comparison-operator GreaterThanThreshold \
    --dimensions Name=FunctionName,Value=$FUNCTION_NAME \
    --evaluation-periods 1 \
    --alarm-actions $SNS_TOPIC_ARN

echo "✅ Bảo mật alarms created!"
EOF

chmod +x scripts/setup-Bảo mật-alarms.sh
./scripts/setup-Bảo mật-alarms.sh

  • Review IAM permissions quarterly
