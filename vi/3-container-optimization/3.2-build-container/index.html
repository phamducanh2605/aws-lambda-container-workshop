cat > docker/optimized/Dockerfile << 'EOF'
# Multi-stage build để tối ưu hóa size
FROM public.ecr.aws/lambda/python:3.11 as builder

# Install build dependencies
RUN yum update -y && \
    yum install -y gcc python3-devel && \
    yum clean all

# Copy requirements và install vào thư mục riêng
COPY requirements.txt .
RUN pip install --no-cache-dir --target /opt/python -r requirements.txt

# Remove unnecessary files để giảm size
RUN find /opt/python -name "*.pyc" -delete && \
    find /opt/python -name "__pycache__" -type d -exec rm -rf {} + && \
    find /opt/python -name "*.so" -exec strip {} \; 2>/dev/null || true

# Production environment stage
FROM public.ecr.aws/lambda/python:3.11

# Copy optimized dependencies
COPY --from=builder /opt/python ${LAMBDA_TASK_ROOT}/

# Copy application code
COPY lambda_function.py ${LAMBDA_TASK_ROOT}/
COPY src/ ${LAMBDA_TASK_ROOT}/src/

# Set environment variables for optimization
ENV PYTHONPATH="${LAMBDA_TASK_ROOT}/src:${PYTHONPATH}"
ENV PYTHONDONTWRITEBYTECODE=1
ENV PYTHONUNBUFFERED=1
ENV PYTHONHASHSEED=0

# Pre-compile Python bytecode
RUN python -m compileall -b ${LAMBDA_TASK_ROOT} && \
    find ${LAMBDA_TASK_ROOT} -name "*.py" -not -path "*/tests/*" -delete

# Set the CMD to your handler
CMD ["lambda_function.lambda_handler"]
EOF

cat > scripts/build-base.sh << 'EOF'
#!/bin/bash

set -e

echo "=== Building Base Container Image ==="

# Load configuration
ECR_REGISTRY=$(grep -o '"registry": "[^"]*' config.json | cut -d'"' -f4)
REPOSITORY="${ECR_REGISTRY}/lambda-workshop/product-api"
TAG="base-$(date +%Y%m%d-%H%M%S)"

echo "Building image: $REPOSITORY:$TAG"

# Build image
docker build \
    -f docker/base/Dockerfile \
    -t $REPOSITORY:$TAG \
    -t $REPOSITORY:base-latest \
    .

echo "✅ Build completed: $REPOSITORY:$TAG"

# Show image size
echo "Image size:"
docker images $REPOSITORY:$TAG --format "table {{.Repository}}:{{.Tag}}\t{{.Size}}"

echo "Image built successfully!"
echo "Tag: $TAG"
EOF

chmod +x scripts/build-base.sh

#!/bin/bash

set -e

echo "=== Building Optimized Container Image ==="

# Load configuration
ECR_REGISTRY=$(grep -o '"registry": "[^"]*' config.json | cut -d'"' -f4)
REPOSITORY="${ECR_REGISTRY}/lambda-workshop/product-api"

# Build image với BuildKit để tối ưu hóa
DOCKER_BUILDKIT=1 docker build \
    -f docker/optimized/Dockerfile \
    -t $REPOSITORY:$TAG \
    -t $REPOSITORY:optimized-latest \
    .

echo "✅ Build completed: $REPOSITORY:$TAG"

# Show image size
echo "Image size:"
docker images $REPOSITORY:$TAG --format "table {{.Repository}}:{{.Tag}}\t{{.Size}}"

echo "Optimized image built successfully!"
echo "Tag: $TAG"
EOF

cat > scripts/push-images.sh << 'EOF'
#!/bin/bash

set -e

IMAGE_TAG=${1:-"latest"}
IMAGE_TYPE=${2:-"optimized"}

if [ "$IMAGE_TYPE" != "base" ] && [ "$IMAGE_TYPE" != "optimized" ]; then
    echo "Usage: $0 [tag] [base|optimized]"
    echo "Example: $0 latest optimized"
    exit 1
fi

echo "=== Pushing Container Images to ECR ==="

# Load configuration
ECR_REGISTRY=$(grep -o '"registry": "[^"]*' config.json | cut -d'"' -f4)
REPOSITORY="${ECR_REGISTRY}/lambda-workshop/product-api"

# Login to ECR
echo "Logging in to ECR..."
aws ecr get-login-password --region us-east-1 | \
    docker login --username AWS --password-stdin $ECR_REGISTRY

# Push specific tag
if [ "$IMAGE_TAG" != "latest" ]; then
    echo "Pushing $REPOSITORY:${IMAGE_TYPE}-${IMAGE_TAG}"
    docker push $REPOSITORY:${IMAGE_TYPE}-${IMAGE_TAG}
fi

# Push latest tag
echo "Pushing $REPOSITORY:${IMAGE_TYPE}-latest"
docker push $REPOSITORY:${IMAGE_TYPE}-latest

echo "✅ Images pushed successfully!"

# Show pushed images
echo "Pushed images:"
aws ecr list-images \
    --repository-name lambda-workshop/product-api \
    --query 'imageIds[*].imageTag' \
    --output table
EOF

chmod +x scripts/push-images.sh

cat > scripts/test-container.sh << 'EOF'
#!/bin/bash

IMAGE_TYPE=${1:-"optimized"}
ECR_REGISTRY=$(grep -o '"registry": "[^"]*' config.json | cut -d'"' -f4)
IMAGE="${ECR_REGISTRY}/lambda-workshop/product-api:${IMAGE_TYPE}-latest"

echo "Kiểm thử container: $IMAGE"

# Start container in background
CONTAINER_ID=$(docker run -d -p 9000:8080 $IMAGE)
echo "Started container: $CONTAINER_ID"

# Wait for container to start
sleep 3

# Test health check
echo "Kiểm thử health check..."
RESPONSE=$(curl -s -XPOST "http://localhost:9000/2015-03-31/functions/function/invocations" \
    -d '{"httpMethod":"GET","path":"/health"}')
echo "Health check response: $RESPONSE"

# Test list products
echo "Kiểm thử list products..."
RESPONSE=$(curl -s -XPOST "http://localhost:9000/2015-03-31/functions/function/invocations" \
    -d '{"httpMethod":"GET","path":"/products"}')
echo "List products response: $RESPONSE"

# Stop container
docker stop $CONTAINER_ID
docker rm $CONTAINER_ID

echo "✅ Container test completed"
EOF

chmod +x scripts/test-container.sh

Kết luận

Bạn đã ho à n th à nh:

  • ✅ Tạo Dockerfile c ơ b ả n và t ố i ư u h ó a
  • ✅ Build Container Images với multi-stage builds
  • ✅ Push images l ê n Amazon ECR
  • ✅ Test containers locally
  • ✅ So s á nh k í ch th ướ c images

Image t ố i ư u h ó a th ườ ng nh ỏ h ơ n 30-50% so với base image nh ờ các k ỹ thuật:

  • Multi-stage builds
  • Loại b ỏ build dependencies
  • Pre-compile Python bytecode
  • Strip binary files

Tiếp theo, chúng ta sẽ tạo Lambda functions từcác Container Images này.
