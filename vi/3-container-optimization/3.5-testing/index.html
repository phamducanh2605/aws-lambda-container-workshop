import json
import pytest
from unittest.mock import patch, MagicMock
import lambda_function

class TestLambdaHandler:
    """Test cases for Lambda handler"""

    def test_health_check(self):
        """Test health check endpoint"""
        event = {
            'httpMethod': 'GET',
            'path': '/health'
        }

        response = lambda_function.lambda_handler(event, None)

        assert response['statusCode'] == 200
        body = json.loads(response['body'])
        assert body['status'] == 'healthy'
        assert 'timestamp' in body
        assert 'version' in body

    def test_cors_headers(self):
        """Test CORS headers are present"""
        event = {
            'httpMethod': 'GET',
            'path': '/health'
        }

        response = lambda_function.lambda_handler(event, None)

        assert 'headers' in response
        headers = response['headers']
        assert headers['Access-Control-Allow-Origin'] == '*'
        assert 'Access-Control-Allow-Methods' in headers
        assert 'Access-Control-Allow-Headers' in headers

    def test_invalid_path(self):
        """Test invalid path returns 404"""
        event = {
            'httpMethod': 'GET',
            'path': '/invalid'
        }

        response = lambda_function.lambda_handler(event, None)

        assert response['statusCode'] == 404
        body = json.loads(response['body'])
        assert body['error'] == 'Not Found'

    @patch('lambda_function.product_service')
    def test_list_products(self, mock_service):
        """Test list products endpoint"""
        # Mock service response
        mock_service.list_products.return_value = {
            'products': [{'id': '1', 'name': 'Test Product'}],
            'total': 1
        }

        event = {
            'httpMethod': 'GET',
            'path': '/products',
            'queryStringParameters': {'limit': '10'}
        }

        response = lambda_function.lambda_handler(event, None)

        assert response['statusCode'] == 200
        mock_service.list_products.assert_called_once_with(10, 0)

    @patch('lambda_function.product_service')
    def test_get_product_found(self, mock_service):
        """Test get single product - found"""
        mock_service.get_product.return_value = {
            'id': '1',
            'name': 'Test Product'
        }

        event = {
            'httpMethod': 'GET',
            'path': '/products/1',
            'pathParameters': {'id': '1'}
        }

        response = lambda_function.lambda_handler(event, None)

        assert response['statusCode'] == 200
        mock_service.get_product.assert_called_once_with('1')

    @patch('lambda_function.product_service')
    def test_get_product_not_found(self, mock_service):
        """Test get single product - not found"""
        mock_service.get_product.return_value = None

        event = {
            'httpMethod': 'GET',
            'path': '/products/999',
            'pathParameters': {'id': '999'}
        }

        response = lambda_function.lambda_handler(event, None)

        assert response['statusCode'] == 404
        body = json.loads(response['body'])
        assert body['error'] == 'Product not found'

    @patch('lambda_function.product_service')
    def test_create_product(self, mock_service):
        """Test create product endpoint"""
        mock_service.create_product.return_value = {
            'id': '123',
            'name': 'New Product',
            'price': 99.99
        }

        event = {
            'httpMethod': 'POST',
            'path': '/products',
            'body': json.dumps({
                'name': 'New Product',
                'price': 99.99,
                'category': 'Test'
            })
        }

        response = lambda_function.lambda_handler(event, None)

        assert response['statusCode'] == 201
        mock_service.create_product.assert_called_once()

    def test_exception_handling(self):
        """Test exception handling returns 500"""
        with patch('lambda_function.product_service') as mock_service:
            mock_service.list_products.side_effect = Exception("Database error")

            event = {
                'httpMethod': 'GET',
                'path': '/products'
            }

            response = lambda_function.lambda_handler(event, None)

            assert response['statusCode'] == 500
            body = json.loads(response['body'])
            assert body['error'] == 'Internal Server Error'

        assert 'offset' in result
        assert 'has_more' in result

        assert product['name'] == update_data['name']
        assert product['price'] == update_data['price']

import pytest
from unittest.mock import patch, MagicMock
from src.monitoring.hiệu suất_metrics import Hiệu suấtMetrics

class TestHiệu suấtMetrics:
    """Test cases for Hiệu suất Metrics"""

    def setup_method(self):
        """Setup test fixtures"""
        with patch('boto3.client'):
            self.metrics = Hiệu suấtMetrics()

    @patch('boto3.client')
    def test_init(self, mock_boto_client):
        """Test metrics initialization"""
        metrics = Hiệu suấtMetrics()

        assert metrics.namespace == 'Lambda/ContainerHiệu suất'
        mock_boto_client.assert_called_with('CloudWatch')

    def test_record_request(self):
        """Test record request metrics"""
        with patch.object(self.metrics, 'put_metric') as mock_put_metric:
            self.metrics.record_request('GET', '/products', 200, 150.5)

            # Should call put_metric multiple times
            assert mock_put_metric.call_count >= 2

            # Check ResponseTime metric
            calls = mock_put_metric.call_args_list
            response_time_call = next(
                call for call in calls
                if call[0][0] == 'ResponseTime'
            )
            assert response_time_call[0][1] == 150.5
            assert response_time_call[0][2] == 'Milliseconds'

    def test_record_request_with_error(self):
        """Test record request with error status"""
        with patch.object(self.metrics, 'put_metric') as mock_put_metric:
            self.metrics.record_request('POST', '/products', 500, 200.0)

            # Should record error metric for 5xx status
            error_calls = [
                call for call in mock_put_metric.call_args_list
                if call[0][0] == 'ErrorCount'
            ]
            assert len(error_calls) > 0

    def test_put_metric(self):
        """Test put metric to CloudWatch"""
        with patch.object(self.metrics.CloudWatch, 'put_metric_data') as mock_put:
            dimensions = [{'Name': 'Method', 'Value': 'GET'}]

            self.metrics.put_metric('TestMetric', 100, 'Count', dimensions)

            mock_put.assert_called_once()
            call_args = mock_put.call_args[1]

            assert call_args['Namespace'] == 'Lambda/ContainerHiệu suất'
            assert len(call_args['MetricData']) == 1

            metric_data = call_args['MetricData'][0]
            assert metric_data['MetricName'] == 'TestMetric'
            assert metric_data['Value'] == 100
            assert metric_data['Unit'] == 'Count'

    def test_put_metric_exception_handling(self):
        """Test put metric handles exceptions gracefully"""
        with patch.object(self.metrics.CloudWatch, 'put_metric_data') as mock_put:
            mock_put.side_effect = Exception("CloudWatch error")

            # Should not raise exception
            try:
                self.metrics.put_metric('TestMetric', 100, 'Count', [])
            except Exception:
                pytest.fail("put_metric should handle exceptions gracefully")

import time
import pytest
import statistics
import lambda_function

@pytest.mark.hiệu suất
class TestHiệu suất:
    """Hiệu suất tests for Lambda function"""

    def test_cold_start_simulation(self):
        """Test Cold Start hiệu suất"""
        # Simulate Cold Start by creating new handler instance
        start_time = time.time()

        event = {
            'httpMethod': 'GET',
            'path': '/health'
        }

        response = lambda_function.lambda_handler(event, None)

        duration = (time.time() - start_time) * 1000  # Convert to ms

        assert response['statusCode'] == 200
        assert duration < 1000  # Should be under 1 second
        print(f"Cold Start duration: {duration:.2f}ms")

    def test_warm_invocation_hiệu suất(self):
        """Test warm invocation hiệu suất"""
        durations = []

        # Warm up
        event = {
            'httpMethod': 'GET',
            'path': '/products'
        }
        lambda_function.lambda_handler(event, None)

        # Measure warm invocations
        for _ in range(10):
            start_time = time.time()
            response = lambda_function.lambda_handler(event, None)
            duration = (time.time() - start_time) * 1000

            assert response['statusCode'] == 200
            durations.append(duration)

        avg_duration = statistics.mean(durations)
        p95_duration = statistics.quantiles(durations, n=20)[18]  # 95th percentile

        assert avg_duration < 100  # Average should be under 100ms
        assert p95_duration < 200  # 95th percentile under 200ms

        print(f"Warm invocation - Avg: {avg_duration:.2f}ms, P95: {p95_duration:.2f}ms")

    def test_memory_usage(self):
        """Test memory usage patterns"""
        import psutil
        import os

        process = psutil.Process(os.getpid())
        initial_memory = process.memory_info().rss

        # Run multiple operations
        for i in range(100):
            event = {
                'httpMethod': 'GET',
                'path': '/products',
                'queryStringParameters': {'limit': '10'}
            }
            lambda_function.lambda_handler(event, None)

        final_memory = process.memory_info().rss
        memory_increase = final_memory - initial_memory

        # Memory increase should be reasonable
        assert memory_increase < 50 * 1024 * 1024  # Less than 50MB increase

        print(f"Memory increase: {memory_increase / 1024 / 1024:.2f}MB")

    @pytest.mark.slow
    def test_sustained_load(self):
        """Test hiệu suất under sustained load"""
        durations = []
        errors = 0

        event = {
            'httpMethod': 'GET',
            'path': '/products'
        }

        # Run for 30 seconds
        end_time = time.time() + 30

        while time.time() < end_time:
            start_time = time.time()

            try:
                response = lambda_function.lambda_handler(event, None)
                duration = (time.time() - start_time) * 1000
                durations.append(duration)

                if response['statusCode'] != 200:
                    errors += 1

            except Exception:
                errors += 1

            time.sleep(0.1)  # 10 RPS

        total_requests = len(durations)
        error_rate = errors / total_requests if total_requests > 0 else 0
        avg_duration = statistics.mean(durations) if durations else 0

        assert error_rate < 0.01  # Less than 1% error rate
        assert avg_duration < 150  # Average under 150ms

        print(f"Sustained load - Requests: {total_requests}, "
              f"Error rate: {error_rate:.2%}, Avg duration: {avg_duration:.2f}ms")

cat > scripts/run-tests.sh << 'EOF'
#!/bin/bash

TEST_TYPE=${1:-"all"}

echo "=== Running Tests: $TEST_TYPE ==="

case $TEST_TYPE in
    "unit")
        echo "Running unit tests..."
        python -m pytest tests/unit/ -v -m "not slow"
        ;;
    "integration")
        echo "Running integration tests..."
        python -m pytest tests/integration/ -v
        ;;
    "hiệu suất")
        echo "Running hiệu suất tests..."
        python -m pytest tests/hiệu suất/ -v -m "hiệu suất"
        ;;
    "all")
        echo "Running all tests..."
        python -m pytest tests/ -v -m "not slow"
        ;;
    "full")
        echo "Running full test suite (including slow tests)..."
        python -m pytest tests/ -v
        ;;
    *)
        echo "Usage: $0 [unit|integration|hiệu suất|all|full]"
        exit 1
        ;;
esac

echo "✅ Tests completed!"
EOF

cat > scripts/coverage-report.sh << 'EOF'
#!/bin/bash

echo "=== Generating Coverage Report ==="

# Run tests with coverage
python -m pytest tests/ --cov=src --cov=lambda_function \
    --cov-report=html:htmlcov \
    --cov-report=xml:coverage.xml \
    --cov-report=term-missing

echo "Coverage reports generated:"
echo "- HTML: htmlcov/index.html"
echo "- XML: coverage.xml"
echo "- Terminal output above"

# Check coverage threshold
COVERAGE=$(python -c "
import xml.etree.ElementTree as ET
tree = ET.parse('coverage.xml')
root = tree.getroot()
print(float(root.attrib['line-rate']) * 100)
")

echo "Current coverage: ${COVERAGE}%"

if (( $(echo "$COVERAGE < 80" | bc -l) )); then
    echo "❌ Coverage below 80% threshold"
    exit 1
else
    echo "✅ Coverage meets 80% threshold"
fi
EOF
