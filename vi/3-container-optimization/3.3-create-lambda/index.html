cat > scripts/create-lambda.sh << 'EOF'
#!/bin/bash

set -e

FUNCTION_NAME=${1:-"workshop-product-api"}
IMAGE_TYPE=${2:-"optimized"}
MEMORY_SIZE=${3:-512}

if [ -z "$FUNCTION_NAME" ]; then
    echo "Usage: $0 <function-name> [base|optimized] [memory-size]"
    echo "Example: $0 workshop-product-api optimized 512"
    exit 1
fi

echo "=== Creating Lambda Function ==="
echo "Function Name: $FUNCTION_NAME"
echo "Image Type: $IMAGE_TYPE"
echo "Memory Size: ${MEMORY_SIZE}MB"

# Load configuration
ECR_REGISTRY=$(grep -o '"registry": "[^"]*' config.json | cut -d'"' -f4)
IMAGE_URI="${ECR_REGISTRY}/lambda-workshop/product-api:${IMAGE_TYPE}-latest"
ACCOUNT_ID=$(aws sts get-caller-identity --query Account --output text)
ROLE_ARN="arn:aws:iam::${ACCOUNT_ID}:role/lambda-container-workshop-role"

echo "Image URI: $IMAGE_URI"
echo "Role ARN: $ROLE_ARN"

# Check if function exists
if AWS Lambda get-function --function-name $FUNCTION_NAME &>/dev/null; then
    echo "Function exists, updating..."

    # Update function code
    AWS Lambda update-function-code \
        --function-name $FUNCTION_NAME \
        --image-uri $IMAGE_URI

    # Update function configuration
    AWS Lambda update-function-configuration \
        --function-name $FUNCTION_NAME \
        --memory-size $MEMORY_SIZE \
        --timeout 30 \
        --environment Variables='{
            "ENVIRONMENT":"workshop",
            "LOG_LEVEL":"INFO"
        }'

    echo "✅ Function updated: $FUNCTION_NAME"
else
    echo "Creating new function..."

    # Create function
    AWS Lambda create-function \
        --function-name $FUNCTION_NAME \
        --role $ROLE_ARN \
        --code ImageUri=$IMAGE_URI \
        --package-type Image \
        --memory-size $MEMORY_SIZE \
        --timeout 30 \
        --Kiến trúc x86_64 \
        --environment Variables='{
            "ENVIRONMENT":"workshop",
            "LOG_LEVEL":"INFO"
        }' \
        --description "Workshop Lambda function using container image ($IMAGE_TYPE)"

    echo "✅ Function created: $FUNCTION_NAME"
fi

# Wait for function to be active
echo "Waiting for function to be active..."
AWS Lambda wait function-active --function-name $FUNCTION_NAME

# Get function info
AWS Lambda get-function \
    --function-name $FUNCTION_NAME \
    --query '{
        FunctionName: Configuration.FunctionName,
        Runtime: Configuration.PackageType,
        MemorySize: Configuration.MemorySize,
        Timeout: Configuration.Timeout,
        CodeSize: Configuration.CodeSize,
        LastModified: Configuration.LastModified
    }' \
    --output table

echo "Function ready for Kiểm thử!"
EOF

chmod +x scripts/create-lambda.sh

cat > scripts/create-api-gateway.sh << 'EOF'
#!/bin/bash

set -e

FUNCTION_NAME=${1:-"workshop-product-api-optimized"}
API_NAME="workshop-product-api"

echo "=== Creating API Gateway ==="

# Create REST API
API_ID=$(aws apigateway create-rest-api \
    --name $API_NAME \
    --description "Workshop Product API" \
    --query 'id' \
    --output text)

echo "API ID: $API_ID"

# Get root resource ID
ROOT_RESOURCE_ID=$(aws apigateway get-resources \
    --rest-api-id $API_ID \
    --query 'items[0].id' \
    --output text)

# Create {proxy+} resource
PROXY_RESOURCE_ID=$(aws apigateway create-resource \
    --rest-api-id $API_ID \
    --parent-id $ROOT_RESOURCE_ID \
    --path-part '{proxy+}' \
    --query 'id' \
    --output text)

# Create ANY method
aws apigateway put-method \
    --rest-api-id $API_ID \
    --resource-id $PROXY_RESOURCE_ID \
    --http-method ANY \
    --authorization-type NONE

# Get Lambda function ARN
ACCOUNT_ID=$(aws sts get-caller-identity --query Account --output text)
LAMBDA_ARN="arn:aws:lambda:us-east-1:${ACCOUNT_ID}:function:${FUNCTION_NAME}"

# Set up integration
aws apigateway put-integration \
    --rest-api-id $API_ID \
    --resource-id $PROXY_RESOURCE_ID \
    --http-method ANY \
    --type AWS_PROXY \
    --integration-http-method POST \
    --uri "arn:aws:apigateway:us-east-1:lambda:path/2015-03-31/functions/${LAMBDA_ARN}/invocations"

# Add Lambda permission
AWS Lambda add-permission \
    --function-name $FUNCTION_NAME \
    --statement-id api-gateway-invoke \
    --action lambda:InvokeFunction \
    --principal apigateway.amazonaws.com \
    --source-arn "arn:aws:execute-api:us-east-1:${ACCOUNT_ID}:${API_ID}/*/*"

# Deploy API
aws apigateway create-Triển khai \
    --rest-api-id $API_ID \
    --stage-name prod

# Get API endpoint
API_ENDPOINT="https://${API_ID}.execute-api.us-east-1.amazonaws.com/prod"
echo "✅ API Gateway created!"
echo "API Endpoint: $API_ENDPOINT"

# Save endpoint to config
echo "API_ENDPOINT=$API_ENDPOINT" >> ~/.bashrc
EOF

chmod +x scripts/create-api-gateway.sh

#!/bin/bash

FUNCTION_NAME=${1:-"workshop-product-api-optimized"}

echo "=== Kiểm thử Lambda Function: $FUNCTION_NAME ==="

# Test 1: Health Check
echo "1. Kiểm thử Health Check..."
AWS Lambda invoke \
    --function-name $FUNCTION_NAME \
    --payload '{"httpMethod":"GET","path":"/health"}' \
    response.json

echo "Response:"
cat response.json | jq .
echo

# Test 2: List Products
echo "2. Kiểm thử List Products..."
AWS Lambda invoke \
    --function-name $FUNCTION_NAME \
    --payload '{"httpMethod":"GET","path":"/products","queryStringParameters":{"limit":"2"}}' \
    response.json

echo "Response:"
cat response.json | jq .
echo

# Test 3: Get Single Product
echo "3. Kiểm thử Get Product..."
AWS Lambda invoke \
    --function-name $FUNCTION_NAME \
    --payload '{"httpMethod":"GET","path":"/products/1","pathParameters":{"id":"1"}}' \
    response.json

echo "Response:"
cat response.json | jq .
echo

# Test 4: Create Product
echo "4. Kiểm thử Create Product..."
AWS Lambda invoke \
    --function-name $FUNCTION_NAME \
    --payload '{
        "httpMethod":"POST",
        "path":"/products",
        "body":"{\"name\":\"Test Product\",\"price\":99.99,\"category\":\"Test\"}"
    }' \
    response.json

echo "Response:"
cat response.json | jq .
echo

# Dọn dẹp
rm -f response.json

echo "✅ Lambda function tests completed!"
EOF

#!/bin/bash

echo "=== Hiệu suất Comparison ==="

OPTIMIZED_FUNCTION="workshop-product-api-optimized"
BASE_FUNCTION="workshop-product-api-base"

echo "Kiểm thử both functions with same payload..."

# Test payload
PAYLOAD='{"httpMethod":"GET","path":"/products","queryStringParameters":{"limit":"5"}}'

echo "1. Kiểm thử Optimized Function..."
time AWS Lambda invoke \
    --function-name $OPTIMIZED_FUNCTION \
    --payload "$PAYLOAD" \
    optimized-response.json

echo "2. Kiểm thử Base Function..."
time AWS Lambda invoke \
    --function-name $BASE_FUNCTION \
    --payload "$PAYLOAD" \
    base-response.json

echo "=== Results ==="
echo "Optimized Function Response:"
cat optimized-response.json | jq .

echo "Base Function Response:"
cat base-response.json | jq .

# Dọn dẹp
rm -f optimized-response.json base-response.json

echo "✅ Hiệu suất comparison completed!"
EOF
