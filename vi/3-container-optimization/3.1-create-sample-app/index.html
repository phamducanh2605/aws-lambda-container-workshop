import json
import os
import time
import boto3
from datetime import datetime
from src.api.product_service import ProductService
from src.monitoring.performance_metrics import PerformanceMetrics

# Khởi tạo connections bên ngoài handler để tái sử dụng
product_service = ProductService()
metrics = PerformanceMetrics()

def lambda_handler(event, context):
    """
    Main Lambda handler cho Product API
    """
    start_time = time.time()

    try:
        # Log request info
        print(f"Request: {json.dumps(event)}")

        # Parse request
        http_method = event.get('httpMethod', 'GET')
        path = event.get('path', '/')
        body = json.loads(event.get('body', '{}')) if event.get('body') else {}
        path_parameters = event.get('pathParameters') or {}
        query_parameters = event.get('queryStringParameters') or {}

        # Route requests
        if path == '/health':
            response = handle_health_check()
        elif path.startswith('/products'):
            response = handle_products(http_method, path, body, path_parameters, query_parameters)
        else:
            response = {
                'statusCode': 404,
                'body': json.dumps({'error': 'Not Found'})
            }

        # Add CORS headers
        response['headers'] = {
            'Content-Type': 'application/json',
            'Access-Control-Allow-Origin': '*',
            'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE, OPTIONS',
            'Access-Control-Allow-Headers': 'Content-Type, Authorization'
        }

        # Record metrics
        duration = (time.time() - start_time) * 1000
        metrics.record_request(http_method, path, response['statusCode'], duration)

        return response

    except Exception as e:
        print(f"Error: {str(e)}")
        return {
            'statusCode': 500,
            'headers': {'Content-Type': 'application/json'},
            'body': json.dumps({'error': 'Internal Server Error'})
        }

def handle_health_check():
    """Health check endpoint"""
    return {
        'statusCode': 200,
        'body': json.dumps({
            'status': 'healthy',
            'timestamp': datetime.utcnow().isoformat(),
            'version': '1.0.0'
        })
    }

def handle_products(method, path, body, path_params, query_params):
    """Handle product-related requests"""

    if method == 'GET':
        if path == '/products':
            # List products
            limit = int(query_params.get('limit', 10))
            offset = int(query_params.get('offset', 0))
            products = product_service.list_products(limit, offset)
            return {
                'statusCode': 200,
                'body': json.dumps(products)
            }
        elif path_params.get('id'):
            # Get single product
            product_id = path_params['id']
            product = product_service.get_product(product_id)
            if product:
                return {
                    'statusCode': 200,
                    'body': json.dumps(product)
                }
            else:
                return {
                    'statusCode': 404,
                    'body': json.dumps({'error': 'Product not found'})
                }

    elif method == 'POST' and path == '/products':
        # Create product
        product = product_service.create_product(body)
        return {
            'statusCode': 201,
            'body': json.dumps(product)
        }

    elif method == 'PUT' and path_params.get('id'):
        # Update product
        product_id = path_params['id']
        product = product_service.update_product(product_id, body)
        if product:
            return {
                'statusCode': 200,
                'body': json.dumps(product)
            }
        else:
            return {
                'statusCode': 404,
                'body': json.dumps({'error': 'Product not found'})
            }

    elif method == 'DELETE' and path_params.get('id'):
        # Delete product
        product_id = path_params['id']
        success = product_service.delete_product(product_id)
        if success:
            return {
                'statusCode': 204,
                'body': ''
            }
        else:
            return {
                'statusCode': 404,
                'body': json.dumps({'error': 'Product not found'})
            }

    return {
        'statusCode': 405,
        'body': json.dumps({'error': 'Method not allowed'})
    }
EOF

import boto3
import json
import uuid
from datetime import datetime
from typing import List, Dict, Optional

class ProductService:
    def __init__(self):
        # Sử dụng DynamoDB local hoặc mock data cho demo
        self.products = [
            {
                'id': '1',
                'name': 'Laptop Dell XPS 13',
                'price': 1299.99,
                'category': 'Electronics',
                'description': 'High-performance ultrabook',
                'created_at': '2025-01-01T00:00:00Z'
            },
            {
                'id': '2',
                'name': 'iPhone 15 Pro',
                'price': 999.99,
                'category': 'Electronics',
                'description': 'Latest iPhone model',
                'created_at': '2025-01-01T00:00:00Z'
            },
            {
                'id': '3',
                'name': 'Nike Air Max',
                'price': 129.99,
                'category': 'Shoes',
                'description': 'Comfortable running shoes',
                'created_at': '2025-01-01T00:00:00Z'
            }
        ]

    def list_products(self, limit: int = 10, offset: int = 0) -> Dict:
        """List products with pagination"""
        total = len(self.products)
        products = self.products[offset:offset + limit]

        return {
            'products': products,
            'total': total,
            'limit': limit,
            'offset': offset,
            'has_more': offset + limit < total
        }

    def get_product(self, product_id: str) -> Optional[Dict]:
        """Get single product by ID"""
        for product in self.products:
            if product['id'] == product_id:
                return product
        return None

    def create_product(self, product_data: Dict) -> Dict:
        """Create new product"""
        product = {
            'id': str(uuid.uuid4()),
            'name': product_data.get('name'),
            'price': product_data.get('price'),
            'category': product_data.get('category'),
            'description': product_data.get('description', ''),
            'created_at': datetime.utcnow().isoformat() + 'Z'
        }

        self.products.append(product)
        return product

    def update_product(self, product_id: str, product_data: Dict) -> Optional[Dict]:
        """Update existing product"""
        for i, product in enumerate(self.products):
            if product['id'] == product_id:
                # Update fields
                product.update({
                    'name': product_data.get('name', product['name']),
                    'price': product_data.get('price', product['price']),
                    'category': product_data.get('category', product['category']),
                    'description': product_data.get('description', product['description']),
                    'updated_at': datetime.utcnow().isoformat() + 'Z'
                })
                self.products[i] = product
                return product
        return None

    def delete_product(self, product_id: str) -> bool:
        """Delete product"""
        for i, product in enumerate(self.products):
            if product['id'] == product_id:
                del self.products[i]
                return True
        return False

import boto3
import os
import time
from datetime import datetime

class PerformanceMetrics:
    def __init__(self):
        self.cloudwatch = boto3.client('cloudwatch')
        self.namespace = 'Lambda/ContainerPerformance'
        self.function_name = os.environ.get('AWS_LAMBDA_FUNCTION_NAME', 'workshop-product-api')

    def record_request(self, method: str, path: str, status_code: int, duration_ms: float):
        """Record request metrics"""
        try:
            # Record response time
            self.put_metric('ResponseTime', duration_ms, 'Milliseconds', [
                {'Name': 'Method', 'Value': method},
                {'Name': 'Path', 'Value': path}
            ])

            # Record request count
            self.put_metric('RequestCount', 1, 'Count', [
                {'Name': 'Method', 'Value': method},
                {'Name': 'StatusCode', 'Value': str(status_code)}
            ])

            # Record errors
            if status_code >= 400:
                self.put_metric('ErrorCount', 1, 'Count', [
                    {'Name': 'StatusCode', 'Value': str(status_code)}
                ])

        except Exception as e:
            print(f"Failed to record metrics: {e}")

    def put_metric(self, metric_name: str, value: float, unit: str, dimensions: list):
        """Send metric to CloudWatch"""
        dimensions.append({
            'Name': 'FunctionName',
            'Value': self.function_name
        })

        self.cloudwatch.put_metric_data(
            Namespace=self.namespace,
            MetricData=[
                {
                    'MetricName': metric_name,
                    'Value': value,
                    'Unit': unit,
                    'Dimensions': dimensions,
                    'Timestamp': datetime.utcnow()
                }
            ]
        )
EOF
