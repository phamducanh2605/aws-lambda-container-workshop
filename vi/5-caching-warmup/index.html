# src/Bộ nhớ đệm (Caching)/app_cache.py
import time
import json
from functools import wraps
from typing import Any, Optional, Dict

class InMemoryCache:
    """Thread-safe in-memory cache với hỗ trợ TTL"""

    def __init__(self, default_ttl: int = 300):
        self._cache: Dict[str, Dict] = {}
        self.default_ttl = default_ttl

    def get(self, key: str) -> Optional[Any]:
        """Lấy giá trị từ cache"""
        if key not in self._cache:
            return None

        entry = self._cache[key]
        if time.time() > entry['expires']:
            del self._cache[key]
            return None

        entry['hits'] += 1
        entry['last_accessed'] = time.time()
        return entry['value']

    def set(self, key: str, value: Any, ttl: Optional[int] = None) -> None:
        """Đặt giá trị vào cache"""
        ttl = ttl or self.default_ttl
        self._cache[key] = {
            'value': value,
            'expires': time.time() + ttl,
            'created': time.time(),
            'last_accessed': time.time(),
            'hits': 0
        }

    def delete(self, key: str) -> bool:
        """Xóa key khỏi cache"""
        if key in self._cache:
            del self._cache[key]
            return True
        return False

    def clear(self) -> None:
        """Xóa tất cả cache entries"""
        self._cache.clear()

    def stats(self) -> Dict:
        """Lấy thống kê cache"""
        total_entries = len(self._cache)
        total_hits = sum(entry['hits'] for entry in self._cache.values())

        return {
            'total_entries': total_entries,
            'total_hits': total_hits,
            'memory_usage_mb': self._estimate_memory_usage()
        }

    def _estimate_memory_usage(self) -> float:
        """Ước tính việc sử dụng bộ nhớ tính bằng MB"""
        import sys
        total_size = 0
        for entry in self._cache.values():
            total_size += sys.getsizeof(entry['value'])
        return total_size / 1024 / 1024

# Global cache instance
app_cache = InMemoryCache(default_ttl=300)

def cached(ttl: int = 300, key_prefix: str = ""):
    """Decorator để cache kết quả function"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # Tạo cache key
            cache_key = f"{key_prefix}{func.__name__}:{hash(str(args) + str(kwargs))}"

            # Thử lấy từ cache
            cached_result = app_cache.get(cache_key)
            if cached_result is not None:
                return cached_result

            # Thực thi function và cache kết quả
            result = func(*args, **kwargs)
            app_cache.set(cache_key, result, ttl)
            return result

        return wrapper
    return decorator

# src/Bộ nhớ đệm (Caching)/redis_cache.py
import redis
import json
import pickle
import time
from typing import Any, Optional
import os

class RedisCache:
    """Redis-based distributed cache"""

    def __init__(self):
        self.redis_client = redis.Redis(
            host=os.environ.get('REDIS_HOST', 'localhost'),
            port=int(os.environ.get('REDIS_PORT', 6379)),
            db=int(os.environ.get('REDIS_DB', 0)),
            decode_responses=False,  # Xử lý binary data
            socket_connect_timeout=5,
            socket_timeout=5,
            retry_on_timeout=True,
            health_check_interval=30
        )

    def get(self, key: str) -> Optional[Any]:
        """Lấy giá trị từ Redis cache"""
        try:
            value = self.redis_client.get(key)
            if value is None:
                return None
            return pickle.loads(value)
        except Exception as e:
            print(f"Redis get error cho key {key}: {e}")
            return None

    def set(self, key: str, value: Any, ttl: int = 300) -> bool:
        """Đặt giá trị vào Redis cache"""
        try:
            serialized_value = pickle.dumps(value)
            return self.redis_client.setex(key, ttl, serialized_value)
        except Exception as e:
            print(f"Redis set error cho key {key}: {e}")
            return False

    def delete(self, key: str) -> bool:
        """Xóa key khỏi Redis cache"""
        try:
            return bool(self.redis_client.delete(key))
        except Exception as e:
            print(f"Redis delete error cho key {key}: {e}")
            return False

    def exists(self, key: str) -> bool:
        """Kiểm tra xem key có tồn tại trong Redis không"""
        try:
            return bool(self.redis_client.exists(key))
        except Exception as e:
            print(f"Redis exists error cho key {key}: {e}")
            return False

    def increment(self, key: str, amount: int = 1) -> Optional[int]:
        """Tăng counter trong Redis"""
        try:
            return self.redis_client.incrby(key, amount)
        except Exception as e:
            print(f"Redis increment error cho key {key}: {e}")
            return None

    def get_stats(self) -> Dict:
        """Lấy thống kê Redis cache"""
        try:
            info = self.redis_client.info()
            return {
                'connected_clients': info.get('connected_clients', 0),
                'used_memory_mb': info.get('used_memory', 0) / 1024 / 1024,
                'keyspace_hits': info.get('keyspace_hits', 0),
                'keyspace_misses': info.get('keyspace_misses', 0),
                'hit_rate': self._calculate_hit_rate(info)
            }
        except Exception as e:
            print(f"Redis stats error: {e}")
            return {}

    def _calculate_hit_rate(self, info: Dict) -> float:
        """Tính toán tỷ lệ cache hit"""
        hits = info.get('keyspace_hits', 0)
        misses = info.get('keyspace_misses', 0)
        total = hits + misses
        return (hits / total * 100) if total > 0 else 0

# Global Redis cache instance
redis_cache = RedisCache()
