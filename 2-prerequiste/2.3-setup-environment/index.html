# Create ECR repository
aws ecr create-repository \
    --repository-name lambda-workshop/product-api \
    --region us-east-1

# Get repository URI
export ECR_REGISTRY=$(aws ecr describe-repositories \
    --repository-names lambda-workshop/product-api \
    --query 'repositories[0].repositoryUri' \
    --output text | cut -d'/' -f1)

echo "ECR Registry: $ECR_REGISTRY"

# Save to file for later use
echo "export ECR_REGISTRY=$ECR_REGISTRY" >> ~/.bashrc

cat > scripts/check-environment.sh << 'EOF'
#!/bin/bash

echo "=== Checking Development Environment ==="

# Check Docker
echo -n "Docker: "
if command -v docker &> /dev/null && docker info &> /dev/null; then
    echo "✅ $(docker --version)"
else
    echo "❌ Not installed or not running"
fi

# Check AWS CLI
echo -n "AWS CLI: "
if command -v aws &> /dev/null; then
    echo "✅ $(aws --version)"
else
    echo "❌ Not installed"
fi

# Check Python
echo -n "Python 3.11: "
if command -v python3.11 &> /dev/null; then
    echo "✅ $(python3.11 --version)"
else
    echo "❌ Not installed"
fi

# Check AWS credentials
echo -n "AWS Credentials: "
if aws sts get-caller-identity &> /dev/null; then
    echo "✅ Account: $(aws sts get-caller-identity --query 'Account' --output text)"
else
    echo "❌ Not configured"
fi

# Check ECR access
echo -n "ECR Access: "
if aws ecr describe-repositories --repository-names lambda-workshop/product-api &> /dev/null; then
    echo "✅ Repository created"
else
    echo "❌ Repository not created"
fi

echo "=== Environment Check Complete ==="
EOF

chmod +x scripts/check-environment.sh

2. Run verification

1
